73template <int dim, int degree>
74inline void
75identitySolver<dim, degree>::init()
76{
77 this->system_matrix->clear();
78 this->system_matrix->initialize(this->matrix_free_handler.get_matrix_free());
79 this->update_system_matrix->clear();
80 this->update_system_matrix->initialize(this->matrix_free_handler.get_matrix_free());
81
82 this->system_matrix->add_global_to_local_mapping(
83 this->residual_global_to_local_solution);
84 this->system_matrix->add_src_solution_subset(this->residual_src);
85
86 this->update_system_matrix->add_global_to_local_mapping(
87 this->newton_update_global_to_local_solution);
88 this->update_system_matrix->add_src_solution_subset(this->newton_update_src);
89
90 // Apply constraints
91 this->constraint_handler.get_constraint(this->field_index)
92 .distribute(*(this->solution_handler.solution_set.at(
93 std::make_pair(this->field_index, dependencyType::NORMAL))));
94}

101template <int dim, int degree>
102inline void
103identitySolver<dim, degree>::solve(const double step_length)
104{
105 auto *solution = this->solution_handler.solution_set.at(
106 std::make_pair(this->field_index, dependencyType::NORMAL));
107
108 // Compute the residual
109 this->system_matrix->compute_residual(*this->residual, *solution);
110 prisms::conditionalOStreams::pout_summary()
111 << " field: " << this->field_index
112 << " Initial residual: " << this->residual->l2_norm() << std::flush;
113
114 // Determine the residual tolerance
115 this->compute_solver_tolerance();
116
117 // Update solver controls
118 this->solver_control.set_tolerance(this->tolerance);
119 dealii::SolverCG<dealii::LinearAlgebra::distributed::Vector<double>> cg(
120 this->solver_control);
121
122 try
123 {
124 *this->newton_update = 0.0;
125 cg.solve(*(this->update_system_matrix),
126 *this->newton_update,
127 *this->residual,
128 dealii::PreconditionIdentity());
129 }
130 catch (...)
131 {
132 prisms::conditionalOStreams::pout_base()
133 << "Warning: linear solver did not converge as per set tolerances.\n";
134 }
135 this->constraint_handler.get_constraint(this->field_index)
136 .set_zero(*this->newton_update);
137
138 prisms::conditionalOStreams::pout_summary()
139 << " Final residual: " << this->solver_control.last_value()
140 << " Steps: " << this->solver_control.last_step() << "\n"
141 << std::flush;
142
143 // Update the solutions
144 (*solution).add(step_length, *this->newton_update);
145 this->solution_handler.update(fieldSolveType::NONEXPLICIT_LINEAR, this->field_index);
146
147 // Apply constraints
148 // This may be redundant with the constraints on the update step.
149 this->constraint_handler.get_constraint(this->field_index).distribute(*solution);
150}
