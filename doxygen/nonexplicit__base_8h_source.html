182template <int dim, int degree>
183inline void
184nonexplicitBase<dim, degree>::compute_subset_attributes(
185 const fieldSolveType &field_solve_type)
186{
187 Assert((field_solve_type == fieldSolveType::NONEXPLICIT_LINEAR ||
188 field_solve_type == fieldSolveType::NONEXPLICIT_SELF_NONLINEAR ||
189 field_solve_type == fieldSolveType::NONEXPLICIT_AUXILIARY ||
190 field_solve_type == fieldSolveType::NONEXPLICIT_CO_NONLINEAR),
191 dealii::ExcMessage(
192 "compute_subset_attributes() should only be used for "
193 "NONEXPLICIT_LINEAR, NONEXPLICIT_SELF_NONLINEAR, NONEXPLICIT_AUXILIARY, and "
194 "NONEXPLICIT_CO_NONLINEAR fieldSolveTypes"));
195
196 subset_attributes.clear();
197
198 for (const auto &[index, variable] : user_inputs.var_attributes)
199 {
200 if (variable.field_solve_type == field_solve_type)
201 {
202 subset_attributes.emplace(index, variable);
203 }
204 }
205}

207template <int dim, int degree>
208inline void
209nonexplicitBase<dim, degree>::compute_shared_dependencies()
210{
211 Assert(subset_attributes.begin()->second.field_solve_type ==
212 fieldSolveType::NONEXPLICIT_CO_NONLINEAR,
213 dealii::ExcMessage("compute_shared_dependencies() should only be used for "
214 "NONEXPLICIT_CO_NONLINEAR fieldSolveTypes"));
215
216 // Compute the shared dependency flags
217 auto &dependency_flag_set = subset_attributes.begin()->second.eval_flag_set_RHS;
218 for (const auto &[index, variable] : subset_attributes)
219 {
220 if (!variable.eval_flag_set_RHS.empty())
221 {
222 for (const auto &[pair, flag] : variable.eval_flag_set_RHS)
223 {
224 dependency_flag_set[pair] |= flag;
225 }
226 }
227 }
228 for (auto &[index, variable] : subset_attributes)
229 {
230 for (const auto &[pair, flag] : dependency_flag_set)
231 {
232 variable.eval_flag_set_RHS[pair] |= flag;
233 }
234 }
235
236 // Compute the shared dependency set
237 auto &dependency_set = subset_attributes.begin()->second.dependency_set_RHS;
238 for (const auto &[index, variable] : subset_attributes)
239 {
240 for (const auto &[index, map] : variable.dependency_set_RHS)
241 {
242 for (const auto &[dependency_type, field_type] : map)
243 {
244 dependency_set[index].emplace(dependency_type, field_type);
245 }
246 }
247 }
248 for (auto &[index, variable] : subset_attributes)
249 {
250 variable.dependency_set_RHS = dependency_set;
251 }
252
253#ifdef DEBUG
254 print();
255#endif
256}

258template <int dim, int degree>
259inline void
260nonexplicitBase<dim, degree>::set_initial_condition()
261{
262 for (const auto &[index, variable] : subset_attributes)
263 {
264 if (variable.pde_type != PDEType::IMPLICIT_TIME_DEPENDENT &&
265 variable.pde_type != PDEType::TIME_INDEPENDENT)
266 {
267 continue;
268 }
269
270 Assert(dof_handler.const_dof_handlers.size() > index,
271 dealii::ExcMessage(
272 "The const DoFHandler set is smaller than the given index = " +
273 std::to_string(index)));
274 Assert(subset_attributes.find(index) != subset_attributes.end(),
275 dealii::ExcMessage(
276 "There is no entry in the attribute subset for the given index = " +
277 std::to_string(index)));
278 Assert(solution_handler.solution_set.find(
279 std::make_pair(index, dependencyType::NORMAL)) !=
280 solution_handler.solution_set.end(),
281 dealii::ExcMessage("There is no solution vector for the given index = " +
282 std::to_string(index) +
283 " and type = " + to_string(dependencyType::NORMAL)));
284
285 dealii::VectorTools::interpolate(
286 mapping,
287 *(dof_handler.const_dof_handlers.at(index)),
288 initialCondition<dim>(index, subset_attributes.at(index).field_type),
289 *(solution_handler.solution_set.at(
290 std::make_pair(index, dependencyType::NORMAL))));
291
292 // TODO: Fix so that we apply some sort of initial condition to all old vector for
293 // all types.
294 if (solution_handler.solution_set.find(
295 std::make_pair(index, dependencyType::OLD_1)) !=
296 solution_handler.solution_set.end())
297 {
298 *(solution_handler.solution_set.at(
299 std::make_pair(index, dependencyType::OLD_1))) =
300 *(solution_handler.solution_set.at(
301 std::make_pair(index, dependencyType::NORMAL)));
302 }
303 }
