83template <int dim, int degree>
84GMGSolver<dim, degree>::GMGSolver(const userInputParameters<dim> &_user_inputs,
85 const uint &_field_index)
86 : baseSolver<dim, degree>(_user_inputs, _field_index)
87 , mg_matrices(dealii::MGLevelObject<LevelMatrixType>(0, 0, _user_inputs))
88{
89 // Get the dirichlet boundary ids for multigrid constriants
90 for (const auto &[component, boundary_condition] :
91 this->user_inputs.boundary_parameters.boundary_condition_list.at(
92 this->field_index))
93 {
94 Assert(component == 0, FeatureNotImplemented("Vector multigrid"));
95
96 for (const auto &[boundary_id, boundary_type] :
97 boundary_condition.boundary_condition_map)
98 {
99 if (boundary_type == boundaryType::DIRICHLET ||
100 boundary_type == boundaryType::NON_UNIFORM_DIRICHLET)
101 {
102 dirichlet_boundary_ids.insert(boundary_id);
103 }
104 }
105 }

108template <int dim, int degree>
109inline void
110GMGSolver<dim, degree>::init_system(
111 const triangulationHandler<dim> &triangulation_handler,
112 dealii::LinearAlgebra::distributed::Vector<double> &solution,
113 dealii::DoFHandler<dim> &dof_handler)
114{
115 this->system_matrix.clear();
116 mg_matrices.clear_elements();
117
118 dof_handler.reinit(triangulation_handler.get_triangulation());
119 dof_handler.distribute_dofs(this->fe);
120 dof_handler.distribute_mg_dofs();
121
122 this->constraint_handler.make_constraints(this->mapping, dof_handler);
123
124 typename dealii::MatrixFree<dim, double>::AdditionalData additional_data;
125 additional_data.tasks_parallel_scheme =
126 dealii::MatrixFree<dim, double>::AdditionalData::none;
127
128 // This should be done according to the residual flags to prevent excess data being
129 // evaluated for the shape functions.
130 additional_data.mapping_update_flags =
131 (update_values | update_gradients | update_JxW_values | update_quadrature_points);
132 std::shared_ptr<dealii::MatrixFree<dim, double>> system_mf_storage(
133 new dealii::MatrixFree<dim, double>());
134 system_mf_storage->reinit(this->mapping,
135 dof_handler,
136 this->constraint_handler.get_constraints(),
137 dealii::QGauss<1>(this->fe.degree + 1),
138 additional_data);
139 this->system_matrix.initialize(system_mf_storage);
140
141 this->system_matrix.initialize_dof_vector(this->change_in_solution);
142 this->system_matrix.initialize_dof_vector(solution);
143 this->system_matrix.initialize_dof_vector(this->residual);
144
145 // Apply boundary conditions to the solution vector as an initial guess
146 this->constraint_handler.get_constraints().distribute(solution);
147
148 const uint nlevels = triangulation_handler.get_n_global_levels();
149 mg_matrices.resize(0, nlevels - 1, this->user_inputs);
150
151 mg_constrained_dofs.initialize(dof_handler);
152 mg_constrained_dofs.make_zero_boundary_constraints(dof_handler, dirichlet_boundary_ids);
153
154 for (uint level = 0; level < nlevels; ++level)
155 {
156 // Setup the constraint set for the multigrid level
157 dealii::AffineConstraints<double> level_constraints(
158 dof_handler.locally_owned_mg_dofs(level),
159 dealii::DoFTools::extract_locally_relevant_level_dofs(dof_handler, level));
160
161 // Loop through boundary indices and apply constraints
162 for (const dealii::types::global_dof_index dof_index :
163 mg_constrained_dofs.get_boundary_indices(level))
164 {
165 level_constraints.constrain_dof_to_zero(dof_index);
166 }
167 level_constraints.close();
168
169 typename dealii::MatrixFree<dim, float>::AdditionalData additional_data;
170 additional_data.tasks_parallel_scheme =
171 dealii::MatrixFree<dim, float>::AdditionalData::none;
172
173 // This should be done according to the residual flags to prevent excess data being
174 // evaluated for the shape functions.
175 additional_data.mapping_update_flags =
176 (update_values | update_gradients | update_JxW_values | update_quadrature_points);
177 additional_data.mg_level = level;
178 std::shared_ptr<dealii::MatrixFree<dim, float>> mg_mf_storage_level =
179 std::make_shared<dealii::MatrixFree<dim, float>>();
180 mg_mf_storage_level->reinit(this->mapping,
181 dof_handler,
182 level_constraints,
183 dealii::QGauss<1>(this->fe.degree + 1),
184 additional_data);
185
186 mg_matrices[level].initialize(mg_mf_storage_level, mg_constrained_dofs, level);
187 }
188
189 conditionalOStreams::pout_base
190 << "Number of degrees of freedom: " << dof_handler.n_dofs() << std::endl;
191 conditionalOStreams::pout_summary()
192 << "Number of degrees of freedom: " << dof_handler.n_dofs() << std::endl;
193}

200template <int dim, int degree>
201inline void
202GMGSolver<dim, degree>::solve(
203 const triangulationHandler<dim> &triangulation_handler,
204 dealii::LinearAlgebra::distributed::Vector<double> &solution,
205 const dealii::DoFHandler<dim> &dof_handler)
206{
207 dealii::MGTransferMatrixFree<dim, float> mg_transfer(mg_constrained_dofs);
208 mg_transfer.build(dof_handler);
209
210 using SmootherType =
211 dealii::PreconditionChebyshev<LevelMatrixType,
212 dealii::LinearAlgebra::distributed::Vector<float>>;
213 dealii::mg::SmootherRelaxation<SmootherType,
214 dealii::LinearAlgebra::distributed::Vector<float>>
215 mg_smoother;
216 dealii::MGLevelObject<typename SmootherType::AdditionalData> smoother_data;
217 smoother_data.resize(0, triangulation_handler.get_n_global_levels() - 1);
218 for (uint level = 0; level < triangulation_handler.get_n_global_levels(); ++level)
219 {
220 if (level > 0)
221 {
222 smoother_data[level].smoothing_range =
223 this->user_inputs.linear_solve_parameters.linear_solve.at(this->field_index)
224 .smoothing_range;
225 smoother_data[level].degree =
226 this->user_inputs.linear_solve_parameters.linear_solve.at(this->field_index)
227 .smoother_iterations;
228 smoother_data[level].eig_cg_n_iterations =
229 this->user_inputs.linear_solve_parameters.linear_solve.at(this->field_index)
230 .eig_cg_n_iterations;
231 }
232 else
233 {
234 smoother_data[0].smoothing_range = 1e-3;
235 smoother_data[0].degree = dealii::numbers::invalid_unsigned_int;
236 smoother_data[0].eig_cg_n_iterations = mg_matrices[0].m();
237 }
238 mg_matrices[level].compute_diagonal();
239 smoother_data[level].preconditioner =
240 mg_matrices[level].get_matrix_diagonal_inverse();
241 }
242 mg_smoother.initialize(mg_matrices, smoother_data);
243
244 dealii::MGCoarseGridApplySmoother<dealii::LinearAlgebra::distributed::Vector<float>>
245 mg_coarse;
246 mg_coarse.initialize(mg_smoother);
247
248 dealii::mg::Matrix<dealii::LinearAlgebra::distributed::Vector<float>> mg_matrix(
249 mg_matrices);
250
251 dealii::MGLevelObject<dealii::MatrixFreeOperators::MGInterfaceOperator<LevelMatrixType>>
252 mg_interface_matrices;
253 mg_interface_matrices.resize(0, triangulation_handler.get_n_global_levels() - 1);
254 for (uint level = 0; level < triangulation_handler.get_n_global_levels(); ++level)
255 {
256 mg_interface_matrices[level].initialize(mg_matrices[level]);
257 }
258 dealii::mg::Matrix<dealii::LinearAlgebra::distributed::Vector<float>> mg_interface(
259 mg_interface_matrices);
260
261 Multigrid<LinearAlgebra::distributed::Vector<float>> mg(mg_matrix,
262 mg_coarse,
263 mg_transfer,
264 mg_smoother,
265 mg_smoother);
266 mg.set_edge_matrices(mg_interface, mg_interface);
267
268 dealii::PreconditionMG<dim,
269 dealii::LinearAlgebra::distributed::Vector<float>,
270 dealii::MGTransferMatrixFree<dim, float>>
271 preconditioner(dof_handler, mg, mg_transfer);
272
273 // Compute the residual of the current solution (Ax-b)
274 this->system_matrix.compute_residual(this->residual, solution);
275
276 // Determine the residual tolerance
277 this->compute_solver_tolerance();
278
279 // Update solver controls
280 this->solver_control.set_tolerance(this->tolerance);
281 dealii::SolverCG<dealii::LinearAlgebra::distributed::Vector<double>> cg(
282 this->solver_control);
283
284 try
285 {
286 this->change_in_solution = 0.0;
287 cg.solve(this->system_matrix,
288 this->change_in_solution,
289 this->residual,
290 preconditioner);
291 this->constraint_handler.get_constraints().set_zero(this->change_in_solution);
292 }
293 catch (...)
294 {
295 conditionalOStreams::pout_base
296 << "Warning: linear solver did not converge as per set tolerances.\n";
297 }
298 solution += this->change_in_solution;
299 this->constraint_handler.get_constraints().distribute(solution);
300
301 conditionalOStreams::pout_summary()
302 << this->solver_control.last_step() << " iterations\n";
303}
