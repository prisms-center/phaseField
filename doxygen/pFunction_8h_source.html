1// pFunction: An interface for PFunctions in PRISMS-PF
2// This class is used to declare and load PFunctions from a pre-existing
3// PLibrary that was created using PRISMS IntegrationTools. The class is a
4// wrapper for the PFunctions in IntegrationTools so that they can be used for
5// vectorized arrays. This class also calls the IntegrationTools checkout
6// function in the constructor to reduce the number of steps the user needs to
7// take. Currently this is only implemented for scalar functions. Vector
8// functions can be treated component by component.
9
10#include <deal.II/base/vectorization.h>

12#include <field_input/IntegrationTools/extern/PLibraryExtern.hh>
13#include <string>

15namespace PFunctions

17 using scalarvalueType = dealii::VectorizedArray<double>;
18
19 class pFunction
20 {
21 public:
22 // Constructor, wraps the IntegrationTools checkout function

24
25 // Returns the value of the function for a given input variable
26 scalarvalueType val(scalarvalueType);

30 scalarvalueType

32
33 // Returns one of the second derivatives of the function for a given input
34 // variable

37
38 private:
39 PRISMS::PFunction<double *, double> fun;
40 };

48 pFunction::val(scalarvalueType var)
49 {
50 scalarvalueType fun_val;
