202template <int dim, int degree, typename number>
203void
204matrixFreeOperator<dim, degree, number>::compute_diagonal()
205{
206 this->inverse_diagonal_entries.reset(
207 new DiagonalMatrix<dealii::LinearAlgebra::distributed::Vector<number>>());
208 dealii::LinearAlgebra::distributed::Vector<number> &inverse_diagonal =
209 this->inverse_diagonal_entries->get_vector();
210 this->data->initialize_dof_vector(inverse_diagonal);
211 uint dummy = 0;
212 this->data->cell_loop(&matrixFreeOperator::local_compute_diagonal,
213 this,
214 inverse_diagonal,
215 dummy);
216
217 this->set_constrained_entries_to_one(inverse_diagonal);
218
219 for (uint i = 0; i < inverse_diagonal.locally_owned_size(); ++i)
220 {
221 Assert(inverse_diagonal.local_element(i) > 0.0,
222 dealii::ExcMessage(
223 "No diagonal entry in a positive definite operator should be zero"));
224 inverse_diagonal.local_element(i) = 1.0 / inverse_diagonal.local_element(i);
225 }
226}
