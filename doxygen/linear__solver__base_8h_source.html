166template <int dim, int degree>
167linearSolverBase<dim, degree>::linearSolverBase(
168 const userInputParameters<dim> &_user_inputs,
169 const variableAttributes &_variable_attributes,
170 const matrixfreeHandler<dim> &_matrix_free_handler,
171 const constraintHandler<dim> &_constraint_handler,
172 solutionHandler<dim> &_solution_handler)
173 : user_inputs(_user_inputs)
174 , variable_attributes(_variable_attributes)
175 , matrix_free_handler(_matrix_free_handler)
176 , constraint_handler(_constraint_handler)
177 , solution_handler(_solution_handler)
178 , field_index(_variable_attributes.field_index)
179 , residual(solution_handler.new_solution_set.at(field_index))
180 , newton_update(solution_handler.solution_set.at(
181 std::make_pair(field_index, dependencyType::CHANGE)))
182 , solver_control(
183 _user_inputs.linear_solve_parameters.linear_solve.at(field_index).max_iterations)
184{
185 // Creating map to match types
186 subset_attributes.emplace(field_index, variable_attributes);
187
188 // Create the implementation of customPDE with the subset of variable attributes
189 system_matrix =
190 std::make_unique<SystemMatrixType>(user_inputs, subset_attributes, field_index);
191 update_system_matrix =
192 std::make_unique<SystemMatrixType>(user_inputs, subset_attributes, field_index);
193
194 // Create the residual subset of solution vectors and add the mapping to customPDE
195 residual_src.push_back(solution_handler.solution_set.at(
196 std::make_pair(field_index, dependencyType::NORMAL)));
197 residual_global_to_local_solution.emplace(std::make_pair(field_index,
198 dependencyType::NORMAL),
199 0);
200 for (const auto &[variable_index, map] : variable_attributes.dependency_set_RHS)
201 {
202 for (const auto &[dependency_type, field_type] : map)
203 {
204 const auto pair = std::make_pair(variable_index, dependency_type);
205
206 Assert(solution_handler.solution_set.find(pair) !=
207 solution_handler.solution_set.end(),
208 dealii::ExcMessage("There is no solution vector for the given index = " +
209 std::to_string(variable_index) +
210 " and type = " + to_string(dependency_type)));
211
212 Assert(solution_handler.new_solution_set.find(variable_index) !=
213 solution_handler.new_solution_set.end(),
214 dealii::ExcMessage(
215 "There is no new solution vector for the given index = " +
216 std::to_string(variable_index)));
217
218 residual_src.push_back(solution_handler.solution_set.at(pair));
219 residual_global_to_local_solution.emplace(pair, residual_src.size() - 1);
220 }
221 }
222
223 // Create the newton update subset of solution vectors and add the mapping to
224 // customPDE. For this one we consider the singular src vector as the residual
225 // vector above and the src subset all other dependencies vectors. This is
226 // complicated, but has to do with the way deal.II's solvers (like CG) handle
227 // iterative updates. For this reason, we have to pass the residual vector as
228 // VectorType src and all other dependencies for the LHS as std::vector<VectorType*>
229 // src_subset.
230 newton_update_src.push_back(solution_handler.new_solution_set.at(field_index));
231 newton_update_global_to_local_solution.emplace(std::make_pair(field_index,
232 dependencyType::CHANGE),
233 0);
234 for (const auto &[variable_index, map] : variable_attributes.dependency_set_LHS)
235 {
236 for (const auto &[dependency_type, field_type] : map)
237 {
238 if (dependency_type == dependencyType::CHANGE)
239 {
240 continue;
241 }
242 const auto pair = std::make_pair(variable_index, dependency_type);
243
244 Assert(solution_handler.solution_set.find(pair) !=
245 solution_handler.solution_set.end(),
246 dealii::ExcMessage("There is no solution vector for the given index = " +
247 std::to_string(variable_index) +
248 " and type = " + to_string(dependency_type)));
249
250 Assert(solution_handler.new_solution_set.find(variable_index) !=
251 solution_handler.new_solution_set.end(),
252 dealii::ExcMessage(
253 "There is no new solution vector for the given index = " +
254 std::to_string(variable_index)));
255
256 newton_update_src.push_back(solution_handler.solution_set.at(pair));
257 newton_update_global_to_local_solution.emplace(pair,
258 newton_update_src.size() - 1);
259 }
260 }
261}

263template <int dim, int degree>
264inline void
265linearSolverBase<dim, degree>::compute_solver_tolerance()
266{
267 tolerance =
268 user_inputs.linear_solve_parameters.linear_solve.at(field_index).tolerance_type ==
269 solverToleranceType::RELATIVE_RESIDUAL_CHANGE
270 ? user_inputs.linear_solve_parameters.linear_solve.at(field_index).tolerance *
271 residual->l2_norm()
272 : user_inputs.linear_solve_parameters.linear_solve.at(field_index).tolerance;
273}
