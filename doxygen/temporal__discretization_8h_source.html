45inline void
46temporalDiscretization::postprocess_and_validate(
47 const std::map<unsigned int, variableAttributes> &var_attributes)
48{
49 // If all of the variables are `TIME_INDEPENDENT`, `AUXILIARY`, or `CONSTANT` then
50 // total_increments should be 1 and final_time should be 0
51 bool only_time_independent_pdes = true;
52 for (const auto &[index, variable] : var_attributes)
53 {
54 if (variable.pde_type == PDEType::EXPLICIT_TIME_DEPENDENT ||
55 variable.pde_type == PDEType::IMPLICIT_TIME_DEPENDENT)
56 {
57 only_time_independent_pdes = false;
58 break;
59 }
60 }
61 if (only_time_independent_pdes)
62 {
63 total_increments = 1;
64 return;
65 }
66
67 // Check that the timestep is greater than zero
68 AssertThrow(dt > 0.0,
69 dealii::ExcMessage(
70 "The timestep must be greater than zero for transient problems."));
71
72 // Pick the maximum specified time since the default values are zero
73 final_time = std::max(final_time, dt * total_increments);
74 total_increments = static_cast<unsigned int>(std::ceil(final_time / dt));
75}
