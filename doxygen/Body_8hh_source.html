33 void
34 read_vtk(const std::string &vtkfile)
35 {
36 // Read vtk file
37 std::ifstream infile(vtkfile.c_str());
38 if (!infile.is_open())
39 {
40 throw std::runtime_error("Could not open VTK file: " + vtkfile);
41 }
42
43 // Read mesh info
44 mesh.read_vtk(infile);
45
46 // Read point data
47 std::istringstream ss;
48 std::string str;
49 std::string name;
50 std::string type;
51 std::string line;
52 int numcomp = 0;
53 unsigned long int Npoints = 0;
54
55 while (std::getline(infile, line))
56 {
57 // Check for Point data
58 if (line.rfind("POINT_DATA", 0) == 0)
59 {
60 ss.clear();
61 ss.str(line);
62 ss >> str >> Npoints;
63 }
64
65 // Check for Scalar data
66 if (line.rfind("SCALARS", 0) == 0)
67 {
68 ss.clear();
69 ss.str(line);
70 ss >> str >> name >> type >> numcomp;
71
72 // Skip LOOKUP_TABLE line that follows SCALAR line
73 std::getline(infile, line);
74
75 // Read data
76 std::vector<double> gid(Npoints);
77 for (auto &value : gid)
78 {
79 infile >> value;
80 }
81
82 // Check dim
83 switch (DIM)
84 {
85 case 1:
86 throw std::runtime_error(
87 "1D dimensional vtk read-in is not currently supported.");
88 break;
89 case 2:
90 break;
91 case 3:
92 break;
93 default:
94 throw std::runtime_error("Invalid dimension for vtk read-in.");
95 }
96
97 // Construct field
98 scalar_field.emplace_back(name, mesh, gid, 0.0);
99 }
100 }
101 }

104 read_RL_vtk(const std::string &vtkfile)
105 {
106 std::cout << "Begin reading vtk file" << std::endl;
107
108 // read in vtk file here
109 std::ifstream infile_mesh(vtkfile.c_str());
110
111 // read mesh info
112 mesh.read_RL_vtk(infile_mesh);
113
114 std::ifstream infile(vtkfile.c_str());
115
116 // read point data
117 std::istringstream ss;
118 std::string str, name, type, line;
119 int numcomp;
120 unsigned long int N_points, Npoints_x, Npoints_y, Npoints_z, Npoints, u, p;
121
122 while (!infile.eof())
123 {
124 std::getline(infile, line);
125
126 if (line[0] == 'X')
127 {
128 if (line.size() > 12 && line.substr(0, 13) == "X_COORDINATES")
129 {
130 // read header line
131 // std::cout << line << "\n";
132 ss.clear();
133 ss.str(line);
134 ss >> str >> Npoints_x >> type;
135
136 std::cout << "Read X_COORDINATES: " << Npoints_x << std::endl;
137
138 // std::cout << " reserve OK" << std::endl;
139 }
140 }
141 if (line[0] == 'Y')
142 {
143 if (line.size() > 12 && line.substr(0, 13) == "Y_COORDINATES")
144 {
145 // read header line
146 // std::cout << line << "\n";
147 ss.clear();
148 ss.str(line);
149 ss >> str >> Npoints_y >> type;
150
151 // read points
152
153 std::cout << "Read Y_COORDINATES: " << Npoints_y << std::endl;
154
155 // std::cout << " reserve OK" << std::endl;
156 }
157 }
158 if (line[0] == 'Z')
159 {
160 if (line.size() > 12 && line.substr(0, 13) == "Z_COORDINATES")
161 {
162 // read header line
163 // std::cout << line << "\n";
164 ss.clear();
165 ss.str(line);
166 ss >> str >> Npoints_z >> type;
167
168 // read points
169
170 std::cout << "Read Z_COORDINATES: " << Npoints_z << std::endl;
171
172 // std::cout << " reserve OK" << std::endl;
173 }
174 }
175
176 if (line[0] == 'S')
177 {
178 if (line.size() > 6 && line.substr(0, 7) == "SCALARS")
179 {
180 ss.clear();
181 ss.str(line);
182 ss >> str >> name >> type >> numcomp;
183
184 // read LOOKUP_TABLE line
185 std::getline(infile, line);
186
187 // read data
188 std::cout << "begin reading data" << std::endl;
189
190 N_points = (Npoints_x) * (Npoints_y) * (Npoints_z);
191
192 if (DIM > 2)
193 {
194 Npoints = 8 * (Npoints_x - 1) * (Npoints_y - 1) * (Npoints_z - 1);
195 }
196 if (DIM == 2)
197 {
198 Npoints = 4 * (Npoints_x - 1) * (Npoints_y - 1);
199 }
200
201 std::vector<float> data(
202 N_points); // NB: data and gid are of different size
203 std::vector<double> gid(Npoints); // gid is the grainID for each node
204 // unsigned int gid[Npoints];
205
206 for (unsigned int i = 0; i < N_points; i++)
207 {
208 infile >> data[i];
209 // std::cout << data[i] << std::endl;
210 }
211
212 std::cout << "beginning grain_id stencil" << std::endl;
213
214 u = 0;
215 if (DIM > 2)
216 {
217 for (unsigned int i = 0; i < (Npoints_z - 1); i++)
218 {
219 for (unsigned int j = 0; j < (Npoints_y - 1); j++)
220 {
221 for (unsigned int k = 0; k < (Npoints_x - 1); k++)
222 {
223 p = k + j * Npoints_x + i * Npoints_x * Npoints_y;
224
225 gid[u] = data[p];
226 gid[u + 1] = data[p + 1];
227 gid[u + 2] = data[p + Npoints_x];
228 gid[u + 3] = data[p + Npoints_x + 1];
229 gid[u + 4] = data[p + Npoints_x * Npoints_y];
230 gid[u + 5] = data[p + Npoints_x * Npoints_y + 1];
231 gid[u + 6] =
232 data[p + Npoints_x + Npoints_x * Npoints_y];
233 gid[u + 7] =
234 data[p + Npoints_x + Npoints_x * Npoints_y + 1];
235
236 u += 8;
237 }
238 }
239 }
240 }
241
242 if (DIM == 2)
243 {
244 for (unsigned int j = 0; j < (Npoints_y - 1); j++)
245 {
246 for (unsigned int k = 0; k < (Npoints_x - 1); k++)
247 {
248 p = k + j * Npoints_x;
249
250 gid[u] = data[p];
251 gid[u + 1] = data[p + 1];
252 gid[u + 2] = data[p + Npoints_x];
253 gid[u + 3] = data[p + Npoints_x + 1];
254
255 u += 4;
256 }
257 }
258 }
259
260 data.clear();
261
262 std::cout << "Construct PField '" << name << "'" << std::endl;
263 scalar_field.push_back(
264 PField<Coordinate, double, DIM>(name, mesh, gid, 0.0));
265 std::cout << " done" << std::endl;
266
267 gid.clear();
268 }
269 }
270 }
271
272 infile.close();
273 }
