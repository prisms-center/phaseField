75inline void
76outputParameters::postprocess_and_validate(
77 const temporalDiscretization &temporal_discretization)
78{
79 // If the user has specified a list and we have list output use that and return early
80 if (condition == "LIST")
81 {
82 for (const auto &increment : user_output_list)
83 {
84 output_list.insert(static_cast<unsigned int>(increment));
85 }
86 return;
87 }
88
89 // If the number of outputs is 0 return early
90 if (n_outputs == 0)
91 {
92 return;
93 }
94
95 // If the number of outputs is greater than the number of increments, force them to be
96 // equivalent
97 n_outputs = std::min(n_outputs, temporal_discretization.total_increments);
98
99 // Determine the output list from the other criteria
100 if (condition == "EQUAL_SPACING")
101 {
102 for (unsigned int iteration = 0;
103 iteration <= temporal_discretization.total_increments;
104 iteration += temporal_discretization.total_increments / n_outputs)
105 {
106 output_list.insert(iteration);
107 }
108 }
109 else if (condition == "LOG_SPACING")
110 {
111 output_list.insert(0);
112 for (unsigned int output = 1; output <= n_outputs; output++)
113 {
114 output_list.insert(static_cast<unsigned int>(std::round(
115 std::pow(static_cast<double>(temporal_discretization.total_increments),
116 static_cast<double>(output) / static_cast<double>(n_outputs)))));
117 }
118 }
119 else if (condition == "N_PER_DECADE")
120 {
121 AssertThrow(temporal_discretization.total_increments > 1,
122 dealii::ExcMessage("For n per decaded spaced outputs, the number of "
123 "increments must be greater than 1."));
124
125 output_list.insert(0);
126 output_list.insert(1);
127 for (unsigned int iteration = 2;
128 iteration <= temporal_discretization.total_increments;
129 iteration++)
130 {
131 const auto decade = static_cast<unsigned int>(std::ceil(std::log10(iteration)));
132 const auto step_size =
133 static_cast<unsigned int>(std::pow(10, decade) / n_outputs);
134 if (iteration % step_size == 0)
135 {
136 output_list.insert(iteration);
137 }
138 }
139 }
140 else
141 {
142 AssertThrow(false, UnreachableCode());
143 }
144}
