103template <int dim>
104void
105triangulationHandler<dim>::generate_mesh(const userInputParameters<dim> &user_inputs)
106{
107 // Generate rectangle
108 dealii::GridGenerator::subdivided_hyper_rectangle(
109 *triangulation,
110 user_inputs.spatial_discretization.subdivisions,
111 dealii::Point<dim>(),
112 dealii::Point<dim>(user_inputs.spatial_discretization.domain_size));
113
114 // Mark boundaries. This is done before global refinement to reduce the number of cells
115 // we have to loop through.
116 mark_boundaries(user_inputs);
117
118 // Mark periodicity
119 mark_periodic(user_inputs);
120
121 // Output triangulation to vtk if in debug mode
122#ifdef DEBUG
123 export_triangulation_as_vtk("triangulation");
124#endif
125
126 // Global refinement
127 triangulation->refine_global(user_inputs.spatial_discretization.refine_factor);
128}

140template <int dim>
141void
142triangulationHandler<dim>::mark_boundaries(
143 const userInputParameters<dim> &user_inputs) const
144{
145 double tolerance = 1e-12;
146
147 // Loop through the cells
148 for (const auto &cell : triangulation->active_cell_iterators())
149 {
150 // Mark the faces (faces_per_cell = 2*dim)
151 for (uint face_number = 0; face_number < dealii::GeometryInfo<dim>::faces_per_cell;
152 ++face_number)
153 {
154 // Direction for quad and hex cells
155 uint direction = std::floor(face_number / 2);
156
157 // Mark the boundary id for x=0, y=0, z=0
158 if (std::fabs(cell->face(face_number)->center()(direction) - 0) < tolerance)
159 {
160 cell->face(face_number)->set_boundary_id(face_number);
161 }
162 // Mark the boundary id for x=max, y=max, z=max
163 else if (std::fabs(
164 cell->face(face_number)->center()(direction) -
165 (user_inputs.spatial_discretization.domain_size[direction])) <
166 tolerance)
167 {
168 cell->face(face_number)->set_boundary_id(face_number);
169 }
170 }
171 }
172}

174template <int dim>
175void
176triangulationHandler<dim>::mark_periodic(const userInputParameters<dim> &user_inputs)
177{
178 // Add periodicity in the triangulation where specified in the boundary conditions. Note
179 // that if one field is periodic all others should be as well.
180 for (const auto &[index, boundary_condition] :
181 user_inputs.boundary_parameters.boundary_condition_list)
182 {
183 for (const auto &[component, condition] : boundary_condition)
184 {
185 for (const auto &[boundary_id, boundary_type] :
186 condition.boundary_condition_map)
187 {
188 if (boundary_type == boundaryType::PERIODIC)
189 {
190 // Skip boundary ids that are odd since those map to the even faces
191 if (boundary_id % 2 != 0)
192 {
193 continue;
194 }
195
196 // Create a vector of matched pairs that we fill and enforce upon the
197 // constaints
198 std::vector<dealii::GridTools::PeriodicFacePair<
199 typename Triangulation::cell_iterator>>
200 periodicity_vector;
201
202 // Determine the direction
203 const uint direction = std::floor(boundary_id / dim);
204
205 // Collect the matched pairs on the coarsest level of the mesh
206 dealii::GridTools::collect_periodic_faces(*triangulation,
207 boundary_id,
208 boundary_id + 1,
209 direction,
210 periodicity_vector);
211
212 // Set constraints
213 triangulation->add_periodicity(periodicity_vector);
214 }
215 }
216 }
217 }
218}
