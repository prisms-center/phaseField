156template <int dim, int degree>
157inline void
158explicitBase<dim, degree>::compute_subset_attributes(
159 const fieldSolveType &field_solve_type)
160{
161 Assert((field_solve_type == fieldSolveType::EXPLICIT ||
162 field_solve_type == fieldSolveType::EXPLICIT_POSTPROCESS ||
163 field_solve_type == fieldSolveType::EXPLICIT_CONSTANT),
164 dealii::ExcMessage(
165 "compute_subset_attributes() should only be used for "
166 "EXPLICIT, EXPLICIT_POSTPROCESS, and EXPLICIT_CONSTANT fieldSolveTypes"));
167
168 subset_attributes.clear();
169
170 for (const auto &[index, variable] : user_inputs.var_attributes)
171 {
172 if (variable.field_solve_type == field_solve_type)
173 {
174 subset_attributes.emplace(index, variable);
175 }
176 }
177}

179template <int dim, int degree>
180inline void
181explicitBase<dim, degree>::compute_shared_dependencies()
182{
183 // Compute the shared dependency flags
184 auto &dependency_flag_set = subset_attributes.begin()->second.eval_flag_set_RHS;
185 for (const auto &[index, variable] : subset_attributes)
186 {
187 if (!variable.eval_flag_set_RHS.empty())
188 {
189 for (const auto &[pair, flag] : variable.eval_flag_set_RHS)
190 {
191 dependency_flag_set[pair] |= flag;
192 }
193 }
194 }
195 for (auto &[index, variable] : subset_attributes)
196 {
197 for (const auto &[pair, flag] : dependency_flag_set)
198 {
199 variable.eval_flag_set_RHS[pair] |= flag;
200 }
201 }
202
203 // Compute the shared dependency set
204 auto &dependency_set = subset_attributes.begin()->second.dependency_set_RHS;
205 for (const auto &[index, variable] : subset_attributes)
206 {
207 for (const auto &[index, map] : variable.dependency_set_RHS)
208 {
209 for (const auto &[dependency_type, field_type] : map)
210 {
211 dependency_set[index].emplace(dependency_type, field_type);
212 }
213 }
214 }
215 for (auto &[index, variable] : subset_attributes)
216 {
217 variable.dependency_set_RHS = dependency_set;
218 }
219
220#ifdef DEBUG
221 print();
222#endif
223}

225template <int dim, int degree>
226inline void
227explicitBase<dim, degree>::set_initial_condition()
228{
229 for (const auto &[index, variable] : subset_attributes)
230 {
231 Assert(dof_handler.const_dof_handlers.size() > index,
232 dealii::ExcMessage(
233 "The const DoFHandler set is smaller than the given index = " +
234 std::to_string(index)));
235 Assert(subset_attributes.find(index) != subset_attributes.end(),
236 dealii::ExcMessage(
237 "There is no entry in the attribute subset for the given index = " +
238 std::to_string(index)));
239 Assert(solution_handler.solution_set.find(
240 std::make_pair(index, dependencyType::NORMAL)) !=
241 solution_handler.solution_set.end(),
242 dealii::ExcMessage("There is no solution vector for the given index = " +
243 std::to_string(index) +
244 " and type = " + to_string(dependencyType::NORMAL)));
245
246 dealii::VectorTools::interpolate(
247 mapping,
248 *(dof_handler.const_dof_handlers.at(index)),
249 initialCondition<dim>(index, subset_attributes.at(index).field_type),
250 *(solution_handler.solution_set.at(
251 std::make_pair(index, dependencyType::NORMAL))));
252 }
253}
