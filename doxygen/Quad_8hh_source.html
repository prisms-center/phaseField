11 class Quad_f : public PSimpleBase<std::vector<PRISMS::Coordinate<2>>, double>
12 {
13 [[nodiscard]] double
14 eval(const std::vector<PRISMS::Coordinate<2>> &var) const override
15 {
16 // var[0]: coordinate to be evaluated
17 // var[1]: nodal coordinate
18 // var[2]: element dimension
19 // var[3]: +/- 1 depending on which 'corner' of quad
20 //
21 // f = (1.0 - e0)*(1.0 - e1)
22 // e = var[3]*(var[0] - var[1])/var[2]
23
24 return (1.0 - var[3][0] * (var[0][0] - var[1][0]) / var[2][0]) *
25 (1.0 - var[3][1] * (var[0][1] - var[1][1]) / var[2][1]);
26 }
27
28 public:
29 Quad_f()
30 {
31 this->_name = "Quad_f";
32 }
33 };

76 {
77 //_var[0]: Coordinate _r; // coordinate to evaluate field at
78 //_var[1]: Coordinate _n; // coordinate of node
79 //_var[2]: Coordinate _h; // quad dimensions
80 //_var[3]: Coordinate _s; // +/- 1, depending on orientation of basis function
81 std::vector<PRISMS::Coordinate<2>> _var;
82
83 public:
84 using size_type = typename Interpolator<Coordinate, 2>::size_type;
85
86 // node_index: index of node in mesh
87 // node_index: index of element in mesh
88 // bfunc: PFuncBase<std::vector<Coordinate>, double>*
89 // node_coord: Coordinate of node
90 // dim: Coordinate containing x and y dimension of element
91 // element_node_index: 0 == bottom left, proceed counter clockwise to 3 == top left of
92 // element
93
94 QuadValues(unsigned long int node_index,
95 unsigned long int element_index,
96 PFuncBase<std::vector<PRISMS::Coordinate<2>>, double> *bfunc,
97 const PRISMS::Coordinate<2> &node_coord,
98 const PRISMS::Coordinate<2> &dim,
99 int element_node_index)
100 : Interpolator<Coordinate, 2>(node_index, element_index, bfunc)
101 {
102 _var.resize(4);
103
104 _var[1][0] = node_coord[0];
105 _var[1][1] = node_coord[1];
106
107 _var[2][0] = dim[0];
108 _var[2][1] = dim[1];
109
110 if (element_node_index == 0)
111 {
112 _var[3][0] = 1.0;
113 _var[3][1] = 1.0;
114 }
115 else if (element_node_index == 1)
116 {
117 _var[3][0] = -1.0;
118 _var[3][1] = 1.0;
119 }
120 else if (element_node_index == 2)
121 {
122 _var[3][0] = -1.0;
123 _var[3][1] = -1.0;
124 }
125 else if (element_node_index == 3)
126 {
127 _var[3][0] = 1.0;
128 _var[3][1] = -1.0;
129 }
130 }
131
132 [[nodiscard]] PRISMS::Coordinate<2>
133 min() const override
134 {
135 PRISMS::Coordinate<2> coord = _var[1];
136
137 if (_var[3][0] == -1.0)
138 {
139 coord[0] -= _var[2][0];
140 }
141 if (_var[3][1] == -1.0)
142 {
143 coord[1] -= _var[2][1];
144 }
145
146 return coord;
147 }
148
149 [[nodiscard]] PRISMS::Coordinate<2>
150 max() const override
151 {
152 PRISMS::Coordinate<2> coord = _var[1];
153
154 if (_var[3][0] == 1.0)
155 {
156 coord[0] += _var[2][0];
157 }
158 if (_var[3][1] == 1.0)
159 {
160 coord[1] += _var[2][1];
161 }
162
163 return coord;
164 }
165
166 bool
167 is_in_range(const Coordinate &coord) override
168 {
169 _var[0][0] = coord[0];
170 _var[0][1] = coord[1];
171
172 for (int i = 0; i < 2; i++)
173 {
174 double e = _var[3][i] * (_var[0][i] - _var[1][i]) / _var[2][i];
175 if (e < 0.0 || e >= 1.0)
176 {
177 return false;
178 }
179 }
180
181 return true;
182 }

192 }
