117template <int dim, int degree>
118inline void
119nonexplicitAuxiliarySolver<dim, degree>::init()
120{
121 this->compute_subset_attributes(fieldSolveType::NONEXPLICIT_AUXILIARY);
122
123 // If the subset attribute is empty return early
124 if (this->subset_attributes.empty())
125 {
126 return;
127 }
128
129 for (const auto &[index, variable] : this->subset_attributes)
130 {
131 // Creating temporary map to match types
132 std::map<unsigned int, variableAttributes> temp;
133 temp.emplace(index, variable);
134 subset_attributes_list.push_back(temp);
135
136 // Create the implementation of customPDE with the subset of variable attributes
137 this->system_matrix[index] =
138 std::make_unique<SystemMatrixType>(this->user_inputs,
139 subset_attributes_list.back(),
140 index);
141
142 // Set up the user-implemented equations and create the residual vectors
143 this->system_matrix.at(index)->clear();
144 this->system_matrix.at(index)->initialize(
145 this->matrix_free_handler.get_matrix_free());
146
147 // Create the subset of solution vectors and add the mapping to customPDE
148 new_solution_subset[index].push_back(
149 this->solution_handler.new_solution_set.at(index));
150 solution_subset[index].push_back(this->solution_handler.solution_set.at(
151 std::make_pair(index, dependencyType::NORMAL)));
152 global_to_local_solution[index].emplace(std::make_pair(index,
153 dependencyType::NORMAL),
154 0);
155 for (const auto &[variable_index, map] :
156 subset_attributes_list.back().begin()->second.dependency_set_RHS)
157 {
158 for (const auto &[dependency_type, field_type] : map)
159 {
160 const auto pair = std::make_pair(variable_index, dependency_type);
161
162 Assert(this->solution_handler.solution_set.find(pair) !=
163 this->solution_handler.solution_set.end(),
164 dealii::ExcMessage(
165 "There is no solution vector for the given index = " +
166 std::to_string(variable_index) +
167 " and type = " + to_string(dependency_type)));
168
169 Assert(this->solution_handler.new_solution_set.find(variable_index) !=
170 this->solution_handler.new_solution_set.end(),
171 dealii::ExcMessage(
172 "There is no new solution vector for the given index = " +
173 std::to_string(variable_index)));
174
175 solution_subset[index].push_back(
176 this->solution_handler.solution_set.at(pair));
177 global_to_local_solution[index].emplace(pair,
178 solution_subset.at(index).size() -
179 1);
180 }
181 }
182 this->system_matrix.at(index)->add_global_to_local_mapping(
183 global_to_local_solution.at(index));
184 }
185}

187template <int dim, int degree>
188inline void
189nonexplicitAuxiliarySolver<dim, degree>::solve()
190{
191 // If the subset attribute is empty return early
192 if (this->subset_attributes.empty())
193 {
194 return;
195 }
196
197 for (const auto &[index, variable] : this->subset_attributes)
198 {
199 // Compute the update
200 this->system_matrix.at(index)->compute_nonexplicit_auxiliary_update(
201 new_solution_subset.at(index),
202 solution_subset.at(index));
203
204 // Scale the update by the respective (SCALAR/VECTOR) invm.
205 new_solution_subset.at(index).at(0)->scale(this->invm_handler.get_invm(index));
206
207 // Update the solutions
208 this->solution_handler.update(fieldSolveType::NONEXPLICIT_AUXILIARY, index);
209
210 // Apply constraints
211 this->constraint_handler.get_constraint(index).distribute(
212 *(this->solution_handler.solution_set.at(
213 std::make_pair(index, dependencyType::NORMAL))));
214 }
215}
