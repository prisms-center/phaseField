18struct linearSolverParameters
19{
20public:
21 // Solver tolerance
22 double tolerance = 1.0e-6;
23
24 // Solver tolerance type
25 solverToleranceType tolerance_type = solverToleranceType::RELATIVE_RESIDUAL_CHANGE;
26
27 // Max number of iterations for the linear solve
28 unsigned int max_iterations = 100;
29
30 // Preconditioner
31 preconditionerType preconditioner = preconditionerType::GMG;
32
33 // Smoothing range for eigenvalues. This denotes the lower bound of eigenvalues that are
34 // smoothed [1.2 λ^max / smoothing_range, 1.2 λ^max], where λ^max is the estimated
35 // maximum eigenvalue. A choice between 5 and 20 is usually useful when the
36 // preconditioner is used as a smoother in multigrid.
37 double smoothing_range = 15.0;
38
39 // Polynomial degree for the Chebyshev smoother
40 unsigned int smoother_degree = 5;
41
42 // Maximum number of CG iterations used to find the maximum eigenvalue
43 unsigned int eig_cg_n_iterations = 10;
44};

74inline void
75linearSolveParameters::print_parameter_summary() const
76{
77 if (!linear_solve.empty())
78 {
79 conditionalOStreams::pout_summary()
80 << "================================================\n"
81 << " Linear Solve Parameters\n"
82 << "================================================\n";
83
84 for (const auto &[index, linear_solver_parameters] : linear_solve)
85 {
86 conditionalOStreams::pout_summary()
87 << "Index: " << index << "\n"
88 << " Tolerance: " << linear_solver_parameters.tolerance << "\n"
89 << " Type: " << to_string(linear_solver_parameters.tolerance_type) << "\n"
90 << " Max iterations: " << linear_solver_parameters.max_iterations << "\n"
91 << " Preconditioner: " << to_string(linear_solver_parameters.preconditioner)
92 << "\n";
93
94 if (linear_solver_parameters.preconditioner == preconditionerType::GMG)
95 {
96 conditionalOStreams::pout_summary()
97 << " Smoothing range: " << linear_solver_parameters.smoothing_range
98 << "\n"
99 << " Smoother degree: " << linear_solver_parameters.smoother_degree
100 << "\n"
101 << " Max eigenvalue CG iterations: "
102 << linear_solver_parameters.eig_cg_n_iterations << "\n";
103 }
104 }
105
106 conditionalOStreams::pout_summary() << "\n" << std::flush;
107 }
108}
