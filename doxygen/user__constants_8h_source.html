220template <int dim>
221inline unsigned int
222userConstants<dim>::compute_tensor_parentheses(
223 const unsigned int &n_elements,
224 const std::vector<std::string> &tensor_elements)
225{
226 unsigned int open_parentheses = 0;
227 unsigned int close_parentheses = 0;
228
229 for (unsigned int element = 0; element < n_elements; element++)
230 {
231 for (const char c : tensor_elements.at(element))
232 {
233 if (c == '(')
234 {
235 ++open_parentheses;
236 }
237 else if (c == ')')
238 {
239 ++close_parentheses;
240 }
241 }
242 }
243
244 if (open_parentheses != close_parentheses)
245 {
246 std::cerr << "PRISMS-PF ERROR: User-defined elastic constant "
247 "list does not have the same number of open and "
248 "close parentheses.\n";
249 abort();
250 }
251
252 return open_parentheses;
253}

255template <int dim>
256inline void
257userConstants<dim>::remove_parentheses(std::vector<std::string> &tensor_elements)
258{
259 for (std::string &element : tensor_elements)
260 {
261 element.erase(std::remove(element.begin(), element.end(), '('), element.end());
262 element.erase(std::remove(element.begin(), element.end(), ')'), element.end());
263 }
264}

267inline dealii::Tensor<1, dim>
268userConstants<dim>::compute_rank_1_tensor_constant(
269 const unsigned int &n_elements,
270 const std::vector<std::string> &tensor_elements)
271{
272 AssertThrow(n_elements > 1 && n_elements < 4,
273 dealii::ExcMessage("PRISMS-PF Error: The columns in user-defined constant "
274 "tensors cannot be longer than 3 elements (internally "
275 "truncated to the number of dimensions)."));
276
277 dealii::Tensor<1, dim> temp;
278 for (unsigned int i = 0; i < dim; i++)
279 {
280 temp[i] = dealii::Utilities::string_to_double(tensor_elements.at(i));
281 }
282
283 return temp;
284}

286template <int dim>
287inline dealii::Tensor<2, dim>
288userConstants<dim>::compute_rank_2_tensor_constant(
289 const unsigned int &n_elements,
290 const std::vector<std::string> &tensor_elements)
291{
292 unsigned int row_length = 0;
293 if (n_elements == 4)
294 {
295 AssertThrow(dim < 3,
296 dealii::ExcMessage(
297 "PRISMS-PF ERROR: User-defined constant tensor does not have "
298 "enough elements. For 3D calculations matrices must be 3x3."));
299
300 row_length = 2;
301 }
302 else if (n_elements == 9)
303 {
304 row_length = 3;
305 }
306 else
307 {
308 AssertThrow(false,
309 dealii::ExcMessage("PRISMS-PF ERROR: User-defined constant tensor does "
310 "not have the correct number of elements, matrices "
311 "must be 2x2 or 3x3."));
312 }
313
314 dealii::Tensor<2, dim> temp;
315 for (unsigned int i = 0; i < dim; i++)
316 {
317 for (unsigned int j = 0; j < dim; j++)
318 {
319 temp[i][j] =
320 dealii::Utilities::string_to_double(tensor_elements.at((i * row_length) + j));
321 }
322 }
323
324 return temp;
325}

327template <int dim>
328inline typename userConstants<dim>::InputVariant
329userConstants<dim>::construct_user_constant(
330 std::vector<std::string> &model_constants_strings)
331{
332 // Ensure that the input includes a value and a type
333 AssertThrow(model_constants_strings.size() > 1,
334 dealii::ExcMessage("PRISMS-PF Error: At least two fields are required for "
335 "user-defined variables (value and type)."));
336
337 std::vector<std::string> model_constants_type_strings =
338 dealii::Utilities::split_string_list(model_constants_strings.at(
339 model_constants_strings.size() - 1),
340 ' ');
341
342 if (model_constants_strings.size() == 2)
343 {
344 return primitive_model_constant(model_constants_strings);
345 }
346 else
347 {
348 if (boost::iequals(model_constants_type_strings.at(0), "tensor"))
349 {
350 const unsigned int n_elements = model_constants_strings.size() - 1;
351
352 const unsigned int open_parentheses =
353 compute_tensor_parentheses(n_elements, model_constants_strings);
354 remove_parentheses(model_constants_strings);
355
356 // Rank 1 tensor
357 if (open_parentheses < 3)
358 {
359 return compute_rank_1_tensor_constant(n_elements, model_constants_strings);
360 }
361 // Rank 2 tensor
362 else if (open_parentheses < 5)
363 {
364 return compute_rank_2_tensor_constant(n_elements, model_constants_strings);
365 }
366 }
367 else if (boost::iequals(model_constants_type_strings.at(1), "elastic") &&
368 boost::iequals(model_constants_type_strings.at(2), "constants"))
369 {
370 const unsigned int n_elements = model_constants_strings.size() - 1;
371
372 remove_parentheses(model_constants_strings);
373
374 // Load in the elastic constants as a vector
375 std::vector<double> temp_elastic_constants;
376 for (unsigned int i = 0; i < n_elements; i++)
377 {
378 temp_elastic_constants.push_back(
379 dealii::Utilities::string_to_double(model_constants_strings.at(i)));
380 }
381
382 const std::string &elastic_const_symmetry = model_constants_type_strings.at(0);
383 dealii::Tensor<2, (2 * dim) - 1 + (dim / 3)> temp =
384 get_Cij_tensor(temp_elastic_constants, elastic_const_symmetry);
385 return temp;
386 }
387 else
388 {
389 AssertThrow(false,
390 dealii::ExcMessage(
391 "PRISMS-PF ERROR: Only user-defined constant tensors may "
392 "have multiple elements."));
393 }
394 return 0;
395 }
396}

398template <int dim>
399inline typename userConstants<dim>::InputVariant
400userConstants<dim>::primitive_model_constant(
401 std::vector<std::string> &model_constants_strings)
402{
403 std::vector<std::string> model_constants_type_strings =
404 dealii::Utilities::split_string_list(model_constants_strings.at(
405 model_constants_strings.size() - 1),
406 ' ');
407
408 if (boost::iequals(model_constants_type_strings.at(0), "double"))
409 {
410 return dealii::Utilities::string_to_double(model_constants_strings.at(0));
411 }
412 else if (boost::iequals(model_constants_type_strings.at(0), "int"))
413 {
414 return dealii::Utilities::string_to_int(model_constants_strings.at(0));
415 }
416 else if (boost::iequals(model_constants_type_strings.at(0), "bool"))
417 {
418 bool temp = boost::iequals(model_constants_strings.at(0), "true");
419 return temp;
420 }
421 else
422 {
423 AssertThrow(false,
424 dealii::ExcMessage(
425 "PRISMS-PF Error: The type for user-defined variables must be "
426 "`double`, `int`, `bool`, `tensor`, or `elastic constants`."));
427 return 0;
428 }

431template <int dim>
432inline dealii::Tensor<2, (2 * dim) - 1 + (dim / 3)>
433userConstants<dim>::get_Cij_tensor(std::vector<double> elastic_constants,
434 const std::string &elastic_const_symmetry) const
435{
436 // First set the material model
437 elasticityModel mat_model = ISOTROPIC;
438 if (elastic_const_symmetry == "isotropic")
439 {
440 mat_model = elasticityModel::ISOTROPIC;
441 }
442 else if (elastic_const_symmetry == "transverse")
443 {
444 mat_model = elasticityModel::TRANSVERSE;
445 }
446 else if (elastic_const_symmetry == "orthotropic")
447 {
448 mat_model = elasticityModel::ORTHOTROPIC;
449 }
450 else if (elastic_const_symmetry == "anisotropic")
451 {
452 mat_model = elasticityModel::ANISOTROPIC;
453 }
454 else
455 {
456 // Should change to an exception
457 std::cerr << "Elastic material model is invalid, please use isotropic, "
458 "transverse, orthotropic, or anisotropic\n";
459 }
460
461 // If the material model is anisotropic for a 2D calculation but the elastic
462 // constants are given for a 3D calculation, change the elastic constant
463 // vector to the 2D form
464 if ((mat_model == ANISOTROPIC) && (dim == 2) && elastic_constants.size() == 21)
465 {
466 std::vector<double> elastic_constants_temp = elastic_constants;
467 elastic_constants.clear();
468 const std::vector<unsigned int> indices_2D = {0, 1, 5, 6, 10, 14};
469 for (const auto &index : indices_2D)
470 {
471 elastic_constants.push_back(elastic_constants_temp.at(index));
472 }
473 }
474
475 return getCIJMatrix(mat_model, elastic_constants);
476}

478template <int dim>
479inline dealii::Tensor<2, (2 * dim) - 1 + (dim / 3)>
480userConstants<dim>::getCIJMatrix(const elasticityModel &model,
481 const std::vector<double> &constants) const
482{
483 // CIJ.fill(0.0);
484 dealii::Tensor<2, (2 * dim) - 1 + (dim / 3)> CIJ;
485
486 conditionalOStreams::pout_base() << "Reading material model:";
487 switch (dim)
488 {
489 case 1:
490 {
491 conditionalOStreams::pout_base() << " 1D ";
492 // 1D models
493 switch (model)
494 {
495 case ISOTROPIC:
496 {
497 conditionalOStreams::pout_base() << " ISOTROPIC \n";
498 CIJ[0][0] = constants[0];
499 break;
500 }
501 default:
502 {
503 exit(-1);
504 }
505 }
506 break;
507 }
508 case 2:
509 {
510 conditionalOStreams::pout_base() << " 2D ";
511 // 2D models
512 switch (model)
513 {
514 case ISOTROPIC:
515 {
516 conditionalOStreams::pout_base() << " ISOTROPIC \n";
517 const double E = constants[0];
518 const double nu = constants[1];
519 const double mu = E / (2 * (1 + nu));
520 const double lambda = nu * E / ((1 + nu) * (1 - 2 * nu));
521 CIJ[0][0] = lambda + 2 * mu;
522 CIJ[1][1] = lambda + 2 * mu;
523 CIJ[2][2] = mu;
524 CIJ[0][1] = CIJ[1][0] = lambda;
525 break;
526 }
527 case ANISOTROPIC:
528 {
529 conditionalOStreams::pout_base() << " ANISOTROPIC \n";
530 CIJ[0][0] = constants[0]; // C11
531 CIJ[1][1] = constants[1]; // C22
532 CIJ[2][2] = constants[2]; // C33
533 CIJ[0][1] = CIJ[1][0] = constants[3]; // C12
534 CIJ[0][2] = CIJ[2][0] = constants[4]; // C13
535 CIJ[1][2] = CIJ[2][1] = constants[5]; // C23
536 break;
537 }
538 default:
539 {
540 exit(-1);
541 }
542 }
543 break;
544 }
545 case 3:
546 {
547 conditionalOStreams::pout_base() << " 3D ";
548 // 3D models
549 switch (model)
550 {
551 case ISOTROPIC:
552 {
553 conditionalOStreams::pout_base() << " ISOTROPIC \n";
554 const double E = constants[0];
555 const double nu = constants[1];
556 const double mu = E / (2 * (1 + nu));
557 const double lambda = nu * E / ((1 + nu) * (1 - 2 * nu));
558 CIJ[0][0] = lambda + 2 * mu;
559 CIJ[1][1] = lambda + 2 * mu;
560 CIJ[2][2] = lambda + 2 * mu;
561 CIJ[3][3] = mu;
562 CIJ[4][4] = mu;
563 CIJ[5][5] = mu;
564 CIJ[0][1] = CIJ[1][0] = lambda;
565 CIJ[0][2] = CIJ[2][0] = lambda;
566 CIJ[1][2] = CIJ[2][1] = lambda;
567 break;
568 }
569 case TRANSVERSE:
570 {
571 conditionalOStreams::pout_base() << " TRANSVERSE \n";
572 CIJ[0][0] = constants[0]; // C11
573 CIJ[1][1] = constants[0]; // C11
574 CIJ[2][2] = constants[1]; // C33
575 CIJ[3][3] = constants[2]; // C44
576 CIJ[4][4] = constants[2]; // C44
577 CIJ[5][5] = (constants[0] - constants[3]) / 2.0; //(C11-C12)/2
578 CIJ[0][1] = CIJ[1][0] = constants[3]; // C12
579 CIJ[0][2] = CIJ[2][0] = constants[4]; // C13
580 CIJ[1][2] = CIJ[2][1] = constants[4]; // C13
581 break;
582 }
583 case ORTHOTROPIC:
584 {
585 conditionalOStreams::pout_base() << " ORTHOTROPIC \n";
586 CIJ[0][0] = constants[0]; // C11
587 CIJ[1][1] = constants[1]; // C22
588 CIJ[2][2] = constants[2]; // C33
589 CIJ[3][3] = constants[3]; // C44
590 CIJ[4][4] = constants[4]; // C55
591 CIJ[5][5] = constants[5]; // C66
592 CIJ[0][1] = CIJ[1][0] = constants[6]; // C12
593 CIJ[0][2] = CIJ[2][0] = constants[7]; // C13
594 CIJ[1][2] = CIJ[2][1] = constants[8]; // C23
595 break;
596 }
597 case ANISOTROPIC:
598 {
599 conditionalOStreams::pout_base() << " ANISOTROPIC \n";
600 CIJ[0][0] = constants[0]; // C11
601 CIJ[1][1] = constants[1]; // C22
602 CIJ[2][2] = constants[2]; // C33
603 CIJ[3][3] = constants[3]; // C44
604 CIJ[4][4] = constants[4]; // C55
605 CIJ[5][5] = constants[5]; // C66
606 CIJ[0][1] = CIJ[1][0] = constants[6]; // C12
607 CIJ[0][2] = CIJ[2][0] = constants[7]; // C13
608 CIJ[0][3] = CIJ[3][0] = constants[8]; // C14
609 CIJ[0][4] = CIJ[4][0] = constants[9]; // C15
610 CIJ[0][5] = CIJ[5][0] = constants[10]; // C16
611 CIJ[1][2] = CIJ[2][1] = constants[11]; // C23
612 CIJ[1][3] = CIJ[3][1] = constants[12]; // C24
613 CIJ[1][4] = CIJ[4][1] = constants[13]; // C25
614 CIJ[1][5] = CIJ[5][1] = constants[14]; // C26
615 CIJ[2][3] = CIJ[3][2] = constants[15]; // C34
616 CIJ[2][4] = CIJ[4][2] = constants[16]; // C35
617 CIJ[2][5] = CIJ[5][2] = constants[17]; // C36
618 CIJ[3][4] = CIJ[4][3] = constants[18]; // C45
619 CIJ[3][5] = CIJ[5][3] = constants[19]; // C46
620 CIJ[4][5] = CIJ[5][4] = constants[20]; // C56
621 break;
622 }
623 default:
624 {
625 exit(-1);
626 }
627 }
628 break;
629 }
630 default:
631 {
632 exit(-1);
633 }
634 }
635 // print CIJ to terminal
636 conditionalOStreams::pout_base() << "Elasticity matrix (Voigt notation):\n";
637 constexpr unsigned int voight_matrix_size = (2 * dim) - 1 + (dim / 3);
638 for (unsigned int i = 0; i < voight_matrix_size; i++)
639 {
640 for (unsigned int j = 0; j < voight_matrix_size; j++)
641 {
642 conditionalOStreams::pout_base() << std::setw(8) << std::setprecision(3)
643 << std::scientific << CIJ[i][j] << " ";
644 }
645 conditionalOStreams::pout_base() << "\n";
646 }
647 conditionalOStreams::pout_base() << "\n";
648 return CIJ;
649}
