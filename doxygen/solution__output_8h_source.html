48template <int dim, typename number>
49solutionOutput<dim, number>::solutionOutput(
50 const dealii::LinearAlgebra::distributed::Vector<number> &solution,
51 const dealii::DoFHandler<dim> &dof_handler,
52 const unsigned int &degree,
53 const std::string &name,
54 const userInputParameters<dim> &user_inputs)
55{
56 // Some stuff to determine the actual name of the output file.
57 const auto n_trailing_digits = static_cast<unsigned int>(
58 std::floor(std::log10(user_inputs.temporal_discretization.total_increments)) + 1);
59
60 // Init data out
61 dealii::DataOut<dim> data_out;
62
63 // Add data vector
64 data_out.add_data_vector(dof_handler, solution, name);
65
66 // Build patches to linearly interpolate from higher order element degrees. Note that
67 // this essentially converts the element to an equal amount of subdivisions in the
68 // output. This does not make subdivisions and element degree equivalent in the
69 // simulation!
70 data_out.build_patches(degree);
71
72 // Set some flags for data output
73 dealii::DataOutBase::VtkFlags flags;
74 flags.time = user_inputs.temporal_discretization.time;
75 flags.cycle = user_inputs.temporal_discretization.increment;
76 flags.print_date_and_time = true;
77#ifdef PRISMS_PF_WITH_ZLIB
78 flags.compression_level = dealii::DataOutBase::CompressionLevel::best_speed;
79#endif
80 data_out.set_flags(flags);
81
82 // Write to file based on the user input. TODO: actually write stuff according to user
83 // input.
84 data_out.write_vtu_with_pvtu_record("./",
85 name,
86 user_inputs.temporal_discretization.increment,
87 MPI_COMM_WORLD,
88 n_trailing_digits);
89}
