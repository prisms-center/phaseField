55template <int dim>
56inline void
57initialCondition<dim>::vector_value(const dealii::Point<dim> &p,
58 dealii::Vector<double> &value) const
59{
60 // Initialize passed variables to zero
61 dealii::Vector<double> vector_value(dim);
62
63 // Pass variables to user-facing function to evaluate
64 for (unsigned int i = 0; i < dim; i++)
65 {
66 custom_initial_condition.set_initial_condition(index,
67 i,
68 p,
69 vector_value(0),
70 vector_value(i));
71 }
72
73 value = vector_value;
74}
