43 void
44 set_initial_condition([[maybe_unused]] const unsigned int &index,
45 [[maybe_unused]] const unsigned int &component,
46 [[maybe_unused]] const dealii::Point<dim> &point,
47 [[maybe_unused]] number &scalar_value,
48 [[maybe_unused]] number &vector_component_value) const override
49 {
50 double dist1 = point.distance(center1);
51 double dist2 = point.distance(center2);
52
53 if (index == 0)
54 {
55 scalar_value = matrix_concentration;
56 scalar_value += 0.5 * (0.125) * (1.0 - std::tanh((dist1 - radius1) / (1.0)));
57 scalar_value += 0.5 * (0.125) * (1.0 - std::tanh((dist2 - radius2) / (1.0)));
58 }
59 else
60 {
61 scalar_value += 0.5 * (1.0 - std::tanh((dist1 - radius1) / (1.0)));
62 scalar_value += 0.5 * (1.0 - std::tanh((dist2 - radius2) / (1.0)));
63 }

66 void
67 compute_rhs(FieldContainer<dim, degree, number> &variable_list,
68 const SimulationTimer &sim_timer,
69 unsigned int solve_block_id) const override
70 {
71 const double dt = sim_timer.get_timestep();
72 if (solve_block_id == 0) // explicit
73 {
74 // --- Getting the values and derivatives of the model variables ---
75
76 const ScalarValue c = variable_list.template get_value<Scalar, OldOne>(0);
77 const ScalarGrad cx = variable_list.template get_gradient<Scalar, OldOne>(0);
78
79 const ScalarValue n = variable_list.template get_value<Scalar, OldOne>(1);
80 const ScalarGrad nx = variable_list.template get_gradient<Scalar, OldOne>(1);
81
82 // --- Setting the expressions for the terms in the governing equations ---
83
84 const ScalarValue c2 = c * c;
85 const ScalarValue c3 = c * c2;
86 const ScalarValue c4 = c2 * c2;
87
88 const ScalarValue n2 = n * n;
89 const ScalarValue n3 = n * n2;
90 const ScalarValue n4 = n2 * n2;
91 const ScalarValue n5 = n * n4;
92
93 // Free energy for each phase and their first and second derivatives
94 ScalarValue fa = (-1.6704 - 4.776 * c + 5.1622 * c2 - 2.7375 * c3 + 1.3687 * c4);
95 ScalarValue fac = (-4.776 + 10.3244 * c - 8.2125 * c2 + 5.4748 * c3);
96 ScalarValue facc = (10.3244 - 16.425 * c + 16.4244 * c2);
97 ScalarValue fb = (5.0 * c2 - 5.9746 * c - 1.5924);
98 ScalarValue fbc = (10.0 * c - 5.9746);
99 ScalarValue fbcc = (10.0);
100
101 // Interpolation function and its derivative
102 ScalarValue h = (10.0 * n3 - 15.0 * n4 + 6.0 * n5);
103 ScalarValue hn = (30.0 * n2 - 60.0 * n3 + 30.0 * n4);
104
105 // Residual equations
106 ScalarGrad mux = (cx * ((1.0 - h) * facc + h * fbcc) + nx * ((fbc - fac) * hn));
107 ScalarValue eq_c = c;
108 ScalarGrad eqx_c = ((-Mc * dt) * mux);
109 ScalarValue eq_n = (n - (dt * Mn) * (fb - fa) * hn);
110 ScalarGrad eqx_n = ((-dt * Kn * Mn) * nx);
111
112 // --- Submitting the terms for the governing equations ---
113
114 // Terms for the equation to evolve the concentration
115 variable_list.set_value_term(0, eq_c);
116 variable_list.set_gradient_term(0, eqx_c);
117
118 // Terms for the equation to evolve the order parameter
119 variable_list.set_value_term(1, eq_n);
120 variable_list.set_gradient_term(1, eqx_n);
121 }
122 else if (solve_block_id == 1) // postprocess
123 {
124 // --- Getting the values and derivatives of the model variables ---
125
126 // c
127 const ScalarValue c = variable_list.template get_value<Scalar, Current>(0);
128 const ScalarGrad cx = variable_list.template get_gradient<Scalar, Current>(0);
129
130 // n
131 const ScalarValue n = variable_list.template get_value<Scalar, Current>(1);
132 const ScalarGrad nx = variable_list.template get_gradient<Scalar, Current>(1);
133
134 // --- Setting the expressions for the terms in the postprocessing expressions
135 // ---
136 const ScalarValue c2 = c * c;
137 const ScalarValue c3 = c * c2;
138 const ScalarValue c4 = c2 * c2;
139
140 const ScalarValue n2 = n * n;
141 const ScalarValue n3 = n * n2;
142 const ScalarValue n4 = n2 * n2;
143 const ScalarValue n5 = n * n4;
144
145 // Free energy for each phase and their first and second derivatives
146 ScalarValue fa = (-1.6704 - 4.776 * c + 5.1622 * c2 - 2.7375 * c3 + 1.3687 * c4);
147 ScalarValue fb = (5.0 * c2 - 5.9746 * c - 1.5924);
148
149 // Interpolation function and its derivative
150 ScalarValue h = (10.0 * n3 - 15.0 * n4 + 6.0 * n5);
151
152 // The homogeneous free energy
153 ScalarValue f_chem = (1.0 - h) * fa + h * fb;
154
155 // The gradient free energy
156 ScalarValue f_grad = 0.5 * Kn * nx * nx;
157
158 // The total free energy
159 ScalarValue f_tot = f_chem + f_grad;
160
161 // The magnitude of the gradient of c
162 ScalarValue mag_grad_c = cx.norm();
163
164 // --- Submitting the terms for the postprocessing expressions ---
165
166 variable_list.set_value_term(2, f_tot);
167 variable_list.set_value_term(3, mag_grad_c);
168 }
169 }

178 number matrix_concentration;
179};
