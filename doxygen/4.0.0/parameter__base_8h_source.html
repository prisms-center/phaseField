52[[nodiscard]] static std::array<std::string, 4>
53generate_aliases(std::string_view name)
54{
55 std::string snake;
56 std::string title;
57 std::string title_snake;
58 std::string pascal;
59 snake.reserve(name.size());
60 title.reserve(name.size());
61 title_snake.reserve(name.size());
62 pascal.reserve(name.size());
63
64 bool start_of_word = true;
65 for (char character : name)
66 {
67 if (character == ' ')
68 {
69 snake += '_';
70 title += ' ';
71 title_snake += '_';
72 start_of_word = true;
73 continue;
74 }
75
76 const char lower = (char) std::tolower((unsigned char) character);
77 const char upper = (char) std::toupper((unsigned char) character);
78
79 snake += lower;
80 title += start_of_word ? upper : lower;
81 title_snake += start_of_word ? upper : lower;
82 pascal += start_of_word ? upper : lower;
83
84 start_of_word = false;
85 }
86
87 return {
88 std::move(snake),
89 std::move(title),
90 std::move(title_snake),
91 std::move(pascal),
92 };
93}

116static void
117declare_entry(dealii::ParameterHandler &parameter_handler,
118 const std::string_view &entry,
119 const std::string_view &default_value,
120 const dealii::Patterns::PatternBase &pattern,
121 const std::string_view &documentation,
122 const std::list<std::string_view> &aliases = {},
123 const std::string_view &warnings = {},
124 const std::string_view &example = {},
125 bool deprecated = false)
126{
127 const auto _entry = std::string(entry);
128
129 // Format the warning and example strings and append them to the documentation
130 auto _doc = std::string(documentation);
131
132 if (!warnings.empty())
133 {
134 _doc += "\n\n@warning " + std::string(warnings) + "\n@end_warning";
135 }
136 if (!example.empty())
137 {
138 _doc += "\n\n@example\n" + std::string(example) + "\n@end_example";
139 }
140
141 parameter_handler.declare_entry(_entry, std::string(default_value), pattern, _doc);
142#if DEAL_II_VERSION_MAJOR >= 9 && DEAL_II_VERSION_MINOR >= 7
143 parameter_handler.mark_as_deprecated(_entry, deprecated);
144#endif
145 declare_aliases_with_generated(parameter_handler, _entry, aliases);
146}
