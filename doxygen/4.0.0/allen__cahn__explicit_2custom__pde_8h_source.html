11class CustomPDE : public PDEOperatorBase<dim, degree, number>

34 void
35 set_initial_condition([[maybe_unused]] const unsigned int &index,
36 [[maybe_unused]] const unsigned int &component,
37 [[maybe_unused]] const dealii::Point<dim> &point,
38 [[maybe_unused]] number &scalar_value,
39 [[maybe_unused]] number &vector_component_value) const override
40 {
41 const dealii::Tensor<1, dim> &mesh_size =
42 get_user_inputs().spatial_discretization.rectangular_mesh.size;
43 constexpr number center[12][3] = {
44 {0.1, 0.3, 0},
45 {0.8, 0.7, 0},
46 {0.5, 0.2, 0},
47 {0.4, 0.4, 0},
48 {0.3, 0.9, 0},
49 {0.8, 0.1, 0},
50 {0.9, 0.5, 0},
51 {0.0, 0.1, 0},
52 {0.1, 0.6, 0},
53 {0.5, 0.6, 0},
54 {1, 1, 0},
55 {0.7, 0.95, 0}
56 };
57 constexpr number rad[12] = {12, 14, 19, 16, 11, 12, 17, 15, 20, 10, 11, 14};
58 number dist = 0.0;
59 for (unsigned int i = 0; i < 12; i++)
60 {
61 dist = 0.0;
62 for (unsigned int dir = 0; dir < dim; dir++)
63 {
64 const number comp_diff = point[dir] - center[i][dir] * mesh_size[dir];
65 dist += comp_diff * comp_diff;
66 }
67 dist = std::sqrt(dist);
68
69 scalar_value += 0.5 * (1.0 - std::tanh((dist - rad[i]) / 1.5));
70 }
71 scalar_value = std::min(scalar_value, static_cast<number>(1.0));
72 }

74 void
75 compute_rhs(FieldContainer<dim, degree, number> &variable_list,
76 const SimulationTimer &sim_timer,
77 unsigned int solve_block_id) const override
78 {
79 if (solve_block_id == 1) // explicit
80 {
81 ScalarValue n_val = variable_list.template get_value<Scalar, OldOne>(0);
82 ScalarGrad n_grad = variable_list.template get_gradient<Scalar, OldOne>(0);
83 ScalarValue f_well = 4.0 * n_val * (n_val - 1.0) * (n_val - 0.5);
84 ScalarValue eq_n = n_val - sim_timer.get_timestep() * m_well * f_well;
85 ScalarGrad eqx_n = -sim_timer.get_timestep() * kappa * m_well * n_grad;
86
87 variable_list.set_value_term(0, eq_n);
88 variable_list.set_gradient_term(0, eqx_n);
89 }
90 else if (solve_block_id == 2) // postprocess
91 {
92 ScalarValue n_val = variable_list.template get_value<Scalar, Current>(0);
93 ScalarGrad n_grad = variable_list.template get_gradient<Scalar, Current>(0);
94 ScalarValue f_tot = 0.0;
95 ScalarValue f_chem =
96 n_val * n_val * n_val * n_val - 2.0 * n_val * n_val * n_val + n_val * n_val;
97 ScalarValue f_grad = 0.0;
98 for (unsigned int i = 0; i < dim; i++)
99 {
100 f_grad += 0.5 * kappa * n_grad[i] * n_grad[i];
101 }
102 f_tot = f_chem + f_grad;
103
104 variable_list.set_value_term(1,
105 std::sqrt(n_grad[0] * n_grad[0] +
106 n_grad[1] * n_grad[1]));
107 variable_list.set_value_term(2, f_tot);
108 }
109 }
110
111 number m_well;
112 number kappa;
113};
