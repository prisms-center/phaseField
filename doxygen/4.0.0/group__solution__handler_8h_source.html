331template <unsigned int dim, typename number>
332template <unsigned int degree>
333inline void
334GroupSolutionHandler<dim, number>::reinit_mg_transfer(
335 const DoFManager<dim, degree> &dof_manager)
336{
337 // 1. Initialize constraints
338 const unsigned int num_blocks = solve_block.field_indices.size();
339 mg_constraints = std::vector<dealii::MGConstrainedDoFs>(num_blocks);
340 mg_transfer = std::vector<dealii::MGTransferMatrixFree<dim, number>>(num_blocks);
341 for (unsigned int block_index = 0; block_index < num_blocks; block_index++)
342 {
343 unsigned int field_index = block_to_global_index[block_index];
344 mg_constraints[block_index].initialize(
345 dof_manager.get_field_dof_handler(field_index));
346 // 2. Initialize MG Transfer
347 mg_transfer[block_index].initialize_constraints(mg_constraints[block_index]);
348 mg_transfer[block_index].build(dof_manager.get_field_dof_handler(field_index));
349 }
350}

352template <unsigned int dim, typename number>
353template <unsigned int degree>
354inline void
355GroupSolutionHandler<dim, number>::mg_transfer_down(
356 const DoFManager<dim, degree> &dof_manager,
357 unsigned int finest_level,
358 bool transfer_old_solutions)
359{
360 const unsigned int num_blocks = solve_block.field_indices.size();
361 for (unsigned int block_index = 0; block_index < num_blocks; block_index++)
362 {
363 unsigned int field_index = block_to_global_index[block_index];
364 dealii::MGLevelObject<SolutionVector<number>> temp_mg_solutions(
365 1 + finest_level - solution_levels.size(),
366 finest_level);
367
368 const auto &dof_handler = dof_manager.get_field_dof_handler(field_index);
369
370 // transfer regular solutions to mg levels
371 mg_transfer[block_index].interpolate_to_mg(dof_handler,
372 temp_mg_solutions,
373 primary_solutions.solutions.block(
374 block_index));
375 // swap to actual mg solution vectors
376 for (unsigned int relative_level = 0; relative_level < solution_levels.size();
377 ++relative_level)
378 {
379 unsigned int level = finest_level - relative_level;
380 solution_levels[relative_level]
381 .solutions.block(block_index)
382 .swap(temp_mg_solutions[level]);
383 }
384
385 if (!transfer_old_solutions)
386 {
387 return;
388 }
389 // Transfer old solutions
390 for (unsigned int age_index = 0;
391 age_index < solution_levels[0].old_solutions.size();
392 ++age_index)
393 {
394 mg_transfer[block_index].interpolate_to_mg(
395 dof_handler,
396 temp_mg_solutions,
397 solution_levels[0].old_solutions[age_index].block(block_index));
398 for (unsigned int relative_level = 0; relative_level < solution_levels.size();
399 ++relative_level)
400 {
401 unsigned int level = finest_level - relative_level;
402 if (age_index < solution_levels[relative_level].old_solutions.size())
403 {
404 solution_levels[relative_level]
405 .old_solutions[age_index]
406 .block(block_index)
407 .swap(temp_mg_solutions[level]);
408 }
409 }
410 }
411 }
412}
