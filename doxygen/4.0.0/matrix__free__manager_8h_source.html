120template <unsigned int dim, typename number>
121template <unsigned int degree>
122void
123MatrixFreeManager<dim, number>::reinit(
124 const DoFManager<dim, degree> &dof_manager,
125 const ConstraintManager<dim, degree, number> &constraint_manager)
126{
127 using AdditionalData = typename MatrixFree<dim, number>::AdditionalData;
128 // TODO: maybe make the flags determined by user dependencies. (Though I doubt this
129 // is a significant source of runtime)
130 static const AdditionalData additional_data(
131 AdditionalData::TasksParallelScheme::partition_partition,
132 0,
133 dealii::update_values | dealii::update_gradients | dealii::update_hessians |
134 dealii::update_JxW_values | dealii::update_quadrature_points);
135
136 const std::array<dealii::DoFHandler<dim>, 2> &generic_dof_handlers =
137 dof_manager.get_dof_handlers();
138
139 {
140 const std::array<dealii::AffineConstraints<number>, 2> &generic_constraints =
141 constraint_manager.get_generic_constraints();
142
143 // Reinit shared MatrixFree
144 shared_matrix_free.reinit(SystemWide<dim, degree>::mapping,
145 dof_manager.get_field_dof_handlers(),
146 constraint_manager.get_field_constraints(),
147 dealii::QGaussLobatto<1>(degree + 1),
148 // should dim really be 1?
149 additional_data);
150
151 // Reinit generic MatrixFree
152 generic_matrix_free.reinit(SystemWide<dim, degree>::mapping,
153 std::vector<const dealii::DoFHandler<dim> *>(
154 {&generic_dof_handlers[0], &generic_dof_handlers[1]}),
155 std::vector<const dealii::AffineConstraints<number> *>(
156 {&generic_constraints[0], &generic_constraints[1]}),
157 dealii::QGaussLobatto<1>(degree +
158 1)); // should dim really be 1?
159 }
160 const unsigned int num_levels = dof_manager.has_mg() ? dof_manager.num_levels() : 0;
161 shared_matrix_free_levels.resize(num_levels);
162 generic_matrix_free_levels.resize(num_levels);
163 for (unsigned int relative_level = 0; relative_level < num_levels; ++relative_level)
164 {
165 const unsigned int fine_level = num_levels - 1;
166 const unsigned int level = fine_level - relative_level;
167 MatrixFree<dim, number> &shared_mg_matrix_free =
168 shared_matrix_free_levels[relative_level];
169 MatrixFree<dim, number> &generic_mg_matrix_free =
170 generic_matrix_free_levels[relative_level];
171
172 const std::array<dealii::AffineConstraints<number>, 2> &generic_constraints =
173 constraint_manager.get_mg_generic_constraints(relative_level);
174
175 AdditionalData shared_additional_data = additional_data;
176 shared_additional_data.mg_level = level;
177
178 // Reinit shared MatrixFree
179 shared_mg_matrix_free.reinit(
180 SystemWide<dim, degree>::mapping,
181 dof_manager.get_field_dof_handlers(),
182 constraint_manager.get_mg_field_constraints(relative_level),
183 dealii::QGaussLobatto<1>(degree + 1), // should dim really be 1?
184 shared_additional_data);
185
186 AdditionalData generic_additional_data;
187 generic_additional_data.mg_level = level;
188
189 // Reinit generic MatrixFree
190 generic_mg_matrix_free.reinit(
191 SystemWide<dim, degree>::mapping,
192 std::vector<const dealii::DoFHandler<dim> *>(
193 {&generic_dof_handlers[0], &generic_dof_handlers[1]}),
194 std::vector<const dealii::AffineConstraints<number> *>(
195 {&generic_constraints[0], &generic_constraints[1]}),
196 dealii::QGaussLobatto<1>(degree + 1), // should dim really be 1?
197 generic_additional_data);
198 }
199}

245template <unsigned int dim, typename number>
246std::vector<std::shared_ptr<const dealii::Utilities::MPI::Partitioner>>
247MatrixFreeManager<dim, number>::get_block_partitioners(
248 const std::set<unsigned int> &field_indices) const
249{
250 // These partitioners basically just provide the number of elements in a distributed way
251 std::vector<std::shared_ptr<const dealii::Utilities::MPI::Partitioner>> partitioners;
252 partitioners.reserve(field_indices.size());
253 for (unsigned int field_index : field_indices)
254 {
255 partitioners.push_back(shared_matrix_free.get_vector_partitioner(field_index));
256 }
257 return partitioners;
258}
