97template <typename Type, typename OtherType>
98std::pair<OtherType, Type>
99string_to_type_pair_with_delimiters(
100 const std::string &string,
101 const std::unordered_map<std::string, Type> &table,
102 const std::unordered_map<std::string, OtherType> &other_table,
103 const std::pair<char, char> &delimiters = {'(', ')'})
104{
105 // Check to see if the string has any delimiters
106 const auto opening_delimiter_position = string.find(delimiters.first);
107 const auto closing_delimiter_position = string.find(delimiters.second);
108
109 bool has_opening_delimiter = opening_delimiter_position != std::string_view::npos;
110 bool has_closing_delimiter = closing_delimiter_position != std::string_view::npos;
111
112 // Some checks for malformed input
113 AssertThrow(closing_delimiter_position >= opening_delimiter_position,
114 dealii::ExcMessage(
115 "Opening delimiter must precede the closing delimiter. You had " +
116 string));
117 AssertThrow(has_closing_delimiter == has_opening_delimiter,
118 dealii::ExcMessage("You must either have a delimiter pair or not. You seem "
119 "to only have one delimiter. You had " +
120 string));
121
122 // Case 2 - no delimiters
123 if (!has_closing_delimiter && !has_opening_delimiter)
124 {
125 const auto second_type = string_to_type("", table);
126 const auto first_type = string_to_type(string, other_table);
127 return {first_type, second_type};
128 }
129
130 // More checks for malformed input
131 AssertThrow(closing_delimiter_position == string.size() - 1,
132 dealii::ExcMessage(
133 "Closing delimiter must be at the end of the string. You had " + string));
134
135 // Case 1 - delimiters
136 std::string_view input_view {string};
137 std::string_view outer_string = input_view.substr(0, opening_delimiter_position);
138 std::string_view inner_string =
139 input_view.substr(opening_delimiter_position + 1,
140 closing_delimiter_position - opening_delimiter_position - 1);
141
142 AssertThrow(!outer_string.empty() && !inner_string.empty(),
143 dealii::ExcMessage(
144 "Inner and outer map entries must not be empty. You had " + string));
145 // NOTE: This will allocate the string views to strings on the heap. Hopefully this
146 // isn't so bad in performance, but I didn't wanna support heterogeneous lookup tables.
147 // I can't imagine this becoming a performance problem right now since we only deal with
148 // strings in the initialization of simulations; however, if it does, it shouldn't be
149 // two hard to support heterogeneous lookup or a table that uses std::string_view.
150 const auto second_type = string_to_type(std::string(outer_string), table);
151 const auto first_type = string_to_type(std::string(inner_string), other_table);
152 return {first_type, second_type};
153}

176template <unsigned int dim, typename T>
177inline DEAL_II_ALWAYS_INLINE void
178compute_stress(const dealii::Tensor<2, voigt_tensor_size<dim>, T> &elasticity_tensor,
179 const dealii::Tensor<1, voigt_tensor_size<dim>, T> &strain,
180 dealii::Tensor<1, voigt_tensor_size<dim>, T> &stress)

189template <unsigned int dim, typename T>
190inline DEAL_II_ALWAYS_INLINE void
191compute_stress(const dealii::Tensor<2, voigt_tensor_size<dim>, T> &elasticity_tensor,
192 const dealii::Tensor<2, dim, T> &strain,
193 dealii::Tensor<2, dim, T> &stress)
194{
195 dealii::Tensor<1, voigt_tensor_size<dim>, T> sigma;
196 dealii::Tensor<1, voigt_tensor_size<dim>, T> epsilon;
197
198 if constexpr (dim == 3)
199 {
200 const int xx_dir = 0;
201 const int yy_dir = 1;
202 const int zz_dir = 2;
203 const int yz_dir = 3;
204 const int xz_dir = 4;
205 const int xy_dir = 5;
206
207 epsilon[xx_dir] = strain[xx_dir][xx_dir];
208 epsilon[yy_dir] = strain[yy_dir][yy_dir];
209 epsilon[zz_dir] = strain[zz_dir][zz_dir];
210 // In Voigt notation: epsilon are engineering shear strains
211 epsilon[yz_dir] = strain[yy_dir][zz_dir] + strain[zz_dir][yy_dir];
212 epsilon[xz_dir] = strain[xx_dir][zz_dir] + strain[zz_dir][xx_dir];
213 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
214
215 // Multiply elasticity_tensor and epsilon to get sigma
216 sigma = elasticity_tensor * epsilon;
217
218 stress[xx_dir][xx_dir] = sigma[xx_dir];
219 stress[yy_dir][yy_dir] = sigma[yy_dir];
220 stress[zz_dir][zz_dir] = sigma[zz_dir];
221
222 stress[yy_dir][zz_dir] = sigma[yz_dir];
223 stress[zz_dir][yy_dir] = sigma[yz_dir];
224
225 stress[xx_dir][zz_dir] = sigma[xz_dir];
226 stress[zz_dir][xx_dir] = sigma[xz_dir];
227
228 stress[xx_dir][yy_dir] = sigma[xy_dir];
229 stress[yy_dir][xx_dir] = sigma[xy_dir];
230 }
231 else if constexpr (dim == 2)
232 {
233 const int xx_dir = 0;
234 const int yy_dir = 1;
235 const int xy_dir = 2;
236
237 epsilon[xx_dir] = strain[xx_dir][xx_dir];
238 epsilon[yy_dir] = strain[yy_dir][yy_dir];
239 // In Voigt notation: epsilon are engineering shear strains
240 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
241
242 // Multiply elasticity_tensor and epsilon to get sigma
243 sigma = elasticity_tensor * epsilon;
244
245 stress[xx_dir][xx_dir] = sigma[xx_dir];
246 stress[yy_dir][yy_dir] = sigma[yy_dir];
247 stress[xx_dir][yy_dir] = sigma[xy_dir];
248 stress[yy_dir][xx_dir] = sigma[xy_dir];
249 }
250 else
251 {
252 const int xx_dir = 0;
253
254 stress[xx_dir][xx_dir] = elasticity_tensor[xx_dir][xx_dir] * strain[xx_dir][xx_dir];
255 }
256}

281inline std::string
282eval_flags_to_string(dealii::EvaluationFlags::EvaluationFlags flag)
283{
284 std::string result;
285
286 if ((flag & dealii::EvaluationFlags::EvaluationFlags::values) != 0U)
287 {
288 result += "values";
289 }
290 if ((flag & dealii::EvaluationFlags::EvaluationFlags::gradients) != 0U)
291 {
292 if (!result.empty())
293 {
294 result += " | ";
295 }
296 result += "gradients";
297 }
298 if ((flag & dealii::EvaluationFlags::EvaluationFlags::hessians) != 0U)
299 {
300 if (!result.empty())
301 {
302 result += " | ";
303 }
304 result += "hessians";
305 }
306
307 if (result.empty())
308 {
309 return "nothing";
310 }
311
312 return result;
313}

315template <unsigned int dim, typename number>
316inline DEAL_II_ALWAYS_INLINE std::vector<number>
317dealii_point_to_vector(const dealii::Point<dim, number> &point)
318{
319 static_assert(dim < 4, "We only allow 3 space dimensions");
320
321 std::vector<number> vec(3, 0.0);
322
323 // NOLINTBEGIN(cppcoreguidelines-pro-bounds-constant-array-index)
324
325 for (unsigned int i = 0; i < dim; ++i)
326 {
327 vec[i] = point[i];
328 }
329
330 // NOLINTEND(cppcoreguidelines-pro-bounds-constant-array-index)
331
332 return vec;
333}
