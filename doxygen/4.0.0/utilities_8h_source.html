111template <typename Type, typename OtherType>
112std::pair<OtherType, Type>
113string_to_type_pair_with_delimiters(
114 const std::string &string,
115 const std::unordered_map<std::string, Type> &table,
116 const std::unordered_map<std::string, OtherType> &other_table,
117 const std::pair<char, char> &delimiters = {'(', ')'})
118{
119 // Check to see if the string has any delimiters
120 const auto opening_delimiter_position = string.find(delimiters.first);
121 const auto closing_delimiter_position = string.find(delimiters.second);
122
123 bool has_opening_delimiter = opening_delimiter_position != std::string_view::npos;
124 bool has_closing_delimiter = closing_delimiter_position != std::string_view::npos;
125
126 // Some checks for malformed input
127 AssertThrow(closing_delimiter_position >= opening_delimiter_position,
128 dealii::ExcMessage(
129 "Opening delimiter must precede the closing delimiter. You had " +
130 string));
131 AssertThrow(has_closing_delimiter == has_opening_delimiter,
132 dealii::ExcMessage("You must either have a delimiter pair or not. You seem "
133 "to only have one delimiter. You had " +
134 string));
135
136 // Case 2 - no delimiters
137 if (!has_closing_delimiter && !has_opening_delimiter)
138 {
139 const auto second_type = string_to_type("", table);
140 const auto first_type = string_to_type(string, other_table);
141 return {first_type, second_type};
142 }
143
144 // More checks for malformed input
145 AssertThrow(closing_delimiter_position == string.size() - 1,
146 dealii::ExcMessage(
147 "Closing delimiter must be at the end of the string. You had " + string));
148
149 // Case 1 - delimiters
150 std::string_view input_view {string};
151 std::string_view outer_string = input_view.substr(0, opening_delimiter_position);
152 std::string_view inner_string =
153 input_view.substr(opening_delimiter_position + 1,
154 closing_delimiter_position - opening_delimiter_position - 1);
155
156 AssertThrow(!outer_string.empty() && !inner_string.empty(),
157 dealii::ExcMessage(
158 "Inner and outer map entries must not be empty. You had " + string));
159 // NOTE: This will allocate the string views to strings on the heap. Hopefully this
160 // isn't so bad in performance, but I didn't wanna support heterogeneous lookup tables.
161 // I can't imagine this becoming a performance problem right now since we only deal with
162 // strings in the initialization of simulations; however, if it does, it shouldn't be
163 // two hard to support heterogeneous lookup or a table that uses std::string_view.
164 const auto second_type = string_to_type(std::string(outer_string), table);
165 const auto first_type = string_to_type(std::string(inner_string), other_table);
166 return {first_type, second_type};
167}

203template <unsigned int dim, typename T>
204inline DEAL_II_ALWAYS_INLINE void
205compute_stress(const dealii::Tensor<2, voigt_tensor_size<dim>, T> &elasticity_tensor,
206 const dealii::Tensor<2, dim, T> &strain,
207 dealii::Tensor<2, dim, T> &stress)
208{
209 dealii::Tensor<1, voigt_tensor_size<dim>, T> sigma;
210 dealii::Tensor<1, voigt_tensor_size<dim>, T> epsilon;
211
212 if constexpr (dim == 3)
213 {
214 const int xx_dir = 0;
215 const int yy_dir = 1;
216 const int zz_dir = 2;
217 const int yz_dir = 3;
218 const int xz_dir = 4;
219 const int xy_dir = 5;
220
221 epsilon[xx_dir] = strain[xx_dir][xx_dir];
222 epsilon[yy_dir] = strain[yy_dir][yy_dir];
223 epsilon[zz_dir] = strain[zz_dir][zz_dir];
224 // In Voigt notation: epsilon are engineering shear strains
225 epsilon[yz_dir] = strain[yy_dir][zz_dir] + strain[zz_dir][yy_dir];
226 epsilon[xz_dir] = strain[xx_dir][zz_dir] + strain[zz_dir][xx_dir];
227 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
228
229 // Multiply elasticity_tensor and epsilon to get sigma
230 sigma = elasticity_tensor * epsilon;
231
232 stress[xx_dir][xx_dir] = sigma[xx_dir];
233 stress[yy_dir][yy_dir] = sigma[yy_dir];
234 stress[zz_dir][zz_dir] = sigma[zz_dir];
235
236 stress[yy_dir][zz_dir] = sigma[yz_dir];
237 stress[zz_dir][yy_dir] = sigma[yz_dir];
238
239 stress[xx_dir][zz_dir] = sigma[xz_dir];
240 stress[zz_dir][xx_dir] = sigma[xz_dir];
241
242 stress[xx_dir][yy_dir] = sigma[xy_dir];
243 stress[yy_dir][xx_dir] = sigma[xy_dir];
244 }
245 else if constexpr (dim == 2)
246 {
247 const int xx_dir = 0;
248 const int yy_dir = 1;
249 const int xy_dir = 2;
250
251 epsilon[xx_dir] = strain[xx_dir][xx_dir];
252 epsilon[yy_dir] = strain[yy_dir][yy_dir];
253 // In Voigt notation: epsilon are engineering shear strains
254 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
255
256 // Multiply elasticity_tensor and epsilon to get sigma
257 sigma = elasticity_tensor * epsilon;
258
259 stress[xx_dir][xx_dir] = sigma[xx_dir];
260 stress[yy_dir][yy_dir] = sigma[yy_dir];
261 stress[xx_dir][yy_dir] = sigma[xy_dir];
262 stress[yy_dir][xx_dir] = sigma[xy_dir];
263 }
264 else
265 {
266 const int xx_dir = 0;
267
268 stress[xx_dir][xx_dir] = elasticity_tensor[xx_dir][xx_dir] * strain[xx_dir][xx_dir];
269 }
270}

279template <unsigned int dim, unsigned int degree, typename T>
280inline DEAL_II_ALWAYS_INLINE T
281stabilization_parameter(const T &timestep,
282 const T &element_volume,
283 const dealii::Tensor<1, dim, T> &velocity,
284 const T &kinematic_viscosity)
285{
286 using dealii::Utilities::fixed_power;
287 using std::cbrt;
288 using std::sqrt;
289
290 constexpr auto degree_modifier = T(1) / T(degree);
291 constexpr auto inv_pi = T(1) / T(M_PI);
292 const auto velocity_l2_norm = velocity.norm_square() + T(1e-12);
293
294 auto element_size = T(0);
295 if constexpr (dim == 1)
296 {
297 element_size = element_volume;
298 }
299 else if constexpr (dim == 2)
300 {
301 element_size = T(2) * sqrt(element_volume * inv_pi);
302 }
303 else if constexpr (dim == 3)
304 {
305 element_size = cbrt(T(6) * element_volume * inv_pi);
306 }
307 else
308 {
309 Assert(false, dealii::ExcMessage("Invalid dimension for the stabilization term."));
310 }
311 element_size *= degree_modifier;
312 const auto element_size_2 = element_size * element_size;
313 const auto element_size_2_inv = T(1) / element_size_2;
314 const auto element_size_4_inv = element_size_2_inv * element_size_2_inv;
315
316 const T term_time = fixed_power<2>(T(2) / timestep);
317 const T term_adv = T(4) * velocity_l2_norm * element_size_2_inv;
318 const T term_diff = T(144) * fixed_power<2>(kinematic_viscosity) * element_size_4_inv;
319
320 return T(1) / sqrt(term_time + term_adv + term_diff);
321}

499inline std::string
500eval_flags_to_string(dealii::EvaluationFlags::EvaluationFlags flag)
501{
502 std::string result;
503
504 if ((flag & dealii::EvaluationFlags::EvaluationFlags::values) != 0U)
505 {
506 result += "values";
507 }
508 if ((flag & dealii::EvaluationFlags::EvaluationFlags::gradients) != 0U)
509 {
510 if (!result.empty())
511 {
512 result += " | ";
513 }
514 result += "gradients";
515 }
516 if ((flag & dealii::EvaluationFlags::EvaluationFlags::hessians) != 0U)
517 {
518 if (!result.empty())
519 {
520 result += " | ";
521 }
522 result += "hessians";
523 }
524
525 if (result.empty())
526 {
527 return "nothing";
528 }
529
530 return result;
