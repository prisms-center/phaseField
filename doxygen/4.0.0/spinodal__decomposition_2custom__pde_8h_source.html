55 void
56 compute_rhs([[maybe_unused]] FieldContainer<dim, degree, number> &variable_list,
57 [[maybe_unused]] const SimulationTimer &sim_timer,
58 [[maybe_unused]] unsigned int solve_block_id) const override
59 {
60 if (solve_block_id == 0) // c
61 {
62 ScalarValue c = variable_list.template get_value<Scalar, OldOne>(0);
63 ScalarGrad mux = variable_list.template get_gradient<Scalar, OldOne>(1);
64
65 ScalarValue eq_c = c;
66 ScalarGrad eqx_c = -McV * sim_timer.get_timestep() * mux;
67
68 variable_list.set_value_term(0, eq_c);
69 variable_list.set_gradient_term(0, eqx_c);
70 }
71 else if (solve_block_id == 1) // mu
72 {
73 ScalarValue c = variable_list.template get_value<Scalar, Current>(0);
74 ScalarGrad cx = variable_list.template get_gradient<Scalar, Current>(0);
75
76 ScalarValue fcV = WcV * c * (c - 1.0) * (c - 0.5);
77
78 ScalarValue eq_mu = fcV;
79 ScalarGrad eqx_mu = KcV * cx;
80
81 variable_list.set_value_term(1, eq_mu);
82 variable_list.set_gradient_term(1, eqx_mu);
83 }
84 else if (solve_block_id == 2) // pp
85 {
86 ScalarValue c = variable_list.template get_value<Scalar, Current>(0);
87 ScalarGrad cx = variable_list.template get_gradient<Scalar, Current>(0);
88
89 ScalarValue f_tot = 0.0;
90 ScalarValue f_chem = c * c * c * c - 2.0 * c * c * c + c * c;
91 ScalarValue f_grad = 0.0;
92
93 for (unsigned int i = 0; i < dim; i++)
94 {
95 f_grad += 0.5 * KcV * cx[i] * cx[i];
96 }
97 f_tot = f_chem + f_grad;
98 variable_list.set_value_term(2, f_tot);
99 }
100 }

104 ScalarValue WcV;
105 number c0;
106 number icamplitude;
107 mutable std::uniform_real_distribution<number> dist;
108};
