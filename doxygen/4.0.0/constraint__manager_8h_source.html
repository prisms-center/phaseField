242template <unsigned int dim, unsigned int degree, typename number>
243inline const std::array<dealii::ComponentMask, dim>
244 ConstraintManager<dim, degree, number>::vector_component_mask = []()
245{
246 std::array<dealii::ComponentMask, dim> masks {};
247 for (unsigned int i = 0; i < dim; ++i)
248 {
249 dealii::ComponentMask temp_mask(dim, false);
250 temp_mask.set(i, true);
251 masks.at(i) = temp_mask;
252 }
253 return masks;
254}();
