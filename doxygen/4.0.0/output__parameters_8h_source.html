60 void
61 add_equal_spacing_outputs(unsigned int num_outputs, unsigned int num_increments)
62 {
63 if (num_increments == 0)
64 {
65 output_list.insert(0);
66 return;
67 }
68 if (!num_outputs)
69 {
70 return;
71 }
72 unsigned int period = std::max(1U, num_increments / num_outputs);
73 for (unsigned int output = 0; output <= num_increments; output += period)
74 {
75 output_list.insert(output);
76 }
77 }
78
82 void
83 add_log_spacing_outputs(unsigned int num_outputs, unsigned int num_increments)
84 {
85 if (!num_outputs)
86 {
87 return;
88 }
89 for (unsigned int output = 1; output <= num_outputs; output++)
90 {
91 output_list.insert(static_cast<unsigned int>(std::round(
92 std::pow(double(num_increments), double(output) / double(num_outputs)))));
93 }
94 }
95
99 void
100 add_n_per_decade_outputs(unsigned int num_outputs, unsigned int num_increments)
101 {
102 if (!num_outputs)
103 {
104 return;
105 }
106 AssertThrow(num_increments > 1,
107 dealii::ExcMessage("For n per decaded spaced outputs, the number of "
108 "increments must be greater than 1."));
109
110 output_list.insert(0);
111 output_list.insert(1);
112 for (unsigned int iteration = 2; iteration <= num_increments; iteration++)
113 {
114 const auto decade = static_cast<unsigned int>(std::ceil(std::log10(iteration)));
115 const auto step_size =
116 static_cast<unsigned int>(std::pow(10, decade) / num_outputs);
117 if (iteration % step_size == 0)
118 {
119 output_list.insert(iteration);
120 }
121 }
122 }

267inline void
268OutputParameters::assign_parameters(dealii::ParameterHandler &parameter_handler,
269 const TemporalDiscretization &temporal_discretization)
270{
271 parameter_handler.enter_subsection("output");
272 {
273 directory = parameter_handler.get("directory");
274 file_name = parameter_handler.get("file name");
275 file_type = parameter_handler.get("file type");
276 patch_subdivisions =
277 static_cast<unsigned int>(parameter_handler.get_integer("subdivisions"));
278
279 std::string condition = parameter_handler.get("condition");
280 unsigned int num_outputs =
281 static_cast<unsigned int>(parameter_handler.get_integer("number"));
282 unsigned int num_increments = temporal_discretization.num_increments;
283 if (condition == "EQUAL_SPACING")
284 {
285 add_equal_spacing_outputs(num_outputs, num_increments);
286 }
287 else if (condition == "LOG_SPACING")
288 {
289 add_log_spacing_outputs(num_outputs, num_increments);
290 }
291 else if (condition == "N_PER_DECADE")
292 {
293 add_n_per_decade_outputs(num_outputs, num_increments);
294 }
295 add_output_list(dealii::Utilities::string_to_int(
296 dealii::Utilities::split_string_list(parameter_handler.get("list"))));
297
298 print_output_period =
299 static_cast<unsigned int>(parameter_handler.get_integer("print step period"));
300 print_timing_with_output =
301 parameter_handler.get_bool("timing information with output");
302 }
303 parameter_handler.leave_subsection();
304}
