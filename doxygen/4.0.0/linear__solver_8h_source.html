75 void

77 {

123 void
124 solve_impl() override
125 {
126 // Zero out the ghosts
127 Timer::start_section("Zero ghosts");
128 solutions.zero_out_ghosts(0);
129 Timer::end_section("Zero ghosts");
130
131 // Set up rhs vector
132 rhs_operator.compute_operator(rhs_vector);
133
134 // Note 1. Use the previous result of the linear solve without nonzero dirichlet
135 // as the initial guess in the next increment. See Note 2. `inhomogeneous_rhs` is
136 // not actually what it is being used as here, we just don't want to allocate a
137 // whole new vector for this purpose
138 solutions.get_solution_full_vector(0).swap(inhomogeneous_rhs);
139 // Get the homogeneous rhs
140 lhs_operator.read_plain = true;
141 lhs_operator.compute_operator(inhomogeneous_rhs, inhomogeneous_values);
142 lhs_operator.read_plain = false;
143 rhs_vector -= inhomogeneous_rhs;
144
145 // Linear solve
146 do_linear_solve(rhs_vector, lhs_operator, solutions.get_solution_full_vector(0));
147
148 // Note 2. Make a copy of the solution to use as the initial guess in the next
149 // increment. See Note 1. `inhomogeneous_rhs` is not actually what it is being
150 // used as here, we just don't want to allocate a whole new vector for this
151 // purpose
152 inhomogeneous_rhs = solutions.get_solution_full_vector(0);
153 // Add back in nonzero dirichlet conditions
154 solutions.get_solution_full_vector(0) += inhomogeneous_values;
155
156 // Apply constraints
157 solutions.apply_constraints(0);
158
159 // Update the ghosts
160 Timer::start_section("Update ghosts");
161 solutions.update_ghosts(0);
162 Timer::end_section("Update ghosts");
163 }

165 int
166 do_linear_solve(BlockVector<number> &b_vector,
167 MFOperator<dim, degree, number> &lhs_matrix,
168 BlockVector<number> &x_vector)
169 {
170 // Linear solve
171 try
172 {
173 if (lin_params.preconditioner == None)
174 {
175 lin_solver.solve(lhs_matrix,
176 x_vector,
177 b_vector,
178 dealii::PreconditionIdentity());
179 }
180 if (lin_params.preconditioner == Chebyshev)
181 {
182 lhs_matrix.reinit_matrix_diagonal(x_vector);
183 lhs_matrix.eval_matrix_diagonal();
184
185 lin_solver.solve(lhs_matrix, x_vector, b_vector, precond_chebyshev);
186 }
187 }
188 catch (...) // TODO: more specific catch so that we dont ignore
189 // non-convergence-related errors.
190 {
191 ConditionalOStreams::pout_base()
192 << "[Increment " << solve_context->get_simulation_timer().get_increment()
193 << "] "
194 << "Warning: linear solver did not converge as per set tolerances before "
195 << lin_params.max_iterations << " iterations.\n";
196 }
197 if (solve_context->get_user_inputs().output_parameters.should_output(
198 solve_context->get_simulation_timer().get_increment()))
199 {
200 ConditionalOStreams::pout_summary()
201 << " Linear solve final residual : "
202 << linear_solver_control.last_value() / normalization_value()
203 << " Linear steps: " << linear_solver_control.last_step() << "\n"
204 << std::flush;
205 }
206 return linear_solver_control.last_step();
207 }

218 double
219 normalization_value()
220 {
221 SolverToleranceType type = lin_params.tolerance_type;
222 double value = 1.0;
223 if (type == RMSEPerField || type == RMSETotal)
224 {
225 value *= std::sqrt(solve_context->get_triangulation_manager().get_volume());
226 }
227 if (type == RMSEPerField || type == IntegratedPerField)
228 {
229 value *= std::sqrt(double(solve_block.field_indices.size()));
230 }
231 return value;
232 }

234 void
235 initialize_preconditioner()
236 {
237 if (lin_params.preconditioner == None)
238 {
239 void(0); // do nothing
240 }
241 if (lin_params.preconditioner == Chebyshev)
242 {
243 initialize_chebyshev();
244 }
245 }

247 void
248 initialize_chebyshev()
249 {
250 precond_data.degree = lin_params.smoother_degree;
251 precond_data.smoothing_range = lin_params.smoothing_range; // ≈ λ_min / λ_max
252 precond_data.eig_cg_n_iterations =
253 lin_params.eig_cg_n_iterations; // estimate λ_max via CG
254 lhs_operator.reinit_matrix_diagonal(solutions.get_solution_full_vector(0));
255 precond_data.preconditioner = lhs_operator.get_matrix_diagonal_inverse();
256
257 precond_chebyshev.initialize(lhs_operator, precond_data);
258 }
