47class MGContext
48{
49public:
50 using PreconditionChebyshev =
51 dealii::PreconditionChebyshev<MFOperator<dim, degree, number>,
52 BlockVector<number>,
53 dealii::DiagonalMatrix<BlockVector<number>>>;
54 using SmootherPrecond = PreconditionChebyshev;
55 using Smoother = dealii::MGSmootherPrecondition<MFOperator<dim, degree, number>,
56 SmootherPrecond,
57 BlockVector<number>>;
58 using MGTransferType =
59 dealii::MGTransferBlockGlobalCoarsening<dim, BlockVector<number>>;
60 // dc = default constructible, ndc = not default constructible
61 dealii::MGLevelObject<MFOperator<dim, degree, number>> mg_lhs_operators; // dc
62 std::vector<dealii::MGConstrainedDoFs> mg_constraints; // dc
63 MGTransferType mg_transfer; // ndc
64 Smoother mg_smoother; // dc
65 dealii::MGCoarseGridApplySmoother<BlockVector<number>> mg_coarse_solver; // dc
66 dealii::mg::Matrix<BlockVector<number>> mg_matrix; // dc
67 dealii::Multigrid<BlockVector<number>> multigrid; // ndc
68 dealii::PreconditionMG<dim, BlockVector<number>, MGTransferType>
69 multigrid_preconditioner; // ndc
70
71 MGContext()
72 : mg_transfer(mg_constraints)
73 , multigrid(mg_matrix,
74 mg_coarse_solver,
75 mg_transfer,
76 mg_smoother,
77 mg_smoother,
78 0,
79 0,
80 dealii::Multigrid<BlockVector<number>>::Cycle::v_cycle)
81 , multigrid_preconditioner(std::vector<const dealii::DoFHandler<dim> *>(),

86 void
87 init(unsigned int min_level,
88 unsigned int max_level,
89 const SolveBlock &solve_block,
90 const LinearSolverParameters &lin_params,
91 const SolveContext<dim, degree, number> &solve_context,
92 const GroupSolutionHandler<dim, number> &solutions)
93 {
94 // 1. Level operators
95 mg_lhs_operators =
96 dealii::MGLevelObject<MFOperator<dim, degree, number>>(min_level, max_level);
97 for (unsigned level = min_level; level <= max_level; ++level)
98 {
99 const unsigned int relative_level = max_level - level;
100 mg_lhs_operators[level].init(solve_context.get_pde_operator(),
101 &PDEOperatorBase<dim, degree, number>::compute_lhs,
102 solve_context.get_field_attributes(),
103 solve_context.get_solution_indexer(),
104 solve_context.get_matrix_free_manager(),
105 solve_context.get_simulation_timer(),
106 solve_block,
107 solve_block.dependencies_lhs);
108 mg_lhs_operators[level].set_scaling_diagonal(
109 lin_params.tolerance_type != AbsoluteResidual,
110 solve_context.get_invm_manager().get_invm_sqrt(
111 solve_context.get_field_attributes(),
112 solve_block.field_indices,
113 relative_level));
114 mg_lhs_operators[level].set_relative_level(relative_level);
115 }
116
117 // 2. MG Constraints (homogeneous)
118 mg_constraints = make_mg_constraints(solve_block, solve_context, solutions);
119
120 // 3. MG Transfer
121 mg_transfer.initialize_constraints(mg_constraints);
122 mg_transfer.build(
123 solve_context.get_dof_manager().get_block_dof_handlers(solve_block.field_indices));
124
125 // 4. MG Smoother
126 dealii::MGLevelObject<typename SmootherPrecond::AdditionalData> smoother_data(
127 min_level,
128 max_level);
129 const auto &chebyshev_params = lin_params.chebyshev_parameters;
130 for (unsigned int level = min_level; level <= max_level; ++level)
131 {
132 unsigned int relative_level = level - min_level;
133 smoother_data[level].smoothing_range = chebyshev_params.smoothing_range;
134 smoother_data[level].degree = chebyshev_params.degree;
135 smoother_data[level].eig_cg_n_iterations = chebyshev_params.eig_cg_n_iterations;
136 smoother_data[level].constraints.close(); // todo
137
138 mg_lhs_operators[level].reinit_matrix_diagonal();
139 smoother_data[level].preconditioner =
140 mg_lhs_operators[level].get_matrix_diagonal_inverse();
141 }
142 mg_smoother.initialize(mg_lhs_operators, smoother_data);
143
144 // 5. Coarse grid solver
145 mg_coarse_solver.initialize(mg_smoother);
146
147 // 6. Multigrid object
148 mg_matrix = dealii::mg::Matrix<BlockVector<number>>(mg_lhs_operators);
149 multigrid = dealii::Multigrid<BlockVector<number>>(
150 mg_matrix,
151 mg_coarse_solver,
152 mg_transfer,
153 mg_smoother,
154 mg_smoother,
155 min_level,
156 max_level,
157 dealii::Multigrid<BlockVector<number>>::Cycle::v_cycle);
158
159 // multigrid_preconditioner =
160 // dealii::PreconditionMG<dim, BlockVector<number>, MGTransferType>(
161 // solve_context.get_dof_manager()
162 // .get_block_dof_handlers(solve_block.field_indices),
163 // multigrid,
164 // mg_transfer);
165 }

170 static std::vector<dealii::MGConstrainedDoFs>
171 make_mg_constraints(const SolveBlock &solve_block,
172 const SolveContext<dim, degree, number> &solve_context,
173 const GroupSolutionHandler<dim, number> &solutions)
174 {
175 std::vector<dealii::MGConstrainedDoFs> mg_constraints(
176 solve_block.field_indices.size());
177 for (unsigned int block_index = 0; block_index < solve_block.field_indices.size();
178 block_index++)
179 {
180 unsigned int field_index = solutions.get_block_to_global_index()[block_index];
181 mg_constraints[block_index].initialize(
182 solve_context.get_dof_manager().get_field_dof_handler(field_index));
183 }
184 const std::vector<FieldAttributes> &field_attributes =
185 solve_context.get_field_attributes();
186 for (unsigned int block_index = 0; block_index < solve_block.field_indices.size();
187 block_index++)
188 {
189 unsigned int field_index = solutions.get_block_to_global_index()[block_index];
190 const FieldAttributes &field = field_attributes[field_index];
191 unsigned int num_comps = (field.field_type == TensorRank::Vector) ? dim : 1;
192 for (unsigned int comp = 0; comp < num_comps; comp++)
193 {
194 std::set<unsigned int> constrained_boundary_ids;
195 const ComponentConditions &comp_bcs =
196 field_attributes[field_index].boundary_conditions.component_constraints.at(
197 comp);
198 for (const auto &[boundary_id, boundary_type] : comp_bcs)
199 {
200 if (boundary_type == Condition::Dirichlet)
201 {
202 constrained_boundary_ids.insert(boundary_id);
203 }
204 }
205 mg_constraints[block_index].make_zero_boundary_constraints(
206 solve_context.get_dof_manager().get_field_dof_handler(field_index),
207 constrained_boundary_ids,
208 num_comps == 1
209 ? ConstraintManager<dim, degree, number>::scalar_empty_mask
210 : ConstraintManager<dim, degree, number>::vector_component_mask.at(comp));
211 }
212 }
213 return mg_constraints;
214 }

306 void
307 solve_impl() override
308 {
309 // Zero out the ghosts
310 Timer::start_section("Zero ghosts");
311 solutions.zero_out_ghosts();
312 Timer::end_section("Zero ghosts");
313
314 // Set up rhs vector
315 rhs_operator.compute_operator(rhs_vector);
316
317 // Set inhomogeneous Dirichlet values. TODO: only update if time-dependent
318 solutions.apply_constraints(inhomogeneous_values);
319
320 // Note 1. Use the previous result of the linear solve without nonzero dirichlet
321 // as the initial guess in the next increment. See Note 2. `inhomogeneous_rhs` is
322 // not actually what it is being used as here, we just don't want to allocate a
323 // whole new vector for this purpose
324 solutions.get_solution_full_vector().swap(inhomogeneous_rhs);
325 // Get the homogeneous rhs
326 lhs_operator.read_plain = true;
327 lhs_operator.compute_operator(inhomogeneous_rhs, inhomogeneous_values);
328 lhs_operator.read_plain = false;
329 rhs_vector -= inhomogeneous_rhs;
330
331 // Linear solve
332 do_linear_solve(rhs_vector, lhs_operator, solutions.get_solution_full_vector());
333
334 // Note 2. Make a copy of the solution to use as the initial guess in the next
335 // increment. See Note 1. `inhomogeneous_rhs` is not actually what it is being
336 // used as here, we just don't want to allocate a whole new vector for this
337 // purpose
338 inhomogeneous_rhs = solutions.get_solution_full_vector();
339 // Add back in nonzero dirichlet conditions
340 solutions.get_solution_full_vector() += inhomogeneous_values;
341
342 // Apply constraints
343 solutions.apply_constraints();
344
345 // Update the ghosts
346 Timer::start_section("Update ghosts");
347 solutions.update_ghosts();
348 Timer::end_section("Update ghosts");
349 }

351 int
352 do_linear_solve(BlockVector<number> &b_vector,
353 MFOperator<dim, degree, number> &lhs_matrix,
354 BlockVector<number> &x_vector)
355 {
356 // Linear solve
357 try
358 {
359 if (lin_params().preconditioner == None)
360 {
361 lin_solver.solve(lhs_matrix,
362 x_vector,
363 b_vector,
364 dealii::PreconditionIdentity());
365 }
366 else if (lin_params().preconditioner == Chebyshev)
367 {
368 lhs_matrix.reinit_matrix_diagonal();
369 lhs_matrix.eval_matrix_diagonal();
370
371 lin_solver.solve(lhs_matrix, x_vector, b_vector, precond_chebyshev);
372 }
373 else if (lin_params().preconditioner == GMG)
374 {
375 // TODO: recalculate diagonals
376 auto &lhs_ops = mg_context.mg_lhs_operators;
377 for (unsigned int level = lhs_ops.min_level(); level <= lhs_ops.max_level();
378 ++level)
379 {
380 lhs_ops[level].reinit_matrix_diagonal(); // todo
381 lhs_ops[level].eval_matrix_diagonal();
382 }
383 lin_solver.solve(lhs_operator, x_vector, b_vector, *multigrid_preconditioner);
384 }
385 }
386 catch (dealii::SolverControl::NoConvergence &exc)
387 {
388 ConditionalOStreams::pout_base()
389 << "[Increment " << solve_context->get_simulation_timer().get_increment()
390 << "] "
391 << "Warning: linear solver did not converge as per set tolerances before "
392 << lin_params().max_iterations << " iterations.\n";
393 }
394 if (solve_context->get_user_inputs().output_parameters.should_output(
395 solve_context->get_simulation_timer().get_increment()))
396 {
397 ConditionalOStreams::pout_summary()
398 << " Linear solve final residual : "
399 << linear_solver_control.last_value() / normalization_value()
400 << " Linear steps: " << linear_solver_control.last_step() << "\n"
401 << std::flush;
402 }
403 return linear_solver_control.last_step();
404 }

415 double
416 normalization_value()
417 {
418 SolverToleranceType type = lin_params().tolerance_type;
419 double value = 1.0;
420 if (type == RMSEPerField || type == RMSETotal)
421 {
422 value *= std::sqrt(solve_context->get_triangulation_manager().get_volume());
423 }
424 if (type == RMSEPerField || type == IntegratedPerField)
425 {
426 value *= std::sqrt(double(solve_block.field_indices.size()));
427 }
428 return value;

462 void
463 initialize_solver()
464 {
465 const auto &richardson_parameters = lin_params().richardson_parameters;
466 const auto &bicgstab_parameters = lin_params().bicgstab_parameters;
467 const auto &gmres_parameters = lin_params().gmres_parameters;
468 const typename dealii::SolverRichardson<BlockVector<number>>::AdditionalData
469 local_richardson_parameters(richardson_parameters.omega,
470 richardson_parameters.use_preconditioned_residual);
471 const typename dealii::SolverBicgstab<BlockVector<number>>::AdditionalData
472 local_bicgstab_parameters(bicgstab_parameters.exact_residual,
473 bicgstab_parameters.breakdown);
474 const typename dealii::SolverGMRES<BlockVector<number>>::AdditionalData
475 local_gmres_parameters(gmres_parameters.max_basis_size,
476 gmres_parameters.right_preconditioning,
477 gmres_parameters.use_default_residual,
478 gmres_parameters.force_re_orthogonalization,
479 gmres_parameters.batched_mode,
480 gmres_parameters.orthogonalization_strategy);
481 const typename dealii::SolverFGMRES<BlockVector<number>>::AdditionalData
482 local_fgmres_parameters(gmres_parameters.max_basis_size,
483 gmres_parameters.orthogonalization_strategy);
484
485 lin_solver.set_data(local_richardson_parameters);
486 lin_solver.set_data(local_bicgstab_parameters);
487 lin_solver.set_data(local_gmres_parameters);
488 lin_solver.set_data(local_fgmres_parameters);
489
490 lin_solver.select(lin_params().solver_type);
491 lin_solver.set_control(linear_solver_control);
492 }
