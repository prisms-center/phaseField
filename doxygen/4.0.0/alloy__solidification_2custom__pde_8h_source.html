43 void
44 set_initial_condition([[maybe_unused]] const unsigned int &index,
45 [[maybe_unused]] const unsigned int &component,
46 [[maybe_unused]] const dealii::Point<dim> &point,
47 [[maybe_unused]] number &scalar_value,
48 [[maybe_unused]] number &vector_component_value) const override
49 {
50 const dealii::Tensor<1, dim> &mesh_size =
51 get_user_inputs().spatial_discretization.rectangular_mesh.size;
52
53 if (index == 0)
54 {
55 // For the concentration field, we just set the initial condition
56 // to have uniform undercooling everywhere.
57 scalar_value = U0;
58 }
59 else if (index == 1)
60 {
61 const std::array<double, 3> center = {
62 {0.0, 0.0, 0.0}
63 };
64 const double rad = 5.0;
65 // For the order parameter, we just place a small seed. Note that
66 // the order parameter ranges from -1 to 1 in this model.
67 double dist = 0.0;
68 for (unsigned int dir = 0; dir < dim; dir++)
69 {
70 const number comp_diff = point[dir] - center[dir] * mesh_size[dir];
71 dist += comp_diff * comp_diff;
72 }
73 dist = std::sqrt(dist);
74
75 scalar_value = -std::tanh((dist - rad) / std::sqrt(2));
76 }
77 }

79 void
80 compute_rhs(FieldContainer<dim, degree, number> &variable_list,
81 const SimulationTimer &sim_timer,
82 unsigned int solve_block_id) const override
83 {
84 const double dt = sim_timer.get_timestep();
85 if (solve_block_id == 0) // explicit U and phi
86 {
87 // The dimensionless solute supersaturation and its derivatives
88 ScalarValue U = variable_list.template get_value<Scalar, OldOne>(0);
89 ScalarGrad Ux = variable_list.template get_gradient<Scalar, OldOne>(0);
90
91 // The order parameter and its derivatives
92 ScalarValue phi = variable_list.template get_value<Scalar, OldOne>(1);
93 ScalarGrad phix = variable_list.template get_gradient<Scalar, OldOne>(1);
94
95 // The auxiliary parameter and its derivatives
96 ScalarValue xi = variable_list.template get_value<Scalar, OldOne>(2);
97
98 // --- Setting the expressions for the terms in the governing equations ---
99
100 // Calculation of interface normal vector
101 ScalarValue normgradn = std::sqrt(phix.norm_square());
102 ScalarGrad normal = phix / (normgradn + regval);
103
104 // The cosine of theta
105 ScalarValue cth = normal[0];
106 // The sine of theta
107 ScalarValue sth = normal[1];
108 // The cosine of 4 theta
109 ScalarValue c4th = (sth * sth * sth * sth) + (cth * cth * cth * cth) -
110 (6.0 * sth * sth * cth * cth);
111
112 // Anisotropic term
113 ScalarValue a_n = 1.0 + (epsilon * c4th);
114
115 // coefficient before phi
116 ScalarValue tau_phi = (1.0 + (1.0 - k) * U) * a_n * a_n;
117
118 // coefficient before U
119 ScalarValue tau_U = (((1.0 + k) / 2.0) - ((1.0 - k) * phi / 2.0));
120
121 // Antitrapping term
122 ScalarGrad j_at;
123 j_at = (1.0 / (2.0 * std::sqrt(2.0))) * (1.0 + (1.0 - k) * U) * (xi / tau_phi) *
124 normal;
125
126 // grad_phi and grad_U dot product term
127 ScalarValue val_term1 = dt * (1.0 + (1.0 - k) * U) * xi / (2.0 * tau_phi * tau_U);
128 ScalarValue val_term2 =
129 dt * ((1.0 - k) / 2.0) / (tau_U * tau_U) *
130 (phix[0] * (Dtilde * ((1.0 - phi) / 2.0) * Ux[0] + j_at[0]) +
131 phix[1] * (Dtilde * ((1.0 - phi) / 2.0) * Ux[1] + j_at[1]));
132
133 // Define required equations
134 ScalarValue eq_U = (U + val_term1 - val_term2);
135
136 ScalarGrad eqx_U =
137 (-1.0 * dt * (Dtilde * ((1.0 - phi) / 2.0) * Ux + j_at) / tau_U);
138
139 ScalarValue eq_phi = phi + (dt * xi / tau_phi);
140
141 // --- Submitting the terms for the governing equations ---
142
143 // Terms for the equation to evolve the concentration
144 variable_list.set_value_term(0, eq_U);
145 variable_list.set_gradient_term(0, eqx_U);
146
147 // Terms for the equation to evolve the order parameter
148 variable_list.set_value_term(1, eq_phi);
149 }
150 else if (solve_block_id == 1) // xi
151 {
152 // --- Getting the values and derivatives of the model variables ---
153
154 ScalarValue U = variable_list.template get_value<Scalar, Current>(0);
155 ScalarValue phi = variable_list.template get_value<Scalar, Current>(1);
156 ScalarGrad phix = variable_list.template get_gradient<Scalar, Current>(1);
157
158 // --- Setting the expressions for the terms in the governing equations ---
159
160 // Calculation of interface normal vector
161 ScalarValue normgradn = std::sqrt(phix.norm_square());
162 ScalarGrad normal = phix / (normgradn + regval);
163
164 // The cosine of theta
165 ScalarValue cth = normal[0];
166 // The sine of theta
167 ScalarValue sth = normal[1];
168
169 // The cosine of 4 theta
170 ScalarValue c4th = (sth * sth * sth * sth) + (cth * cth * cth * cth) -
171 (6.0 * sth * sth * cth * cth);
172 // The sine of 4 theta
173 ScalarValue s4th = (4.0 * sth * cth * cth * cth) - (4.0 * sth * sth * sth * cth);
174
175 // Anisotropic term
176 ScalarValue a_n = 1.0 + (epsilon * c4th);
177
178 // gradient energy coefficient, its derivative and square
179 ScalarValue a_d = -4.0 * epsilon * s4th;
180
181 // dimensionless temperature changes
182 ScalarValue y = variable_list.get_q_point_location()[1]; // The y-component
183 ScalarValue t_n = sim_timer.get_time(); // The time
184 ScalarValue tep = ((y - y0 - Vtilde * t_n) / ltilde);
185
186 // The anisotropy term that enters in to the equation for xi
187 ScalarGrad aniso;
188 aniso[0] = a_n * a_n * phix[0] - a_n * a_d * phix[1];
189 aniso[1] = a_n * a_n * phix[1] + a_n * a_d * phix[0];
190
191 // Define the terms in the equations
192 ScalarValue eq_xi =
193 phi - (phi * phi * phi) -
194 (lambda * (1.0 - phi * phi) * (1.0 - phi * phi) * (U + tep + U_off));
195
196 ScalarGrad eqx_xi = -aniso;
197
198 // --- Submitting the terms for the governing equations ---
199
200 variable_list.set_value_term(2, eq_xi);
201 variable_list.set_gradient_term(2, eqx_xi);
202 }
203 else if (solve_block_id == 2) // c postprocess
204 {
205 ScalarValue U = variable_list.template get_value<Scalar, Current>(0);
206 ScalarValue phi = variable_list.template get_value<Scalar, Current>(1);
207
208 ScalarValue c = (c0 / 2.0 / (1 + U0 - U0 * k)) * ((1.0 + k) - (1.0 - k) * phi) *
209 ((1.0) + (1.0 - k) * U);
210
211 variable_list.set_value_term(3, c);
212 }
213 }

216 number U0;
217 number U_off;
218 number y0;
219 number epsilon;
220 number k;
221 number lambda;
222 number Dtilde;
223 number Vtilde;
224 number ltilde;
225 number regval;
226};
