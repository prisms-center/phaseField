54 void
55 compute_rhs([[maybe_unused]] FieldContainer<dim, degree, number> &variable_list,
56 [[maybe_unused]] const SimulationTimer &sim_timer,
57 [[maybe_unused]] unsigned int solve_block_id) const override
58 {
59 if (solve_block_id == 1) // linear residual
60 {
61 ScalarValue dist_from_inclusion = variable_list.get_q_point_location().norm();
62 ScalarValue inclusion_radius(10.0);
63
64 VectorGrad transformation_strain;
65 using std::tanh;
66 ScalarValue strain_value =
67 0.01 * (0.5 + 0.5 * tanh(10.0 * (dist_from_inclusion - inclusion_radius)));
68 for (unsigned int i = 0; i < dim; i++)
69 {
70 transformation_strain[i][i] = strain_value;
71 }
72 VectorGrad stress;
73 compute_stress<dim, ScalarValue>(stiffness, transformation_strain, stress);
74 variable_list.set_gradient_term(0, -stress);
75 }
76 }

82 {
83 if (solve_block_id == 1) // linear lhs
84 {
85 VectorGrad ux = variable_list.template get_symmetric_gradient<Vector, LHS>(0);
86 VectorGrad stress;
87 compute_stress<dim, ScalarValue>(stiffness, ux, stress);
88 variable_list.set_gradient_term(0, stress);
89 }
90 }
