51struct DependencyExtents
52{
53 unsigned int oldest_age = 0;
54 unsigned int max_mg_level = 0;
55
56 // TODO: update this to account multigrid levels. Will need more
57 // than std::list<DependencyMap>
58 DependencyExtents(const std::set<unsigned int> &field_indices,
59 const std::list<DependencyMap> &all_dependeny_sets)
60 {
61 for (const DependencyMap &dependency_set : all_dependeny_sets)
62 {
63 for (unsigned int field_index : field_indices)
64 {
65 const auto &dep_pair_it = dependency_set.find(field_index);
66 if (dep_pair_it != dependency_set.end())
67 {
68 oldest_age = std::max(oldest_age, oldest(dep_pair_it->second));
69 }
70 }
71 }
72 }
73};

75struct NewDependencyExtents
76{
77 unsigned int max_age = 0;
78 std::vector<unsigned int> max_age_per_level;
79
80 NewDependencyExtents(const std::set<unsigned int> &field_indices,
81 const std::list<SolveBlock> &solve_blocks)
82 {
83 for (const SolveBlock &solve_block : solve_blocks)
84 {
85 const unsigned int num_levels =
86 solve_block.linear_solver_parameters.preconditioner == GMG
87 ? solve_block.linear_solver_parameters.mg_depth
88 : 0;
89 for (unsigned int field_index : field_indices)
90 {
91 {
92 const auto &dep_pair_it = solve_block.dependencies_rhs.find(field_index);
93 if (dep_pair_it != solve_block.dependencies_rhs.end())
94 {
95 unsigned int age = oldest2(dep_pair_it->second);
96 if (age != -1)
97 {
98 max_age = std::max(max_age, age);
99 }
100 // ignore mg_dependencies for rhs
101 }
102 }
103 {
104 const auto &dep_pair_it = solve_block.dependencies_lhs.find(field_index);
105 if (dep_pair_it != solve_block.dependencies_lhs.end())
106 {
107 // mg dependencies TODO: when issue #1182 is closed, move this inside
108 // the other if-statement
109 if (num_levels > max_age_per_level.size())
110 {
111 max_age_per_level.resize(num_levels, 0);
112 }
113 unsigned int age = oldest2(dep_pair_it->second);
114 if (age != -1)
115 {
116 max_age = std::max(max_age, age);
117 for (unsigned int relative_level = 0; relative_level < num_levels;
118 ++relative_level)
119 {
120 max_age_per_level[relative_level] =
121 std::max(max_age_per_level[relative_level], age);
122 }
123 }
124 }
125 }
126 }
127 }
128 }
129};
