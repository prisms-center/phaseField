514template <unsigned int dim, unsigned int degree, typename number>
515template <TensorRank Rank>
516FieldContainer<dim, degree, number>::FEEValuationDeps<Rank>::FEEValuationDeps(
517 const Dependency &dependency,
518 const std::pair<const SolutionLevel<dim, number> *, unsigned int> &mf_id_pair,
519 bool is_dst)
520 : solution_level(mf_id_pair.first)
521 , block_index(mf_id_pair.second)
522{
523 // Make an FEEvaluation if the current solution needs to be evaluated
524 if (dependency.flag)
525 {
526 fe_eval = std::make_shared<FEEDepPair>(FEEval<Rank>(solution_level->matrix_free,
527 block_index),
528 dependency.flag);
529 }
530 // Make FEEvaluations for the the old solutions
531 fe_eval_old.resize(dependency.old_flags.size(), nullptr);
532 for (unsigned int age = 0; age < dependency.old_flags.size(); ++age)
533 {
534 if (dependency.old_flags.at(age)) // kinda redundant... maybe remove?
535 {
536 fe_eval_old[age] =
537 std::make_shared<FEEDepPair>(FEEval<Rank>(solution_level->matrix_free,
538 block_index),
539 dependency.old_flags.at(age));
540 }
541 }
542 if (dependency.src_flag || is_dst)
543 {
544 fe_eval_src_dst =
545 std::make_shared<FEEDepPair>(FEEval<Rank>(solution_level->matrix_free,
546 block_index),
547 dependency.src_flag);
548 }
549}

658template <unsigned int dim, unsigned int degree, typename number>
659template <TensorRank Rank>
660inline void
661FieldContainer<dim, degree, number>::FEEValuationDeps<Rank>::eval(
662 const BlockVector<number> *_src_solutions,
663 bool plain)
664{
665 // NOTE: `read_dof_values_plain` must be called here so that constraints aren't
666 // implicitly applied. This allows us to have inhomogeneous constraints.
667 if (fe_eval)
668 {
669 fe_eval->first.read_dof_values_plain(solution_level->solutions.block(block_index));
670 fe_eval->first.evaluate(fe_eval->second);
671 }
672 for (unsigned int age = 0; age < fe_eval_old.size(); ++age)
673 {
674 if (FEEDepPairPtr &old_fe_eval = fe_eval_old[age])
675 {
676 old_fe_eval->first.read_dof_values_plain(
677 solution_level->old_solutions[age].block(block_index));
678 old_fe_eval->first.evaluate(old_fe_eval->second);
679 }
680 }
681 if (fe_eval_src_dst && fe_eval_src_dst->second != EvalFlags::nothing)
682 {
683 if (plain)
684 {
685 fe_eval_src_dst->first.read_dof_values_plain(
686 _src_solutions->block(block_index));
687 }
688 else
689 {
690 fe_eval_src_dst->first.read_dof_values(_src_solutions->block(block_index));
691 }
692 fe_eval_src_dst->first.evaluate(fe_eval_src_dst->second);
693 }
694}

696template <unsigned int dim, unsigned int degree, typename number>
697template <TensorRank Rank>
698inline void
699FieldContainer<dim, degree, number>::FEEValuationDeps<Rank>::reinit_and_eval(
700 unsigned int cell,
701 const BlockVector<number> *_src_solutions,
702 bool plain)
703{
704 // NOTE: `read_dof_values_plain` must be called here so that constraints aren't
705 // implicitly applied. This allows us to have inhomogeneous constraints.
706 if (fe_eval)
707 {
708 fe_eval->first.reinit(cell);
709 fe_eval->first.read_dof_values_plain(solution_level->solutions.block(block_index));
710 fe_eval->first.evaluate(fe_eval->second);
711 }
712 for (unsigned int age = 0; age < fe_eval_old.size(); ++age)
713 {
714 if (FEEDepPairPtr &old_fe_eval = fe_eval_old[age])
715 {
716 old_fe_eval->first.reinit(cell);
717 old_fe_eval->first.read_dof_values_plain(
718 solution_level->old_solutions[age].block(block_index));
719 old_fe_eval->first.evaluate(old_fe_eval->second);
720 }
721 }
722 if (fe_eval_src_dst)
723 {
724 fe_eval_src_dst->first.reinit(cell);
725 if (fe_eval_src_dst->second != EvalFlags::nothing)
726 {
727 if (plain)
728 {
729 fe_eval_src_dst->first.read_dof_values_plain(
730 _src_solutions->block(block_index));
731 }
732 else
733 {
734 fe_eval_src_dst->first.read_dof_values(_src_solutions->block(block_index));
735 }
736 fe_eval_src_dst->first.evaluate(fe_eval_src_dst->second);
737 }
738 }
739}
