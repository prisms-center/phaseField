33inline void
34add_equal_spacing_outputs(unsigned int n_outputs,
35 unsigned int n_increments,
36 std::set<unsigned int> &output_list)
37{
38 output_list.clear();
39 if (n_increments == 0)
40 {
41 output_list.insert(0);
42 return;
43 }
44 if (!n_outputs)
45 {
46 return;
47 }
48
49 unsigned int period = std::max(1U, n_increments / n_outputs);
50 for (unsigned int output = 0; output <= n_increments; output += period)
51 {
52 output_list.insert(output);
53 }
54}

56inline void
57add_log_spacing_outputs(unsigned int n_outputs,
58 unsigned int n_increments,
59 std::set<unsigned int> &output_list)
60{
61 output_list.clear();
62 if (n_increments == 0)
63 {
64 output_list.insert(0);
65 return;
66 }
67 if (!n_outputs)
68 {
69 return;
70 }
71
72 for (unsigned int output = 1; output <= n_outputs; output++)
73 {
74 output_list.insert((unsigned int) (std::round(
75 std::pow(double(n_increments), double(output) / double(n_outputs)))));
76 }
77}

79inline void
80add_n_per_decade_outputs(unsigned int n_outputs,
81 unsigned int n_increments,
82 std::set<unsigned int> &output_list)
83{
84 output_list.clear();
85 if (n_increments == 0)
86 {
87 output_list.insert(0);
88 return;
89 }
90 if (!n_outputs)
91 {
92 return;
93 }
94
95 output_list.insert(0);
96 output_list.insert(1);
97 for (unsigned int iteration = 2; iteration <= n_increments; iteration++)
98 {
99 const auto decade = (unsigned int) (std::ceil(std::log10(iteration)));
100 const auto step_size = (unsigned int) (std::pow(10, decade) / n_outputs);
101 if (iteration % step_size == 0)
102 {
103 output_list.insert(iteration);
104 }
105 }
106}
