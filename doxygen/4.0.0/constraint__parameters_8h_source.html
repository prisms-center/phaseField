43condition_from_string(const std::string &boundary_string)
44{
45 if (boundary_string == "Natural")
46 {
47 return Condition::Natural;
48 }
49 if (boundary_string == "Dirichlet")
50 {
51 return Condition::Dirichlet;
52 }
53 if (boundary_string == "Neumann")
54 {
55 return Condition::Neumann;
56 }
57 if (boundary_string == "Periodic")
58 {
59 return Condition::Periodic;
60 }
61 AssertThrow(false, dealii::ExcMessage("Invalid boundary condition " + boundary_string));
62 return Condition::Natural;
63}
