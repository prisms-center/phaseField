102template <int dim, int degree>
103inline void
104nonexplicitLinearSolver<dim, degree>::init()
105{
106 this->compute_subset_attributes(fieldSolveType::NONEXPLICIT_LINEAR);
107
108 // If the subset attribute is empty return early
109 if (this->subset_attributes.empty())
110 {
111 return;
112 }
113
114 this->set_initial_condition();
115
116 for (const auto &[index, variable] : this->subset_attributes)
117 {
118 if (this->user_inputs.linear_solve_parameters.linear_solve.at(index)
119 .preconditioner == preconditionerType::GMG)
120 {
121 gmg_solvers.emplace(
122 index,
123 std::make_unique<GMGSolver<dim, degree>>(this->user_inputs,
124 variable,
125 this->matrix_free_handler,
126 this->constraint_handler,
127 this->triangulation_handler,
128 this->dof_handler,
129 this->mg_matrix_free_handler,
130 this->solution_handler));
131 gmg_solvers.at(index)->init();
132 }
133 else
134 {
135 identity_solvers.emplace(
136 index,
137 std::make_unique<identitySolver<dim, degree>>(this->user_inputs,
138 variable,
139 this->matrix_free_handler,
140 this->constraint_handler,
141 this->solution_handler));
142 identity_solvers.at(index)->init();
143 }
144 }
145}

147template <int dim, int degree>
148inline void
149nonexplicitLinearSolver<dim, degree>::solve()
150{
151 // If the subset attribute is empty return early
152 if (this->subset_attributes.empty())
153 {
154 return;
155 }
156
157 for (const auto &[index, variable] : this->subset_attributes)
158 {
159 if (this->user_inputs.linear_solve_parameters.linear_solve.at(index)
160 .preconditioner == preconditionerType::GMG)
161 {
162 gmg_solvers.at(index)->solve();
163 }
164 else
165 {
166 identity_solvers.at(index)->solve();
167 }
168 }
