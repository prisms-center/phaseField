66inline void
67identitySolver<dim, degree>::init_system(
68 const triangulationHandler<dim> &triangulation_handler,
69 dealii::LinearAlgebra::distributed::Vector<double> &solution,
70 dealii::DoFHandler<dim> &dof_handler)
71{
72 this->system_matrix.clear();
73
74 dof_handler.reinit(triangulation_handler.get_triangulation());
75 dof_handler.distribute_dofs(this->fe);
76
77 this->constraint_handler.make_constraints(this->mapping, dof_handler);
78
79 typename dealii::MatrixFree<dim, double>::AdditionalData additional_data;
80 additional_data.tasks_parallel_scheme =
81 dealii::MatrixFree<dim, double>::AdditionalData::none;
82
83 // This should be done according to the residual flags to prevent excess data being
84 // evaluated for the shape functions.
85 additional_data.mapping_update_flags =
86 (update_values | update_gradients | update_JxW_values | update_quadrature_points);
87 std::shared_ptr<dealii::MatrixFree<dim, double>> system_mf_storage(
88 new dealii::MatrixFree<dim, double>());
89 system_mf_storage->reinit(this->mapping,
90 dof_handler,
91 this->constraint_handler.get_constraints(),
92 dealii::QGauss<1>(this->fe.degree + 1),
93 additional_data);
94 this->system_matrix.initialize(system_mf_storage);
95
96 this->system_matrix.initialize_dof_vector(this->change_in_solution);
97 this->system_matrix.initialize_dof_vector(solution);
98 this->system_matrix.initialize_dof_vector(this->residual);
99
100 // Apply boundary conditions to the solution vector as an initial guess
101 this->constraint_handler.get_constraints().distribute(solution);
102
103 conditionalOStreams::pout_base
104 << "Number of degrees of freedom: " << dof_handler.n_dofs() << std::endl;
105 conditionalOStreams::pout_summary()
106 << "Number of degrees of freedom: " << dof_handler.n_dofs() << std::endl;
107}

114template <int dim, int degree>
115inline void
116identitySolver<dim, degree>::solve(
117 [[maybe_unused]] const triangulationHandler<dim> &triangulation_handler,
118 dealii::LinearAlgebra::distributed::Vector<double> &solution,
119 [[maybe_unused]] const dealii::DoFHandler<dim> &dof_handler)
120{
121 // Compute the residual of the current solution (Ax-b)
122 this->system_matrix.compute_residual(this->residual, solution);
123
124 // Determine the residual tolerance
125 this->compute_solver_tolerance();
126
127 // Setup solver controls
128 this->solver_control.set_tolerance(this->tolerance);
129 dealii::SolverCG<dealii::LinearAlgebra::distributed::Vector<double>> cg(
130 this->solver_control);
131
132 try
133 {
134 this->change_in_solution = 0.0;
135 cg.solve(this->system_matrix,
136 this->change_in_solution,
137 this->residual,
138 PreconditionIdentity());
139 this->constraint_handler.get_constraints().set_zero(this->change_in_solution);
140 }
141 catch (...)
142 {
143 conditionalOStreams::pout_base
144 << "Warning: linear solver did not converge as per set tolerances.\n";
145 }
146 solution += this->change_in_solution;
147 this->constraint_handler.get_constraints().distribute(solution);
148
149 conditionalOStreams::pout_summary()
150 << this->solver_control.last_step() << " iterations\n";
151}
