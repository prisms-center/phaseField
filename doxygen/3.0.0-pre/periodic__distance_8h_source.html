23template <unsigned int dim, typename real>
24real
25distance(const dealii::Point<dim, real> &point1,
26 const dealii::Point<dim, real> &point2,
27 const UserInputParameters<dim> &user_inputs)
28{
29 const BoundaryParameters<dim> &boundary_params = user_inputs.get_boundary_parameters();
30 const SpatialDiscretization<dim> &spatial_discretization =
31 user_inputs.get_spatial_discretization();
32 if (!boundary_params.has_periodic_boundaries())
33 {
34 return point1.distance(point2);
35 }
36 // else
37 real dist = real(0.0);
38 for (unsigned int d = 0; d < dim; ++d)
39 {
40 using std::min;
41 real delta = point2[d] - point1[d];
42 if (boundary_params.get_periodicity()[d])
43 {
44 const real length = spatial_discretization.get_size()[d];
45 const real half_length = length / 2.0;
46 delta = pmod(delta - half_length, length) - half_length;
47 }
48 dist += delta * delta;
49 }
50 using std::sqrt;
51 return sqrt(dist);
52}
