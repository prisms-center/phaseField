128 Assert(multigrid_on, dealii::ExcNotInitialized());
129 const unsigned int relative_level = level - global_mg_level.first;
130 Assert(relative_level < mg_levels.size(),
131 dealii::ExcMessage("The multigrid hierarchy does not contain level = " +
132 std::to_string(level)));
133 auto iterator = std::find(mg_levels[relative_level].begin(),
134 mg_levels[relative_level].end(),
135 global_index);
136 Assert(iterator != mg_levels[relative_level].end(),
137 dealii::ExcMessage("The multigrid level does not contain index = " +
138 std::to_string(global_index)));
139 return std::distance(mg_levels[relative_level].begin(), iterator);
140 }

176template <unsigned int dim>
177inline MGInfo<dim>::MGInfo(const UserInputParameters<dim> &_user_inputs)
178 : user_inputs(&_user_inputs)
179{
180 const Max max_mg_level =
181 user_inputs->get_spatial_discretization().get_has_adaptivity()
182 ? user_inputs->get_spatial_discretization().get_max_refinement()
183 : user_inputs->get_spatial_discretization().get_global_refinement();
184 Min min_mg_level = UINT_MAX;
185
186 std::set<Types::Index> fields_with_multigrid;
187 std::map<Types::Index, Min> min_levels;
188 for (const auto &[index, linear_solver] :
189 user_inputs->get_linear_solve_parameters().get_linear_solve_parameters())
190 {
191 if (linear_solver.preconditioner == PreconditionerType::GMG)
192 {
193 const Min min_level = linear_solver.min_mg_level;
194 fields_with_multigrid.insert(index);
195 min_levels.emplace(index, min_level);
196 min_mg_level = std::min(min_mg_level, min_level);
197 }
198 }
199
200 global_mg_level = std::make_pair(min_mg_level, max_mg_level);
201
202 if (fields_with_multigrid.empty())
203 {
204 multigrid_on = false;
205 return;
206 }
207 multigrid_on = true;
208
209 // Create the set of tuples that let us know which fields have what multigrid levels.
210 // Note that we don't have to do this if the DependencyType is Change, although we must
211 // always include DependencyType::Normal.
212 std::set<std::tuple<Types::Index, DependencyType, Min>> all_lhs_fields;
213 for (const auto &index : fields_with_multigrid)
214 {
215 const auto &variable = user_inputs->get_variable_attributes().at(index);
216
217 // First add the DependencyType::Change
218 all_lhs_fields.insert(
219 std::make_tuple(index, DependencyType::Change, min_levels.at(index)));
220
221 // Then add the LHS dependencies. For now I just add all of them and go back to trim
222 // the duplicates that have the same variable but different Min multigrid levels.
223 Types::Index field_index = 0;
224 for (const auto &dependency_set : variable.get_eval_flag_set_lhs())
225 {
226 Types::Index dep_index = 0;
227 for (const auto &value : dependency_set)
228 {
229 if (static_cast<DependencyType>(dep_index) == DependencyType::Change ||
230 value == dealii::EvaluationFlags::EvaluationFlags::nothing)
231 {
232 dep_index++;
233 continue;
234 }
235 AssertThrow(
236 static_cast<DependencyType>(dep_index) == DependencyType::Normal ||
237 static_cast<DependencyType>(dep_index) == DependencyType::Change,
238 FeatureNotImplemented("The dependency type = " +
239 to_string(static_cast<DependencyType>(dep_index)) +
240 " is not supported"));
241
242 all_lhs_fields.insert(
243 std::make_tuple(field_index,
244 static_cast<DependencyType>(dep_index),
245 min_levels.at(index)));
246
247 dep_index++;
248 }
249
250 field_index++;
251 }
252 }
253 // Trim fields that have the same first and second entry of the tuple.
254 std::map<std::pair<Types::Index, DependencyType>, Min> lowest_mg_level;
255 for (const auto &tuple : all_lhs_fields)
256 {
257 auto key = std::make_pair(std::get<0>(tuple), std::get<1>(tuple));
258 const Min min_level = std::get<2>(tuple);
259
260 auto iterator = lowest_mg_level.find(key);
261 if (iterator == lowest_mg_level.end() || min_level < iterator->second)
262 {
263 lowest_mg_level[key] = min_level;
264 }
265 }
266 for (const auto &[key, min_level] : lowest_mg_level)
267 {
268 lhs_fields.insert(std::make_tuple(key.first, key.second, min_level));
269 }
270
271 mg_levels.resize(global_mg_level.second - global_mg_level.first + 1);
272 for (const auto &[index, dependency, min_level] : lhs_fields)
273 {
274 const unsigned int relative_level = min_level - global_mg_level.first;
275 for (unsigned int level = relative_level; level < mg_levels.size(); level++)
276 {
277 mg_levels[level].push_back(index);
278 }
279 }
280}

282template <unsigned int dim>
283void
284MGInfo<dim>::print()
285{
286 ConditionalOStreams::pout_summary()
287 << "================================================\n"
288 << " MGInfo\n"
289 << "================================================\n";
290 if (!multigrid_on)
291 {
292 ConditionalOStreams::pout_summary()
293 << " There are no fields with multigrid enabled\n\n"
294 << std::flush;
295 return;
296 }
297 ConditionalOStreams::pout_summary() << " Global min = " << global_mg_level.first
298 << " and max = " << global_mg_level.second << "\n"
299 << " LHS dependency fields:\n";
300 for (const auto &[field, dependency, minimum] : lhs_fields)
301 {
302 ConditionalOStreams::pout_summary()
303 << " Index " << field << " Type " << to_string(dependency) << " Min "
304 << minimum << "\n";
305 }
306 unsigned int outer_index = 0;
307 ConditionalOStreams::pout_summary() << " LHS index hierarchy:\n";
308 for (const auto &vector : mg_levels)
309 {
310 ConditionalOStreams::pout_summary() << " Outer index " << outer_index << "\n";
311 unsigned int inner_index = 0;
312 for (const auto &index : vector)
313 {
314 ConditionalOStreams::pout_summary()
315 << " Inner index " << inner_index << " Field index: " << index << "\n";
316 inner_index++;
317 }
318 outer_index++;
319 }
320 ConditionalOStreams::pout_summary() << "\n" << std::flush;
321}
