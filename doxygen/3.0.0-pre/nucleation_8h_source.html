100 const double delta_t = nuc_params.get_nucleation_period() * time_info.get_timestep();
101 auto &rng = user_inputs.get_miscellaneous_parameters().rng;
102
103 // Set up FEValues
104 static const dealii::QGaussLobatto<dim> quadrature(degree + 1);
105 static const unsigned int num_quad_points = quadrature.size();
106 dealii::FESystem<dim> fe_system(dealii::FE_Q<dim>(dealii::QGaussLobatto<1>(degree + 1)),
107 1);

109 quadrature,
110 dealii::UpdateFlags::update_values |
111 dealii::UpdateFlags::update_JxW_values);
112 std::list<Nucleus<dim>> new_nuclei_list;
113 // Loop over nucleation rate variables and attempt seeding at each cell
114 for (const auto &[index, variable] : user_inputs.get_variable_attributes())
115 {
116 if (!variable.is_nucleation_rate())
117 {
118 continue;
119 }
120 std::uniform_int_distribution<unsigned int> nucleating_index_dist(
121 0,
122 variable.get_nucleating_field_indices().size() - 1);
123 // Perform nucleation logic here
124 // This is where you would check conditions and create nuclei
125 for (const auto &cell : solver_context.get_triangulation_handler()
126 .get_triangulation()
127 .active_cell_iterators())

131 continue;

134 // Grab the DoFHandler iterator

136 solver_context.get_dof_handler().get_dof_handler(index));

139 fe_values.reinit(dof_iterator);
140 // Get the values for a scalar field
141 fe_values.get_function_values(*(solver_context.get_solution_handler()
142 .get_solution_vector(index,
143 DependencyType::Normal)),
144 values);
145 double nuc_rate = 0.0;
146 double cell_volume = 0.0;
147 for (unsigned int q_point = 0; q_point < num_quad_points; ++q_point)
148 {
149 nuc_rate += values[q_point] * fe_values.get_quadrature().weight(q_point);
150 cell_volume += fe_values.JxW(q_point);

162 0.0,
163 1.0);
164 nucleus_location_unit_cell[d] = uniform_unit_interval(rng);
165 }
166 dealii::Point<dim> nucleus_location =
167 solver_context.get_mapping()
168 .transform_unit_to_real_cell(cell, nucleus_location_unit_cell);
169 double seed_time = time_info.get_time();
170 unsigned int seed_increment = time_info.get_increment();
171 unsigned int nucleating_index =
172 variable.get_nucleating_field_indices()[nucleating_index_dist(rng)];
173
174 new_nuclei_list.emplace_back(nucleating_index,
175 nucleus_location,
176 seed_time,
177 seed_increment);
178 }
179 }
180 }
181 return gather_exclude_broadcast_nuclei(new_nuclei_list, nuclei, user_inputs);

248template <unsigned int dim, unsigned int degree, typename number>
249inline bool
250NucleationHandler<dim, degree, number>::gather_exclude_broadcast_nuclei(
251 std::list<Nucleus<dim>> &new_nuclei_list,
252 std::vector<Nucleus<dim>> &global_nuclei,
253 const UserInputParameters<dim> &user_inputs)
254{
255 // dont waste time if no nuclei appeared
256 if (!bool(dealii::Utilities::MPI::sum(new_nuclei_list.size(), MPI_COMM_WORLD)))
257 {
258 return false;
259 }
260
261 // Set up refs
262 const NucleationParameters &nuc_params = user_inputs.get_nucleation_parameters();
263 const TemporalDiscretization &time_info = user_inputs.get_temporal_discretization();
264 RNGEngine &rng = user_inputs.get_miscellaneous_parameters().rng;
265
266 // Gather new nuclei to root process
267 std::vector<Nucleus<dim>> new_nuclei(new_nuclei_list.begin(), new_nuclei_list.end());
268 mpi_gather_nuclei(new_nuclei);
269 bool any_nucleation_occurred = false;
270 if (dealii::Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
271 {
272 // Remove nuclei within their exclusion distance and add to nuclei list
273 ConditionalOStreams::pout_base()
274 << "[Increment " << time_info.get_increment() << "] : Nucleation\n"
275 << " " << new_nuclei.size() << " nuclei generated before exclusion.\n"
276 << " Excluding nuclei...\n";
277 unsigned int count = 0;
278
279 // remove bias from cell order
280 std::shuffle(new_nuclei.begin(), new_nuclei.end(), rng);
281
282 while (!new_nuclei.empty())
283 {
284 const Nucleus<dim> &nuc = new_nuclei.back();
285 bool valid =
286 std::none_of(global_nuclei.begin(),
287 global_nuclei.end(),
288 [&](const Nucleus<dim> &existing_nucleus)
289 {
290 const double distance =
291 prisms::distance<dim, double>(nuc.location,
292 existing_nucleus.location,
293 user_inputs);
294 return nuc_params.check_active(existing_nucleus, time_info) &&
295 (distance < nuc_params.get_exclusion_distance() ||
296 (nuc.field_index == existing_nucleus.field_index &&
297 distance <
298 nuc_params.get_same_field_exclusion_distance()));
299 });
300 if (valid)
301 {
302 // Note: Using push_back() in a loop is not good use for
303 // vectors. We also don't want to use reserve() on the upper bound
304 // because that could allocate much more space than needed. I originally
305 // was using a std::list to avoid this issue, but that is unfriendly to
306 // the MPI functions. One solution could be to convert between data
307 // structures as needed, but that also adds overhead. For now, I will
308 // assume that the total number of nuclei is not enough to
309 // cause significant performance issues.
310 global_nuclei.push_back(nuc);
311 ConditionalOStreams::pout_base()
312 << " New nucleus at: " << nuc.location << "\n";
313 ++count;
314 any_nucleation_occurred = true;
315 }
316 new_nuclei.pop_back();
317 }
318 ConditionalOStreams::pout_base() << " " << count
319 << " new nuclei after exclusion.\n"
320 " "
321 << global_nuclei.size() << " total nuclei.\n\n"
322 << std::flush;
323 }
324 MPI_Bcast(&any_nucleation_occurred, 1, MPI_CXX_BOOL, 0, MPI_COMM_WORLD);
325 mpi_broadcast_nuclei(global_nuclei);
326 return any_nucleation_occurred;
327}

329template <unsigned int dim, unsigned int degree, typename number>
330inline void
331NucleationHandler<dim, degree, number>::mpi_gather_nuclei(
332 std::vector<Nucleus<dim>> &local_nuclei)
333{
334 // Step 1: Share how many nuclei each rank has
335 int rank = dealii::Utilities::MPI::this_mpi_process(MPI_COMM_WORLD);
336 int num_procs = dealii::Utilities::MPI::n_mpi_processes(MPI_COMM_WORLD);
337 int local_count = local_nuclei.size();
338 std::vector<int> nuclei_counts_per_rank;
339 if (rank == 0)
340 {
341 nuclei_counts_per_rank.resize(num_procs);
342 }
343
344 MPI_Gather(&local_count,
345 1,
346 MPI_INT,
347 nuclei_counts_per_rank.data(),
348 1,
349 MPI_INT,
350 0,
351 MPI_COMM_WORLD);
352
353 // Step 2: Compute displacements and allocate receive buffer on root
354 std::vector<int> recv_displacements;
355 std::vector<Nucleus<dim>> gathered_nuclei;
356 if (rank == 0)
357 {
358 recv_displacements.resize(num_procs);
359 recv_displacements[0] = 0;
360 for (int r = 1; r < num_procs; ++r)
361 recv_displacements[r] = recv_displacements[r - 1] + nuclei_counts_per_rank[r - 1];
362
363 int total_count = recv_displacements.back() + nuclei_counts_per_rank.back();
364 gathered_nuclei.resize(total_count);
365 }
366
367 // Step 3: Gather all nuclei into root's buffer
368 MPI_Gatherv(local_nuclei.data(),
369 local_count,
370 Nucleus<dim>::mpi_datatype(),
371 gathered_nuclei.data(),
372 nuclei_counts_per_rank.data(),
373 recv_displacements.data(),
374 Nucleus<dim>::mpi_datatype(),
375 0,
376 MPI_COMM_WORLD);
377 if (rank == 0)
378 {
379 local_nuclei = gathered_nuclei;
380 }
381}

383template <unsigned int dim, unsigned int degree, typename number>
384inline void
385NucleationHandler<dim, degree, number>::mpi_broadcast_nuclei(
386 std::vector<Nucleus<dim>> &local_nuclei)
387{
388 int rank = dealii::Utilities::MPI::this_mpi_process(MPI_COMM_WORLD);
389 int count = local_nuclei.size();
390 MPI_Bcast(&count, 1, MPI_INT, 0, MPI_COMM_WORLD);
391
392 if (rank != 0)
393 {
394 local_nuclei.resize(count);
395 }
396
397 MPI_Bcast(local_nuclei.data(), count, Nucleus<dim>::mpi_datatype(), 0, MPI_COMM_WORLD);
398}
