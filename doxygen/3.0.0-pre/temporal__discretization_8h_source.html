143inline void
144TemporalDiscretization::postprocess_and_validate(
145 const std::map<unsigned int, VariableAttributes> &var_attributes)
146{
147 // If all of the variables are `TimeIndependent`, `Auxiliary`, or `Constant` then
148 // total_increments should be 0 and final_time should be 0
149 bool only_time_independent_pdes = true;
150 for (const auto &[index, variable] : var_attributes)
151 {
152 if (variable.is_postprocess())
153 {
154 continue;
155 }
156 if (variable.get_pde_type() == PDEType::ExplicitTimeDependent ||
157 variable.get_pde_type() == PDEType::ImplicitTimeDependent)
158 {
159 only_time_independent_pdes = false;
160 break;
161 }
162 }
163 if (only_time_independent_pdes)
164 {
165 total_increments = 0;
166 return;
167 }
168
169 // Check that the timestep is greater than zero
170 AssertThrow(dt > 0.0,
171 dealii::ExcMessage(
172 "The timestep must be greater than zero for transient problems."));
173
174 // Pick the maximum specified time since the default values are zero
175 final_time = std::max(final_time, dt * total_increments);
176 total_increments = static_cast<unsigned int>(std::ceil(final_time / dt));
177}

187 << "Total increments: " << total_increments << "\n"
188 << "Final time: " << final_time << "\n\n"
189 << std::flush;
