93template <unsigned int dim, typename number>
94inline ReadBinary<dim, number>::ReadBinary(
95 const InitialConditionFile &_ic_file,
96 const SpatialDiscretization<dim> &_spatial_discretization)
97 : ReadFieldBase<dim, number>(_ic_file, _spatial_discretization)
98{
99 check_size(this->ic_file);
100 std::ifstream dataFile(this->ic_file.filename, std::ios::binary);
101 AssertThrow(dataFile,
102 dealii::ExcMessage("Could not open binary file: " +
103 this->ic_file.filename));
104 data.reserve(n_values);
105 const unsigned int bufsize = sizeof(number);
106 char buf[bufsize];
107 number numbuf;
108 for (dealii::types::global_dof_index j = 0; j < n_values; ++j)
109 {
110 dataFile.read(buf, bufsize);
111 memcpy(&numbuf, &buf, bufsize);
112 data.push_back(numbuf);
113 }
114 dataFile.close();
115 for (unsigned int j = 0; j < dim; ++j)
116 {
117 dNdx[j] = static_cast<number>(this->ic_file.n_data_points[j] - 1) /
118 this->spatial_discretization.get_size()[j];
119 }
120}

123template <unsigned int dim, typename number>
124inline void
125ReadBinary<dim, number>::check_size(const InitialConditionFile &ic_file)
126{
127 std::uintmax_t file_size = std::filesystem::file_size(ic_file.filename);
128 // Calculate expected size: number of points * size of each point
129 n_values = 1;
130 for (unsigned int j = 0; j < dim; j++)
131 {
132 n_values *= static_cast<std::uintmax_t>(ic_file.n_data_points[j]);
133 }
134
135 std::uintmax_t expected_size = n_values * sizeof(number);
136 // Make sure expected size is not zero
137 AssertThrow(expected_size != 0,
138 dealii::ExcMessage(
139 std::string("Expected input array size is zero, check that the ") +
140 std::string("number of data points in each used direction is set "
141 "correctly in the input file.")));
142 // Make sure the size matches for either a scalar or vector
143 AssertThrow(((file_size == expected_size) || (file_size == dim * expected_size)),
144 dealii::ExcMessage(
145 std::string("Expected binary file size (") +
146 std::to_string(expected_size) + std::string(" bytes for scalar or ") +
147 std::to_string(3 * expected_size) +
148 std::string(" bytes for vector) does not match actual file size (") +
149 std::to_string(file_size) + std::string(" bytes).")));
150}

152template <unsigned int dim, typename number>
153inline void
154ReadBinary<dim, number>::write_file(const std::vector<number> &data,
155 const InitialConditionFile &ic_file)
156{
157 std::ofstream dataFile(ic_file.filename, std::ios::binary);
158 AssertThrow(dataFile,
159 dealii::ExcMessage("Could not open binary file: " + ic_file.filename));
160 const unsigned int bufsize = sizeof(number);
161 char buf[bufsize];
162 for (dealii::types::global_dof_index j = 0; j < data.size(); ++j)
163 {
164 memcpy(&buf, &data[j], bufsize);
165 dataFile.write(buf, bufsize);
166 }
167 dataFile.close();
168}

170// function to get values at specified index
171// encapsulates the assumed storage order of the input data (fortran-style)
172template <unsigned int dim, typename number>
173inline dealii::Vector<number>
174ReadBinary<dim, number>::get_vals(
175 const std::array<dealii::types::global_dof_index, dim> &multi_index,
176 const unsigned int n_components)
177{
178 dealii::Vector<number> value(n_components);
179 // index of the scalar or first component of vector
180 dealii::types::global_dof_index flat_index = 0;
181 dealii::types::global_dof_index subspace_size = 1;
182 for (unsigned int j = 0; j < dim; ++j)
183 {
184 flat_index += multi_index[j] * subspace_size;
185 subspace_size *= this->ic_file.n_data_points[j];
186 }
187 flat_index *= n_components;
188 for (unsigned int component = 0; component < n_components; ++component)
189 {
190 value[component] = this->data[flat_index + component];
191 }
192 return value;
193}

195template <unsigned int dim, typename number>
196inline dealii::Vector<number>
197ReadBinary<dim, number>::interpolate(const dealii::Point<dim> &point,
198 const unsigned int n_pt_values)
199{
200 std::vector<std::array<uint, 2>> indices(dim,
201 std::array<uint, 2> {
202 {0u, 0u}
203 }); // lower and upper indices in
204 // each dimension
205 std::vector<std::array<number, 2>> weights(
206 dim,
207 std::array<number, 2> {
208 {number(0.0), number(0.0)}
209 }); // lower and upper weights in each dimension
210 for (unsigned int j = 0; j < dim; ++j)
211 {
212 indices[j][0] = (dealii::types::global_dof_index) std::floor(dNdx[j] * point[j]);
213 AssertThrow(indices[j][0] >= 0,
214 dealii::ExcMessage(
215 "negative index encountered in ReadBinary::interpolate"));
216 AssertThrow(indices[j][0] < this->ic_file.n_data_points[j],
217 dealii::ExcMessage(
218 "index out of bounds encountered in ReadBinary::interpolate"));
219 if (indices[j][0] == this->ic_file.n_data_points[j] - 1)
220 {
221 weights[j][0] = 1.0;
222 weights[j][1] = 0.0;
223 indices[j][1] = indices[j][0];
224 }
225 else
226 {
227 indices[j][1] = indices[j][0] + 1;
228 // Calculate the fractional position between grid points
229 number pt_lower = static_cast<number>(indices[j][0]) / dNdx[j];
230 weights[j][1] = (point[j] - pt_lower) * dNdx[j]; // 1/dNdx[j] = grid spacing;
231 weights[j][0] = 1.0 - weights[j][1];
232 }
233 }
234 dealii::Vector<number> pt_values(n_pt_values);
235 pt_values = 0.0;
236 // 1 << dim is 2^dim, i.e. number of corners of the cell
237 for (unsigned int k = 0; k < (1 << dim); ++k)
238 {
239 std::array<dealii::types::global_dof_index, dim> pt_indices;
240 number pt_wgt = 1.0;
241 for (unsigned int j = 0; j < dim; ++j)
242 {
243 // bit j of k tells us whether to use lower (0)
244 // or upper (1) index/weight in dimension j
245 bool bit = (k >> j & 1);
246 pt_indices[j] = indices[j][bit];
247 pt_wgt *= weights[j][bit];
248 }
249 dealii::Vector<number> values_temp = this->get_vals(pt_indices, n_pt_values);
250 for (unsigned int n_pt_index = 0; n_pt_index < n_pt_values; ++n_pt_index)
251 {
252 pt_values[n_pt_index] += pt_wgt * values_temp[n_pt_index];
253 }
254 }
255 return pt_values;
256}
