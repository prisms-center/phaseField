112template <unsigned int dim, typename number>
113inline ReadBinary<dim, number>::ReadBinary(
114 const InitialConditionFile &_ic_file,
115 const SpatialDiscretization<dim> &_spatial_discretization)
116 : ReadFieldBase<dim, number>(_ic_file, _spatial_discretization)
117{
118 // Make sure the dataset format is correct
119 AssertThrow(this->ic_file.dataset_format == DataFormatType::FlatBinary,
120 dealii::ExcMessage("Dataset format must be FlatBinary"));
121
122 // Check that only one field is being read in
123 AssertThrow(this->ic_file.file_variable_names.size() == 1 &&
124 this->ic_file.simulation_variable_names.size() == 1,
125 dealii::ExcMessage("Only one field can be read in from a binary file"));
126
127 // Make sure we have a rectangular domain
128 AssertThrow(_spatial_discretization.get_type() ==
129 SpatialDiscretization<dim>::TriangulationType::Rectangular,
130 dealii::ExcMessage(
131 "Only rectangular domains are supported for binary input files"));
132
133 // Compute the total number of points in the binary file
134 for (unsigned int d : std::views::iota(0U, dim))
135 {
136 n_points *= this->ic_file.n_data_points[d];
137 }
138
139 // Check that the binary matches an expected size
140 check_file_size();
141
142 // Read in the binary file
143 std::ifstream data_file(this->ic_file.filename, std::ios::binary);
144 AssertThrow(data_file,
145 dealii::ExcMessage("Could not open binary file: " +
146 this->ic_file.filename));
147
148 // Reserve space in the data vector
149 data.reserve(n_values);
150
151 // Read in the data
152 for (dealii::types::global_dof_index i : std::views::iota(0U, n_values))
153 {
154 std::array<char, sizeof(number)> buffer;
155 data_file.read(buffer.data(), sizeof(number));
156 data.push_back(std::bit_cast<number>(buffer));
157 }
158 data_file.close();
159}

161template <unsigned int dim, typename number>
162inline void
163ReadBinary<dim, number>::check_file_size()
164{
165 // Grab the file size of the binary file in bytes
166 auto file_size = std::filesystem::file_size(this->ic_file.filename);
167
168 // Compute the expected size of the binary file. This is simply the number of points
169 // mutliplied by the size of each point in bytes.
170 auto expected_size_scalar = static_cast<std::uintmax_t>(n_points * sizeof(number));
171 auto expected_size_vector = static_cast<std::uintmax_t>(dim * expected_size_scalar);
172
173 // Make sure expected size is not zero
174 AssertThrow(
175 expected_size_scalar != 0 && expected_size_vector != 0,
176 dealii::ExcMessage(
177 "Expected input array size is zero, check that the number of data points "
178 "in each used direction is set correctly in the input file for your binary file. "
179 "You likely have the number of data points set to zero in all directions."));
180 // Make sure the size matches for either a scalar or vector
181 AssertThrow(file_size == expected_size_scalar || file_size == expected_size_vector,
182 dealii::ExcMessage(
183 "Expected binary file size (" + std::to_string(expected_size_scalar) +
184 " bytes for scalar or " + std::to_string(expected_size_vector) +
185 " bytes for vector) does not match actual file size (" +
186 std::to_string(file_size) + " bytes)."));
187
188 // Set the number of values
189 n_values = file_size == expected_size_scalar ? n_points : dim * n_points;
190}

192template <unsigned int dim, typename number>
193inline void
194ReadBinary<dim, number>::write_file(const std::vector<number> &data,
195 const InitialConditionFile &ic_file)
196{
197 // Try to open the file
198 std::ofstream data_file(ic_file.filename, std::ios::binary);
199 AssertThrow(data_file,
200 dealii::ExcMessage("Could not open binary file: " + ic_file.filename));
201
202 // Write the data
203 for (dealii::types::global_dof_index j : std::views::iota(0U, data.size()))
204 {
205 auto buffer = std::bit_cast<std::array<char, sizeof(number)>>(data[j]);
206 data_file.write(buffer.data(), sizeof(number));
207 }
208 data_file.close();
209}

211template <unsigned int dim, typename number>
212inline dealii::Vector<number>
213ReadBinary<dim, number>::get_value(const dealii::types::global_dof_index index,
214 const unsigned int n_components)
215{
216 // Create vector to hold the value with the correct number of components
217 dealii::Vector<number> value(n_components);
218
219 // Check that the number of components matches what we expect from the number of values
220 Assert(n_values % n_components == 0,
221 dealii::ExcMessage("The number of components requested in the get_value call "
222 "does not match the number of values in the binary file."));
223
224 // Fill the value vector
225 for (unsigned int component : std::views::iota(0U, n_components))
226 {
227 Assert(((n_components * index) + component) < data.size(),
228 dealii::ExcMessage("Index out of bounds in ReadBinary::get_vals"));
229 value[component] = data[(n_components * index) + component];
230 }
231
232 return value;
233}

235template <unsigned int dim, typename number>
236inline dealii::Vector<number>
237ReadBinary<dim, number>::interpolate(const dealii::Point<dim> &point,
238 const unsigned int n_components)
239{
240 Assert(n_components == 1 || n_components == dim,
241 dealii::ExcMessage(
242 "Number of components for interpolation must be 1 (scalar) or dim (vector)"));
243
244 // Create a vector to hold the interpolated value
245 dealii::Vector<number> value(n_components);
246
247 // Get the spatial discretization
248 const auto &spatial_discretization = this->spatial_discretization;
249
250 // Compute the spacing in each direction
251 std::array<number, dim> spacing;
252 for (unsigned int d : std::views::iota(0U, dim))
253 {
254 spacing[d] = spatial_discretization.get_size()[d] /
255 static_cast<number>(this->ic_file.n_data_points[d]);
256 }
257
258 // Compute the indices of the lower corner of the cell containing the point
259 std::array<dealii::types::global_dof_index, dim> lower_indices;
260 for (unsigned int d : std::views::iota(0U, dim))
261 {
262 lower_indices[d] =
263 static_cast<dealii::types::global_dof_index>(std::floor(point[d] / spacing[d]));
264 // Make sure we don't go out of bounds
265 if (lower_indices[d] >= this->ic_file.n_data_points[d] - 1)
266 {
267 lower_indices[d] = this->ic_file.n_data_points[d] - 2;
268 }
269 }
270
271 // Compute the weights for interpolation in each direction
272 std::array<number, dim> weights;
273 for (unsigned int d : std::views::iota(0U, dim))
274 {
275 weights[d] = (point[d] - lower_indices[d] * spacing[d]) / spacing[d];
276 }
277
278 // Perform multilinear interpolation based on the dimension
279 if constexpr (dim == 1)
280 {
281 // Here is the map of the nodes in 1D:
282 // 0 — 1
283
284 // Grab the index of the lower left corner of the cell
285 auto lower_index = lower_indices[0];
286
287 // Get the indices of the two points in 1D
288 auto node_index_0 = lower_index;
289 auto node_index_1 = lower_index + 1;
290
291 // Get the values at these points
292 auto value_0 = get_value(node_index_0, n_components);
293 auto value_1 = get_value(node_index_1, n_components);
294
295 // Interpolate
296 for (unsigned int c : std::views::iota(0U, n_components))
297 {
298 value[c] = (1.0 - weights[0]) * value_0[c] + weights[0] * value_1[c];
299 }
300 }
301 else if constexpr (dim == 2)
302 {
303 // Here is the map of the nodes in 2D:
304 // 01 — 11
305 // | |
306 // 00 — 10
307
308 // Grab the row length in the x-direction (0th direction)
309 auto row_length_0 = this->ic_file.n_data_points[0];
310
311 // Grab the index of the lower left corner of the cell
312 auto lower_index = lower_indices[0] + (lower_indices[1] * row_length_0);
313
314 // Get the indices of the four points in 2D
315 auto node_index_00 = lower_index;
316 auto node_index_10 = lower_index + 1;
317
318 auto node_index_01 = lower_index + row_length_0;
319 auto node_index_11 = lower_index + row_length_0 + 1;
320
321 // Get the values at these points
322 auto value_00 = get_value(node_index_00, n_components);
323 auto value_10 = get_value(node_index_10, n_components);
324
325 auto value_01 = get_value(node_index_01, n_components);
326 auto value_11 = get_value(node_index_11, n_components);
327
328 // Interpolate
329 for (unsigned int c : std::views::iota(0U, n_components))
330 {
331 value[c] = (1.0 - weights[0]) * (1.0 - weights[1]) * value_00[c] +
332 weights[0] * (1.0 - weights[1]) * value_10[c] +
333 (1.0 - weights[0]) * weights[1] * value_01[c] +
334 weights[0] * weights[1] * value_11[c];
335 }
336 }
337 else if constexpr (dim == 3)
338 {
339 // Here is the map of the nodes in 3D:
340 //
341 // 011 ———— 111
342 // / | / |
343 // / 001 / |
344 // 010 ——— 110 101
345 // | / | /
346 // | / | /
347 // 000 ———— 100
348
349 // Grab the row length in the x-direction (0th direction)
350 auto row_length_0 = this->ic_file.n_data_points[0];
351 // Grab the row length in the y-direction (1st direction)
352 auto row_length_1 = this->ic_file.n_data_points[1];
353
354 // Grab the index of the lower left corner of the cell
355 auto lower_index = lower_indices[0] + (lower_indices[1] * row_length_0) +
356 (lower_indices[2] * row_length_0 * row_length_1);
357
358 // Get the indices of the eight points in 3D
359 auto node_index_000 = lower_index;
360 auto node_index_100 = lower_index + 1;
361
362 auto node_index_010 = lower_index + row_length_0;
363 auto node_index_110 = lower_index + row_length_0 + 1;
364
365 auto node_index_001 = lower_index + (row_length_0 * row_length_1);
366 auto node_index_101 = lower_index + (row_length_0 * row_length_1) + 1;
367
368 auto node_index_011 = lower_index + (row_length_0 * row_length_1) + row_length_0;
369 auto node_index_111 =
370 lower_index + (row_length_0 * row_length_1) + row_length_0 + 1;
371
372 // Get the values at these points
373 auto value_000 = get_value(node_index_000, n_components);
374 auto value_100 = get_value(node_index_100, n_components);
375
376 auto value_010 = get_value(node_index_010, n_components);
377 auto value_110 = get_value(node_index_110, n_components);
378
379 auto value_001 = get_value(node_index_001, n_components);
380 auto value_101 = get_value(node_index_101, n_components);
381
382 auto value_011 = get_value(node_index_011, n_components);
383 auto value_111 = get_value(node_index_111, n_components);
384
385 // Interpolate
386 for (unsigned int c : std::views::iota(0U, n_components))
387 {
388 value[c] =
389 (1.0 - weights[0]) * (1.0 - weights[1]) * (1.0 - weights[2]) * value_000[c] +
390 weights[0] * (1.0 - weights[1]) * (1.0 - weights[2]) * value_100[c] +
391 (1.0 - weights[0]) * weights[1] * (1.0 - weights[2]) * value_010[c] +
392 weights[0] * weights[1] * (1.0 - weights[2]) * value_110[c] +
393 (1.0 - weights[0]) * (1.0 - weights[1]) * weights[2] * value_001[c] +
394 weights[0] * (1.0 - weights[1]) * weights[2] * value_101[c] +
395 (1.0 - weights[0]) * weights[1] * weights[2] * value_011[c] +
396 weights[0] * weights[1] * weights[2] * value_111[c];
397 }
398 }
399
400 return value;
401}

403template <unsigned int dim, typename number>
404inline number
405ReadBinary<dim, number>::get_scalar_value(const dealii::Point<dim> &point,
406 [[maybe_unused]] const std::string &scalar_name)
407{
408 Assert(n_values == n_points,
409 dealii::ExcMessage("The number of points should match the number of values in a "
410 "binary file for a scalar field. Make sure the file size is "
411 "correct and you are trying to access a scalar field."));
412 return interpolate(point, 1)[0];
413}
414
415template <unsigned int dim, typename number>
416inline dealii::Vector<number>
417ReadBinary<dim, number>::get_vector_value(const dealii::Point<dim> &point,
418 [[maybe_unused]] const std::string &vector_name)
419{
420 Assert((n_values / dim) == n_points,
421 dealii::ExcMessage("The number of points should match the number of values "
422 "divided by the dimension in a binary file for a vector "
423 "field. Make sure the file size is correct and you are "
424 "trying to access a vector field."));
425
426 return interpolate(point, dim);
427}

432{
433 for (dealii::types::global_dof_index i : std::views::iota(0U, n_values))

438 }
439 ConditionalOStreams::pout_summary() << std::flush;
440}
