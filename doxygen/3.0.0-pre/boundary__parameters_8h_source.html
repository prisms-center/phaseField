39 {
40 UndefinedBoundary,
41 Natural,

43 Periodic,
44 Neumann,
45 NonuniformDirichlet,
46 NonuniformNeumann,
47 TimeDependentNonuniformDirichlet,
48 TimeDependentNonuniformNeumann,
49 };

54 bool
55 operator==(const BoundaryCondition &other) const
56 {
57 return boundary_condition_map == other.boundary_condition_map &&
58 dirichlet_value_map == other.dirichlet_value_map;
59 }

82 [[nodiscard]] double
83 get_dirichlet_value(dealii::types::boundary_id boundary_id) const
84 {
85 Assert(boundary_condition_map.at(boundary_id) == Type::Dirichlet,
86 dealii::ExcMessage("Boundary condition is not dirichlet"));

97 {
98 Assert(boundary_condition_map.at(boundary_id) == Type::Dirichlet,
99 dealii::ExcMessage("Boundary condition is not dirichlet"));
100 dirichlet_value_map.emplace(boundary_id, boundary_value);
101 }

106 [[nodiscard]] std::string
107 to_string(Type boundary_type) const
108 {
109 switch (boundary_type)
110 {
111 case Type::UndefinedBoundary:
112 return "UndefinedBoundary";
113 case Type::Natural:
114 return "Natural";
115 case Type::Dirichlet:
116 return "Dirichlet";
117 case Type::Periodic:
118 return "Periodic";
119 case Type::Neumann:
120 return "Neumann";
121 case Type::NonuniformDirichlet:
122 return "NonuniformDirichlet";
123 case Type::NonuniformNeumann:
124 return "NonuniformNeumann";
125 case Type::TimeDependentNonuniformDirichlet:
126 return "TimeDependentNonuniformDirichlet";
127 case Type::TimeDependentNonuniformNeumann:
128 return "TimeDependentNonuniformNeumann";
129 default:
130 return "UNKNOWN";
131 }
132 }

134private:
135 // Map of boundary conditions and domain boundary for which they correspond to. For a
136 // simple geometry like a square the boundary ids are marked, in order, by x=0, x=max,
137 // y=0, y=max. More complex geometries can have somewhat arbitrary ordering, but will
138 // render some of our assertions moot.
139 std::map<dealii::types::boundary_id, Type> boundary_condition_map;
140
141 // A map of boundary values for dirichlet boundary conditions
142 std::map<dealii::types::boundary_id, double> dirichlet_value_map;
143};

304template <unsigned int dim>
305inline void
306BoundaryParameters<dim>::postprocess_and_validate(
307 std::map<unsigned int, VariableAttributes> &var_attributes)
308{
309 for (const auto &[index, variable] : var_attributes)
310 {
311 // Ensure that boundary conditions are specified for all variables and their
312 // components
313 if (variable.field_info.tensor_rank == FieldInfo::TensorRank::Vector)
314 {
315 for (unsigned int i = 0; i < dim; i++)
316 {
317 // For postprocess and constant fields, the boundary condition is built-in
318 // to whatever the user sets. Thus, we set it to natural.
319 if (variable.is_postprocess() ||
320 variable.get_pde_type() == PDEType::Constant)
321 {
322 set_boundary("Natural", variable.get_field_index(), i);
323 }
324 else
325 {
326 AssertThrow(bc_list.contains(variable.get_field_index()),
327 dealii::ExcMessage("Invalid entry"));
328 AssertThrow(bc_list.at(variable.get_field_index()).contains(i),
329 dealii::ExcMessage("Invalid entry"));
330 AssertThrow(!bc_list.at(variable.get_field_index()).at(i).empty(),
331 dealii::ExcMessage(
332 "Boundary conditions must be specified "
333 "for all components in all vector field."));
334
335 set_boundary(bc_list.at(variable.get_field_index()).at(i),
336 variable.get_field_index(),
337 i);
338 }
339 }
340 }
341 else
342 {
343 if (variable.is_postprocess() || variable.get_pde_type() == PDEType::Constant)
344 {
345 set_boundary("Natural", variable.get_field_index(), 0);
346 }
347 else
348 {
349 AssertThrow(bc_list.contains(variable.get_field_index()),
350 dealii::ExcMessage("Invalid entry"));
351 AssertThrow(bc_list.at(variable.get_field_index()).contains(0),
352 dealii::ExcMessage("Invalid entry"));
353 AssertThrow(!bc_list.at(variable.get_field_index()).at(0).empty(),
354 dealii::ExcMessage("Boundary conditions must be specified "
355 "for all scalar fields."));
356
357 set_boundary(bc_list.at(variable.get_field_index()).at(0),
358 variable.get_field_index(),
359 0);
360 }
361 }
362 }
363
364 // Set periodicity flags
365 periodicity.fill(false);
366 for (const auto &[index, component_map] : boundary_condition_list)
367 {
368 for (const auto &[component, boundary_condition] : component_map)
369 {
370 for (const auto &[domain_id, boundary_type] :
371 boundary_condition.get_boundary_condition_map())
372 {
373 if (boundary_type == BoundaryCondition::Type::Periodic)
374 {
375 periodicity[domain_id / 2] = true;
376 }
377 }
378 }
379 }
380
381 // Validate boundary conditions
382 validate_boundary_conditions();
383
384 // Clear the bc_list now that it's no longer necessary
385 bc_list.clear();
386
387#ifdef ADDITIONAL_OPTIMIZATIONS
388 // Check if any fields are degenerate in terms of boundary conditions
389 // TODO (landinjm): Clean this up
390 for (const auto &[index_1, variable_1] : var_attributes)
391 {
392 // Skip is the degenerate index has already been assigned
393 if (variable_1.get_degenerate_field_index() != Numbers::invalid_index)
394 {
395 continue;
396 }
397 if (variable_1.is_postprocess())
398 {
399 continue;
400 }
401
402 const auto field_type_1 = variable_1.field_info.tensor_rank;
403
404 for (auto &[index_2, variable_2] : var_attributes)
405 {
406 if (variable_2.is_postprocess())
407 {
408 continue;
409 }
410
411 bool is_degenerate = false;
412
413 const auto field_type_2 = variable_2.field_info.tensor_rank;
414
415 is_degenerate = field_type_1 == field_type_2 &&
416 check_degenerate_boundary_conditions(index_1, index_2);
417
418 if (is_degenerate)
419 {
420 ConditionalOStreams::pout_verbose()
421 << "Field " << variable_1.get_name()
422 << " has the same boundary conditions as " << variable_2.get_name()
423 << ". Using optimizations...\n";
424 variable_2.set_degenerate_field_index(index_1);
425 }
426 }
427 }
428#endif
429}

431template <unsigned int dim>
432inline bool
433BoundaryParameters<dim>::check_degenerate_boundary_conditions(
434 const Types::Index &index_1,
435 const Types::Index &index_2) const
436{
437 // If the indices are the same return false
438 if (index_1 == index_2)
439 {
440 return false;
441 }
442
443 Assert(boundary_condition_list.contains(index_1),
444 dealii::ExcMessage("Invalid entry for index = " + std::to_string(index_1)));
445 Assert(boundary_condition_list.contains(index_2),
446 dealii::ExcMessage("Invalid entry for index = " + std::to_string(index_2)));
447
448 bool is_degenerate = false;
449
450 // First check the boundary_condition_list
451 const auto &boundary_condition_1 = boundary_condition_list.at(index_1);
452 const auto &boundary_condition_2 = boundary_condition_list.at(index_2);
453
454 is_degenerate = boundary_condition_1 == boundary_condition_2;
455
456 // Check the pinned points
457 if (pinned_point_list.contains(index_1) && pinned_point_list.contains(index_2))
458 {
459 is_degenerate =
460 is_degenerate && pinned_point_list.at(index_1) == pinned_point_list.at(index_2);
461 }
462
463 return is_degenerate;
464}

466template <unsigned int dim>
467inline void
468BoundaryParameters<dim>::print_parameter_summary() const
469{
470 ConditionalOStreams::pout_summary()
471 << "================================================\n"
472 << " Boundary Parameters\n"
473 << "================================================\n";
474
475 for (const auto &[index, component_map] : boundary_condition_list)
476 {
477 ConditionalOStreams::pout_summary() << "Index: " << index << "\n";
478 for (const auto &[component, boundary_condition] : component_map)
479 {
480 ConditionalOStreams::pout_summary() << " Component: " << component << "\n";
481 for (const auto &[domain_id, boundary_type] :
482 boundary_condition.get_boundary_condition_map())
483 {
484 ConditionalOStreams::pout_summary()
485 << " Boundary id: " << domain_id << " "
486 << "Condition: " << boundary_condition.to_string(boundary_type);
487 if (boundary_type == BoundaryCondition::Type::Dirichlet)
488 {
489 ConditionalOStreams::pout_summary()
490 << " = " << boundary_condition.get_dirichlet_value(domain_id);
491 }
492 ConditionalOStreams::pout_summary() << "\n";
493 }
494 }
495 }
496
497 if (!pinned_point_list.empty())
498 {
499 ConditionalOStreams::pout_summary() << "Pinned field index: ";
500 }
501 for (const auto &[index, point_value_pair] : pinned_point_list)
502 {
503 ConditionalOStreams::pout_summary() << index << "\n"
504 << " Value: ";
505
506 // Handle variant value printing
507 std::visit(
508 [&](const auto &value)
509 {
510 if constexpr (std::is_same_v<std::decay_t<decltype(value)>, double>)
511 {
512 ConditionalOStreams::pout_summary() << value;
513 }
514 else
515 {
516 for (unsigned int i = 0; i < value.size(); ++i)
517 {
518 ConditionalOStreams::pout_summary() << value[i] << " ";
519 }
520 }
521 },
522 point_value_pair.first);
523
524 ConditionalOStreams::pout_summary()
525 << "\n Point: " << point_value_pair.second << "\n";
526 }
527 ConditionalOStreams::pout_summary() << "\n" << std::flush;
528}

530template <unsigned int dim>
531inline void
532BoundaryParameters<dim>::set_boundary(const std::string &bc_string,
533 const Types::Index &index,
534 const unsigned int &component)
535{
536 // Split string
537 auto bc_string_list = dealii::Utilities::split_string_list(bc_string);
538
539 // Check that there is either 1 or 2*dim entries in the vector. This can be changed
540 // later to support other geometries.
541 AssertThrow(bc_string_list.size() == 1 ||
542 bc_string_list.size() == static_cast<std::size_t>(2 * dim),
543 dealii::ExcMessage(
544 "Either 1 or 2*dim boundary conditions must be specified."));
545
546 // If there is only 1 boundary condition resize BC_string_list, copying the first
547 // entry.
548 if (bc_string_list.size() == 1)
549 {
550 bc_string_list.resize(static_cast<std::size_t>(2 * dim), bc_string_list[0]);
551 }
552
553 // Assign boundary condition
554 BoundaryCondition condition;
555 for (unsigned int i = 0; i < (2 * dim); i++)
556 {
557 const std::string dirichlet = "Dirichlet";
558 const std::string neumann = "Neumann";
559
560 if (boost::iequals(bc_string_list[i], "Natural"))
561 {
562 condition.add_boundary_condition(i, BoundaryCondition::Type::Natural);
563 }
564 else if (boost::iequals(bc_string_list[i].substr(0, dirichlet.size()), dirichlet))
565 {
566 condition.add_boundary_condition(i, BoundaryCondition::Type::Dirichlet);
567 std::string dirichlet_value =
568 bc_string_list[i].substr(dirichlet.size() + 1, bc_string_list[i].size());
569 dirichlet_value = dealii::Utilities::trim(dirichlet_value);
570 condition.add_boundary_condition(i,
571 dealii::Utilities::string_to_double(
572 dirichlet_value));
573 }
574 else if (boost::iequals(bc_string_list[i], "Periodic"))
575 {
576 condition.add_boundary_condition(i, BoundaryCondition::Type::Periodic);
577 }
578 else if (boost::iequals(bc_string_list[i].substr(0, neumann.size()), neumann))
579 {
580 AssertThrow(false, FeatureNotImplemented("Neumann boundary conditions"));
581 }
582 else if (boost::iequals(bc_string_list[i], "NonuniformDirichlet"))
583 {
584 condition.add_boundary_condition(i,
585 BoundaryCondition::Type::NonuniformDirichlet);
586 }
587 else if (boost::iequals(bc_string_list[i], "NonuniformNeumann"))
588 {
589 AssertThrow(false,
590 FeatureNotImplemented("Nonuniform neumann boundary conditions"));
591 }
592 else if (boost::iequals(bc_string_list[i], "TimeDependentNonuniformDirichlet"))
593 {
594 condition.add_boundary_condition(
595 i,
596 BoundaryCondition::Type::TimeDependentNonuniformDirichlet);
597 time_dependent_bc_list.insert(index);
598 }
599 else if (boost::iequals(bc_string_list[i], "TimeDependentNonuniformNeumann"))
600 {
601 AssertThrow(false,
602 FeatureNotImplemented(
603 "Time-dependent neumann boundary conditions"));
604 time_dependent_bc_list.insert(index);
605 }
606 else
607 {
608 AssertThrow(false,
609 dealii::ExcMessage("Invalid boundary condition " +
610 bc_string_list[i]));
611 }
612 // If periodic boundary conditions are used, ensure that they are applied on
613 // both sides of the domain.
614 if (i % 2 == 0)
615 {
616 AssertThrow(boost::iequals(bc_string_list[i], "Periodic") ==
617 boost::iequals(bc_string_list[i + 1], "Periodic"),
618 dealii::ExcMessage("Periodic boundary condition must be "
619 "specified on both sides of domain"));
620 }
621 }
622
623 boundary_condition_list[index].emplace(component, condition);
624}

626template <unsigned int dim>
627inline void
628BoundaryParameters<dim>::validate_boundary_conditions() const
629{
630 // Throw a warning if the pinned point is not on a vertex
631 // TODO (landinjm): How do we want to handle this?
632 for (const auto &[index, point_value_pair] : pinned_point_list)
633 {
634 // Disable this TODO (landinjm): Fix
635 /*Assert(point_value_pair.second == dealii::Point<dim>(),
636 dealii::ExcMessage("Pinned point must be on the origin"));*/
637
638 // Validate that vector values have the correct size
639 std::visit(
640 [&](const auto &value)
641 {
642 if constexpr (std::is_same_v<std::decay_t<decltype(value)>,
643 std::vector<double>>)
644 {
645 AssertThrow(value.size() == dim,
646 dealii::ExcMessage("Vector value size must match dimension"));
647 }
648 },
649 point_value_pair.first);
650 }
651
652 // Throw a warning if only some fields have periodic boundary conditions
653 std::vector<bool> periodic_ids(static_cast<dealii::types::boundary_id>(2 * dim), false);
654 for (const auto &[index, component_map] : boundary_condition_list)
655 {
656 for (const auto &[component, boundary_condition] : component_map)
657 {
658 for (const auto &[domain_id, boundary_type] :
659 boundary_condition.get_boundary_condition_map())
660 {
661 if (boundary_type == BoundaryCondition::Type::Periodic)
662 {
663 periodic_ids[domain_id] = true;
664 }
665 }
666 }
667 }
668 for (const auto &[index, component_map] : boundary_condition_list)
669 {
670 for (const auto &[component, boundary_condition] : component_map)
671 {
672 for (const auto &[domain_id, boundary_type] :
673 boundary_condition.get_boundary_condition_map())
674 {
675 if (boundary_type != BoundaryCondition::Type::Periodic &&
676 periodic_ids[domain_id])
677 {
678 // TODO (landinjm): This needs to be fixed to ignore postprocess fields.
679 // Disable it for now.
680 /*AssertThrow(
681 false,
682 dealii::ExcMessage(
683 "All fields for a given domain id (side) must have periodic "
684 "boundary conditions if any field has periodic boundary "
685 "conditions"));*/
686 }
687 }
688 }
689 }
