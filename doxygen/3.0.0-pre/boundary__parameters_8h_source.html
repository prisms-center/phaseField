58 dirichlet_value_map == other.dirichlet_value_map;

97 {
98 Assert(boundary_condition_map.at(boundary_id) == Type::Dirichlet,

106 [[nodiscard]] std::string
107 to_string(Type boundary_type) const
108 {
109 switch (boundary_type)
110 {
111 case Type::UndefinedBoundary:
112 return "UndefinedBoundary";
113 case Type::Natural:
114 return "Natural";
115 case Type::Dirichlet:
116 return "Dirichlet";
117 case Type::Periodic:
118 return "Periodic";
119 case Type::Neumann:
120 return "Neumann";
121 case Type::NonuniformDirichlet:
122 return "NonuniformDirichlet";
123 case Type::NonuniformNeumann:
124 return "NonuniformNeumann";
125 case Type::TimeDependentNonuniformDirichlet:
126 return "TimeDependentNonuniformDirichlet";
127 case Type::TimeDependentNonuniformNeumann:
128 return "TimeDependentNonuniformNeumann";
129 default:
130 return "UNKNOWN";
131 }
132 }

134private:
135 // Map of boundary conditions and domain boundary for which they correspond to. For a
136 // simple geometry like a square the boundary ids are marked, in order, by x=0, x=max,
137 // y=0, y=max. More complex geometries can have somewhat arbitrary ordering, but will
138 // render some of our assertions moot.
139 std::map<dealii::types::boundary_id, Type> boundary_condition_map;
140
141 // A map of boundary values for dirichlet boundary conditions
142 std::map<dealii::types::boundary_id, double> dirichlet_value_map;
143};

278template <unsigned int dim>
279inline void
280BoundaryParameters<dim>::postprocess_and_validate(
281 std::map<unsigned int, VariableAttributes> &var_attributes)
282{
283 for (const auto &[index, variable] : var_attributes)
284 {
285 // Ensure that boundary conditions are specified for all variables and their
286 // components
287 if (variable.get_field_type() == FieldType::Vector)
288 {
289 for (unsigned int i = 0; i < dim; i++)
290 {
291 // For postprocess and constant fields, the boundary condition is built-in
292 // to whatever the user sets. Thus, we set it to natural.
293 if (variable.is_postprocess() ||
294 variable.get_pde_type() == PDEType::Constant)
295 {
296 set_boundary("Natural", variable.get_field_index(), i);
297 }
298 else
299 {
300 AssertThrow(bc_list.contains(variable.get_field_index()),
301 dealii::ExcMessage("Invalid entry"));
302 AssertThrow(bc_list.at(variable.get_field_index()).contains(i),
303 dealii::ExcMessage("Invalid entry"));
304 AssertThrow(!bc_list.at(variable.get_field_index()).at(i).empty(),
305 dealii::ExcMessage(
306 "Boundary conditions must be specified "
307 "for all components in all vector field."));
308
309 set_boundary(bc_list.at(variable.get_field_index()).at(i),
310 variable.get_field_index(),
311 i);
312 }
313 }
314 }
315 else
316 {
317 if (variable.is_postprocess() || variable.get_pde_type() == PDEType::Constant)
318 {
319 set_boundary("Natural", variable.get_field_index(), 0);
320 }
321 else
322 {
323 AssertThrow(bc_list.contains(variable.get_field_index()),
324 dealii::ExcMessage("Invalid entry"));
325 AssertThrow(bc_list.at(variable.get_field_index()).contains(0),
326 dealii::ExcMessage("Invalid entry"));
327 AssertThrow(!bc_list.at(variable.get_field_index()).at(0).empty(),
328 dealii::ExcMessage("Boundary conditions must be specified "
329 "for all scalar fields."));
330
331 set_boundary(bc_list.at(variable.get_field_index()).at(0),
332 variable.get_field_index(),
333 0);
334 }
335 }
336 }
337
338 // Validate boundary conditions
339 validate_boundary_conditions();
340
341 // Clear the bc_list now that it's no longer necessary
342 bc_list.clear();
343
344#ifdef ADDITIONAL_OPTIMIZATIONS
345 // Check if any fields are degenerate in terms of boundary conditions
346 // TODO (landinjm): Clean this up
347 for (const auto &[index_1, variable_1] : var_attributes)
348 {
349 // Skip is the degenerate index has already been assigned
350 if (variable_1.get_degenerate_field_index() != Numbers::invalid_index)
351 {
352 continue;
353 }
354 if (variable_1.is_postprocess())
355 {
356 continue;
357 }
358
359 const auto field_type_1 = variable_1.get_field_type();
360
361 for (auto &[index_2, variable_2] : var_attributes)
362 {
363 if (variable_2.is_postprocess())
364 {
365 continue;
366 }
367
368 bool is_degenerate = false;
369
370 const auto field_type_2 = variable_2.get_field_type();
371
372 is_degenerate = field_type_1 == field_type_2 &&
373 check_degenerate_boundary_conditions(index_1, index_2);
374
375 if (is_degenerate)
376 {
377 ConditionalOStreams::pout_verbose()
378 << "Field " << variable_1.get_name()
379 << " has the same boundary conditions as " << variable_2.get_name()
380 << ". Using optimizations...\n";
381 variable_2.set_degenerate_field_index(index_1);
382 }
383 }
384 }
385#endif
386}

388template <unsigned int dim>
389inline bool
390BoundaryParameters<dim>::check_degenerate_boundary_conditions(
391 const Types::Index &index_1,
392 const Types::Index &index_2) const
393{
394 // If the indices are the same return false
395 if (index_1 == index_2)
396 {
397 return false;
398 }
399
400 Assert(boundary_condition_list.contains(index_1),
401 dealii::ExcMessage("Invalid entry for index = " + std::to_string(index_1)));
402 Assert(boundary_condition_list.contains(index_2),
403 dealii::ExcMessage("Invalid entry for index = " + std::to_string(index_2)));
404
405 bool is_degenerate = false;
406
407 // First check the boundary_condition_list
408 const auto &boundary_condition_1 = boundary_condition_list.at(index_1);
409 const auto &boundary_condition_2 = boundary_condition_list.at(index_2);
410
411 is_degenerate = boundary_condition_1 == boundary_condition_2;
412
413 // Check the pinned points
414 if (pinned_point_list.contains(index_1) && pinned_point_list.contains(index_2))
415 {
416 is_degenerate =
417 is_degenerate && pinned_point_list.at(index_1) == pinned_point_list.at(index_2);
418 }
419
420 return is_degenerate;
421}

423template <unsigned int dim>
424inline void
425BoundaryParameters<dim>::print_parameter_summary() const
426{
427 ConditionalOStreams::pout_summary()
428 << "================================================\n"
429 << " Boundary Parameters\n"
430 << "================================================\n";
431
432 for (const auto &[index, component_map] : boundary_condition_list)
433 {
434 ConditionalOStreams::pout_summary() << "Index: " << index << "\n";
435 for (const auto &[component, boundary_condition] : component_map)
436 {
437 ConditionalOStreams::pout_summary() << " Component: " << component << "\n";
438 for (const auto &[domain_id, boundary_type] :
439 boundary_condition.get_boundary_condition_map())
440 {
441 ConditionalOStreams::pout_summary()
442 << " Boundary id: " << domain_id << " "
443 << "Condition: " << boundary_condition.to_string(boundary_type);
444 if (boundary_type == BoundaryCondition::Type::Dirichlet)
445 {
446 ConditionalOStreams::pout_summary()
447 << " = " << boundary_condition.get_dirichlet_value(domain_id);
448 }
449 ConditionalOStreams::pout_summary() << "\n";
450 }
451 }
452 }
453
454 if (!pinned_point_list.empty())
455 {
456 ConditionalOStreams::pout_summary() << "Pinned field index: ";
457 }
458 for (const auto &[index, point_value_pair] : pinned_point_list)
459 {
460 ConditionalOStreams::pout_summary() << index << "\n"
461 << " Value: ";
462
463 // Handle variant value printing
464 std::visit(
465 [&](const auto &value)
466 {
467 if constexpr (std::is_same_v<std::decay_t<decltype(value)>, double>)
468 {
469 ConditionalOStreams::pout_summary() << value;
470 }
471 else
472 {
473 for (unsigned int i = 0; i < value.size(); ++i)
474 {
475 ConditionalOStreams::pout_summary() << value[i] << " ";
476 }
477 }
478 },
479 point_value_pair.first);
480
481 ConditionalOStreams::pout_summary()
482 << "\n Point: " << point_value_pair.second << "\n";
483 }
484 ConditionalOStreams::pout_summary() << "\n" << std::flush;
485}

487template <unsigned int dim>
488inline void
489BoundaryParameters<dim>::set_boundary(const std::string &bc_string,
490 const Types::Index &index,
491 const unsigned int &component)
492{
493 // Split string
494 auto bc_string_list = dealii::Utilities::split_string_list(bc_string);
495
496 // Check that there is either 1 or 2*dim entries in the vector. This can be changed
497 // later to support other geometries.
498 AssertThrow(bc_string_list.size() == 1 ||
499 bc_string_list.size() == static_cast<std::size_t>(2 * dim),
500 dealii::ExcMessage(
501 "Either 1 or 2*dim boundary conditions must be specified."));
502
503 // If there is only 1 boundary condition resize BC_string_list, copying the first
504 // entry.
505 if (bc_string_list.size() == 1)
506 {
507 bc_string_list.resize(static_cast<std::size_t>(2 * dim), bc_string_list[0]);
508 }
509
510 // Assign boundary condition
511 BoundaryCondition condition;
512 for (unsigned int i = 0; i < (2 * dim); i++)
513 {
514 const std::string dirichlet = "Dirichlet";
515 const std::string neumann = "Neumann";
516
517 if (boost::iequals(bc_string_list[i], "Natural"))
518 {
519 condition.add_boundary_condition(i, BoundaryCondition::Type::Natural);
520 }
521 else if (boost::iequals(bc_string_list[i].substr(0, dirichlet.size()), dirichlet))
522 {
523 condition.add_boundary_condition(i, BoundaryCondition::Type::Dirichlet);
524 std::string dirichlet_value =
525 bc_string_list[i].substr(dirichlet.size() + 1, bc_string_list[i].size());
526 dirichlet_value = dealii::Utilities::trim(dirichlet_value);
527 condition.add_boundary_condition(i,
528 dealii::Utilities::string_to_double(
529 dirichlet_value));
530 }
531 else if (boost::iequals(bc_string_list[i], "Periodic"))
532 {
533 condition.add_boundary_condition(i, BoundaryCondition::Type::Periodic);
534 }
535 else if (boost::iequals(bc_string_list[i].substr(0, neumann.size()), neumann))
536 {
537 AssertThrow(false, FeatureNotImplemented("Neumann boundary conditions"));
538 }
539 else if (boost::iequals(bc_string_list[i], "NonuniformDirichlet"))
540 {
541 condition.add_boundary_condition(i,
542 BoundaryCondition::Type::NonuniformDirichlet);
543 }
544 else if (boost::iequals(bc_string_list[i], "NonuniformNeumann"))
545 {
546 AssertThrow(false,
547 FeatureNotImplemented("Nonuniform neumann boundary conditions"));
548 }
549 else if (boost::iequals(bc_string_list[i], "TimeDependentNonuniformDirichlet"))
550 {
551 condition.add_boundary_condition(
552 i,
553 BoundaryCondition::Type::TimeDependentNonuniformDirichlet);
554 time_dependent_bc_list.insert(index);
555 }
556 else if (boost::iequals(bc_string_list[i], "TimeDependentNonuniformNeumann"))
557 {
558 AssertThrow(false,
559 FeatureNotImplemented(
560 "Time-dependent neumann boundary conditions"));
561 time_dependent_bc_list.insert(index);
562 }
563 else
564 {
565 AssertThrow(false,
566 dealii::ExcMessage("Invalid boundary condition " +
567 bc_string_list[i]));
568 }
569 // If periodic boundary conditions are used, ensure that they are applied on
570 // both sides of the domain.
571 if (i % 2 == 0)
572 {
573 AssertThrow(boost::iequals(bc_string_list[i], "Periodic") ==
574 boost::iequals(bc_string_list[i + 1], "Periodic"),
575 dealii::ExcMessage("Periodic boundary condition must be "
576 "specified on both sides of domain"));
577 }
578 }
579
580 boundary_condition_list[index].emplace(component, condition);
581}

583template <unsigned int dim>
584inline void
585BoundaryParameters<dim>::validate_boundary_conditions() const
586{
587 // Throw a warning if the pinned point is not on a vertex
588 // TODO (landinjm): How do we want to handle this?
589 for (const auto &[index, point_value_pair] : pinned_point_list)
590 {
591 // Disable this TODO (landinjm): Fix
592 /*Assert(point_value_pair.second == dealii::Point<dim>(),
593 dealii::ExcMessage("Pinned point must be on the origin"));*/
594
595 // Validate that vector values have the correct size
596 std::visit(
597 [&](const auto &value)
598 {
599 if constexpr (std::is_same_v<std::decay_t<decltype(value)>,
600 std::vector<double>>)
601 {
602 AssertThrow(value.size() == dim,
603 dealii::ExcMessage("Vector value size must match dimension"));
604 }
605 },
606 point_value_pair.first);
607 }
608
609 // Throw a warning if only some fields have periodic boundary conditions
610 std::vector<bool> periodic_ids(static_cast<dealii::types::boundary_id>(2 * dim), false);
611 for (const auto &[index, component_map] : boundary_condition_list)
612 {
613 for (const auto &[component, boundary_condition] : component_map)
614 {
615 for (const auto &[domain_id, boundary_type] :
616 boundary_condition.get_boundary_condition_map())
617 {
618 if (boundary_type == BoundaryCondition::Type::Periodic)
619 {
620 periodic_ids[domain_id] = true;
621 }
622 }
623 }
624 }
625 for (const auto &[index, component_map] : boundary_condition_list)
626 {
627 for (const auto &[component, boundary_condition] : component_map)
628 {
629 for (const auto &[domain_id, boundary_type] :
630 boundary_condition.get_boundary_condition_map())
631 {
632 if (boundary_type != BoundaryCondition::Type::Periodic &&
633 periodic_ids[domain_id])
634 {
635 // TODO (landinjm): This needs to be fixed to ignore postprocess fields.
636 // Disable it for now.
637 /*AssertThrow(
638 false,
639 dealii::ExcMessage(
640 "All fields for a given domain id (side) must have periodic "
641 "boundary conditions if any field has periodic boundary "
642 "conditions"));*/
643 }
644 }
645 }
646 }
