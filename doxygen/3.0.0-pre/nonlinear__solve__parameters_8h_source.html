81inline void
82NonlinearSolveParameterSet::postprocess_and_validate()
83{
84 for (const auto &[index, nonlinear_solver_parameters] : nonlinear_solve)
85 {
86 AssertThrow(
87 nonlinear_solver_parameters.step_length > 0.0 &&
88 nonlinear_solver_parameters.step_length <= 1.0,
89 dealii::ExcMessage(
90 "Step length must be greater than 0.0 and less than or equal to 1.0"));
91
92 AssertThrow(nonlinear_solver_parameters.tolerance_value > 0,
93 dealii::ExcMessage("Tolerance must be greater than 0.0"));
94 }
95}

97inline void
98NonlinearSolveParameterSet::print_parameter_summary() const
99{
100 if (!nonlinear_solve.empty())
101 {
102 ConditionalOStreams::pout_summary()
103 << "================================================\n"
104 << " Nonlinear Solve Parameters\n"
105 << "================================================\n";
106
107 for (const auto &[index, nonlinear_solver_parameters] : nonlinear_solve)
108 {
109 ConditionalOStreams::pout_summary()
110 << "Index: " << index << "\n"
111 << " Max iterations: " << nonlinear_solver_parameters.max_iterations << "\n"
112 << " Step length: " << nonlinear_solver_parameters.step_length << "\n";
113 }
114
115 ConditionalOStreams::pout_summary() << "\n" << std::flush;
116 }
117}
