104inline void
105CheckpointParameters::postprocess_and_validate(
106 const TemporalDiscretization &temporal_discretization)
107{
108 // If the user has specified a list and we have list checkpoint use that and return
109 // early
110 if (condition == "LIST")
111 {
112 for (const auto &increment : user_checkpoint_list)
113 {
114 checkpoint_list.insert(static_cast<unsigned int>(increment));
115 }
116 return;
117 }
118
119 // If the number of checkpoints is 0 return early
120 if (n_checkpoints == 0)
121 {
122 return;
123 }
124
125 // If the number of outputs is greater than the number of increments, force them to be
126 // equivalent
127 n_checkpoints = std::min(n_checkpoints, temporal_discretization.get_total_increments());
128
129 // If the number of increments is 0, the number of checkpoints should be 0, so we return
130 // early
131 if (temporal_discretization.get_total_increments() == 0)
132 {
133 return;
134 }
135
136 // Determine the output list from the other criteria
137 if (condition == "EQUAL_SPACING")
138 {
139 for (unsigned int iteration = 0;
140 iteration <= temporal_discretization.get_total_increments();
141 iteration += temporal_discretization.get_total_increments() / n_checkpoints)
142 {
143 checkpoint_list.insert(iteration);
144 }
145 }
146 else if (condition == "LOG_SPACING")
147 {
148 checkpoint_list.insert(0);
149 for (unsigned int output = 1; output <= n_checkpoints; output++)
150 {
151 checkpoint_list.insert(static_cast<unsigned int>(std::round(
152 std::pow(static_cast<double>(temporal_discretization.get_total_increments()),
153 static_cast<double>(output) / static_cast<double>(n_checkpoints)))));
154 }
155 }
156 else if (condition == "N_PER_DECADE")
157 {
158 AssertThrow(temporal_discretization.get_total_increments() > 1,
159 dealii::ExcMessage("For n per decaded spaced outputs, the number of "
160 "increments must be greater than 1."));
161
162 checkpoint_list.insert(0);
163 checkpoint_list.insert(1);
164 for (unsigned int iteration = 2;
165 iteration <= temporal_discretization.get_total_increments();
166 iteration++)
167 {
168 const auto decade = static_cast<unsigned int>(std::ceil(std::log10(iteration)));
169 const auto step_size =
170 static_cast<unsigned int>(std::pow(10, decade) / n_checkpoints);
171 if (iteration % step_size == 0)
172 {
173 checkpoint_list.insert(iteration);
174 }
175 }
176 }
177 else
178 {
179 AssertThrow(false, UnreachableCode());
180 }
181}

183inline void
184CheckpointParameters::print_parameter_summary() const
185{
186 ConditionalOStreams::pout_summary()
187 << "================================================\n"
188 << " Checkpoint Parameters\n"
189 << "================================================\n"
190 << "Checkpoint condition: " << condition << "\n"
191 << "Number of checkpoints: " << n_checkpoints << "\n";
192
193 ConditionalOStreams::pout_summary() << "Checkpoint iteration list: ";
194 for (const auto &iteration : checkpoint_list)
195 {

197 }
198 ConditionalOStreams::pout_summary() << "\n\n" << std::flush;
