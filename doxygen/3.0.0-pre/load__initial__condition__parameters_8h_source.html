131inline void
132LoadInitialConditionParameters::postprocess_and_validate()
133{
134 for (const auto &ic_file : ic_files)
135 {
136 // Check that the file variables are the same length as the simulation variables
137 AssertThrow(ic_file.file_variable_names.size() ==
138 ic_file.simulation_variable_names.size(),
139 dealii::ExcMessage("The number of file variables must be the same as "
140 "the number of simulation variables"));
141 }
142
143 // TODO (landinjm): Check that there are no duplicate field names so we don't double
144 // assign.
145}

147inline void
148LoadInitialConditionParameters::print_parameter_summary() const
149{
150 if (read_initial_conditions_from_file)
151 {
152 ConditionalOStreams::pout_summary()
153 << "================================================\n"
154 << " Load IC Parameters\n"
155 << "================================================\n";
156
157 for (const auto &ic_file : ic_files)
158 {
159 ConditionalOStreams::pout_summary()
160 << "File name: " << ic_file.filename << "\n"
161 << "File extension: " << ic_file.file_extension << "\n"
162 << "Grid type: " << ic_file.grid_type << "\n"
163 << "File variable names: ";
164 for (const auto &file_variable_name : ic_file.file_variable_names)
165 {
166 ConditionalOStreams::pout_summary() << file_variable_name << " ";
167 }
168 ConditionalOStreams::pout_summary() << "\n"
169 << "Simulation variable names: ";
170 for (const auto &simulation_variable_name : ic_file.simulation_variable_names)
171 {
172 ConditionalOStreams::pout_summary() << simulation_variable_name << " ";
173 }
174 }
175
176 ConditionalOStreams::pout_summary() << "\n" << std::flush;
177 }
178}
