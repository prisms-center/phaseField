55template <unsigned int dim, typename T>
56inline void
57compute_stress(const dealii::Tensor<2, voigt_tensor_size<dim>, T> &elasticity_tensor,
58 const dealii::Tensor<2, dim, T> &strain,
59 dealii::Tensor<2, dim, T> &stress)
60{
61 dealii::Tensor<1, voigt_tensor_size<dim>, T> sigma;
62 dealii::Tensor<1, voigt_tensor_size<dim>, T> epsilon;
63
64 if constexpr (dim == 3)
65 {
66 const int xx_dir = 0;
67 const int yy_dir = 1;
68 const int zz_dir = 2;
69 const int yz_dir = 3;
70 const int xz_dir = 4;
71 const int xy_dir = 5;
72
73 epsilon[xx_dir] = strain[xx_dir][xx_dir];
74 epsilon[yy_dir] = strain[yy_dir][yy_dir];
75 epsilon[zz_dir] = strain[zz_dir][zz_dir];
76 // In Voigt notation: epsilonngineering shear strain=zz_dir*strain
77 epsilon[yz_dir] = strain[yy_dir][zz_dir] + strain[zz_dir][yy_dir];
78 epsilon[xz_dir] = strain[xx_dir][zz_dir] + strain[zz_dir][xx_dir];
79 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
80
81 // Multiply elasticity_tensor and epsilon to get sigma
82 sigma = elasticity_tensor * epsilon;
83
84 stress[xx_dir][xx_dir] = sigma[xx_dir];
85 stress[yy_dir][yy_dir] = sigma[yy_dir];
86 stress[zz_dir][zz_dir] = sigma[zz_dir];
87
88 stress[yy_dir][zz_dir] = sigma[yz_dir];
89 stress[zz_dir][yy_dir] = sigma[yz_dir];
90
91 stress[xx_dir][zz_dir] = sigma[xz_dir];
92 stress[zz_dir][xx_dir] = sigma[xz_dir];
93
94 stress[xx_dir][yy_dir] = sigma[xy_dir];
95 stress[yy_dir][xx_dir] = sigma[xy_dir];
96 }
97 else if constexpr (dim == 2)
98 {
99 const int xx_dir = 0;
100 const int yy_dir = 1;
101 const int xy_dir = 2;
102
103 epsilon[xx_dir] = strain[xx_dir][xx_dir];
104 epsilon[yy_dir] = strain[yy_dir][yy_dir];
105 // In Voigt notation: epsilonngineering shear strain=xy_dir*strain
106 epsilon[xy_dir] = strain[xx_dir][yy_dir] + strain[yy_dir][xx_dir];
107
108 // Multiply elasticity_tensor and epsilon to get sigma
109 sigma = elasticity_tensor * epsilon;
110
111 stress[xx_dir][xx_dir] = sigma[xx_dir];
112 stress[yy_dir][yy_dir] = sigma[yy_dir];
113 stress[xx_dir][yy_dir] = sigma[xy_dir];
114 stress[yy_dir][xx_dir] = sigma[xy_dir];
115 }
116 else
117 {
118 const int xx_dir = 0;
119
120 stress[xx_dir][xx_dir] = elasticity_tensor[xx_dir][xx_dir] * strain[xx_dir][xx_dir];
121 }
122}

147inline std::string
148eval_flags_to_string(dealii::EvaluationFlags::EvaluationFlags flag)
149{
150 std::string result;
151
152 if ((flag & dealii::EvaluationFlags::EvaluationFlags::values) != 0U)
153 {
154 result += "values";
155 }
156 if ((flag & dealii::EvaluationFlags::EvaluationFlags::gradients) != 0U)
157 {
158 if (!result.empty())
159 {
160 result += " | ";
161 }
162 result += "gradients";
163 }
164 if ((flag & dealii::EvaluationFlags::EvaluationFlags::hessians) != 0U)
165 {
166 if (!result.empty())
167 {
168 result += " | ";
169 }
170 result += "hessians";
171 }
172
173 if (result.empty())
174 {
175 return "nothing";
176 }
177
178 return result;
179}

181template <unsigned int dim, typename number>
182inline std::vector<number>
183dealii_point_to_vector(const dealii::Point<dim, number> &point)
184{
185 static_assert(dim < 4, "We only allow 3 space dimensions");
186
187 std::vector<number> vec(3, 0.0);
188
189 // NOLINTBEGIN(cppcoreguidelines-pro-bounds-constant-array-index)
190
191 for (unsigned int i = 0; i < dim; ++i)
192 {
193 vec[i] = point[i];
194 }
195
196 // NOLINTEND(cppcoreguidelines-pro-bounds-constant-array-index)
197
198 return vec;
199}
