21 explicit GridRefiner(
22 GridRefinementContext<dim, degree, number> &grid_refinement_context)
23 : grid_refinement_context(grid_refinement_context)
24 {
25 fe_values_flags.resize(static_cast<Types::Index>(FieldType::Vector) + 1,
26 dealii::UpdateFlags::update_default);
27
28 for (const auto &criterion : grid_refinement_context.get_user_inputs()
29 .get_spatial_discretization()
30 .get_refinement_criteria())
31 {
32 // Grab the index and field type
33 const Types::Index index = criterion.get_index();
34 const FieldType local_field_type = grid_refinement_context.get_user_inputs()
35 .get_variable_attributes()
36 .at(index)
37 .get_field_type();
38
39 if (criterion.get_criterion() & GridRefinement::RefinementFlags::Value)
40 {
41 fe_values_flags[static_cast<Types::Index>(local_field_type)] |=
42 dealii::UpdateFlags::update_values;
43 }
44 else if (criterion.get_criterion() & GridRefinement::RefinementFlags::Gradient)
45 {
46 fe_values_flags[static_cast<Types::Index>(local_field_type)] |=
47 dealii::UpdateFlags::update_gradients;
48 }
49 }
50 };

90 void
91 init(const std::map<FieldType, dealii::FESystem<dim>> &fe_system)
92 {
93 // Return early if adaptive meshing is disabled
94 if (!grid_refinement_context.get_user_inputs()
95 .get_spatial_discretization()
96 .get_has_adaptivity())
97 {
98 ConditionalOStreams::pout_base() << " grid refinment disabled...\n"
99 << std::flush;
100 return;
101 }
102
103 // Set quadrature rule
104 const dealii::QGaussLobatto<dim> quadrature(degree + 1);
105
106 for (const auto &[field_type, system] : fe_system)
107 {
108 const unsigned int spacedim = field_type == FieldType::Scalar ? 1 : dim;
109
110 // Recreate the FESystem for the field type. I hate to do this, but we have to
111 // ensure that this proper lifetime management and FESystem has it's copy
112 // constructor deleted.
113 fe_systems.try_emplace(field_type,
114 dealii::FE_Q<dim>(dealii::QGaussLobatto<1>(degree + 1)),
115 spacedim);
116
117 // Create the FEValues using the stored FESystem
118 fe_values.try_emplace(field_type,
119 fe_systems.at(field_type),
120 quadrature,
121 fe_values_flags[static_cast<Types::Index>(field_type)]);
122 }
123
124 // Get the number of quadrature points
125 num_quad_points = quadrature.size();
126
127 // Get the min and max global refinements
128 max_refinement = grid_refinement_context.get_user_inputs()
129 .get_spatial_discretization()
130 .get_max_refinement();
131 min_refinement = grid_refinement_context.get_user_inputs()
132 .get_spatial_discretization()
133 .get_min_refinement();
134 }

147 void
148 do_adaptive_refinement()
149 {
150 // Return early if adaptive meshing is disabled
151 if (!grid_refinement_context.get_user_inputs()
152 .get_spatial_discretization()
153 .get_has_adaptivity())
154 {
155 return;
156 }
157
158 Assert(num_quad_points != 0,
159 dealii::ExcMessage("The init() function must be called before trying to "
160 "perform adaptive refinement"));
161
162 // Step 1
163 mark_cells_for_refinement_and_coarsening();
164
165 // Step 2
166 refine_grid();
167
168 // Step 3
169 grid_refinement_context.get_triangulation_handler().reinit();
170 grid_refinement_context.get_dof_handler().reinit(
171 grid_refinement_context.get_triangulation_handler(),
172 grid_refinement_context.get_finite_element_systems(),
173 grid_refinement_context.get_multigrid_info());
174 grid_refinement_context.get_constraint_handler().make_constraints(
175 grid_refinement_context.get_mapping(),
176 grid_refinement_context.get_dof_handler().get_dof_handlers());
177 if (grid_refinement_context.get_multigrid_info().has_multigrid())
178 {
179 const unsigned int min_level =
180 grid_refinement_context.get_multigrid_info().get_mg_min_level();
181 const unsigned int max_level =
182 grid_refinement_context.get_multigrid_info().get_mg_max_level();
183 for (unsigned int level = min_level; level <= max_level; ++level)
184 {
185 grid_refinement_context.get_constraint_handler().make_mg_constraints(
186 grid_refinement_context.get_mapping(),
187 grid_refinement_context.get_dof_handler().get_mg_dof_handlers(level),
188 level);
189 }
190 }
191
192 grid_refinement_context.get_matrix_free_container().template reinit<degree, 1>(
193 grid_refinement_context.get_mapping(),
194 grid_refinement_context.get_dof_handler(),
195 grid_refinement_context.get_constraint_handler(),
196 dealii::QGaussLobatto<1>(degree + 1));
197
198 // Clear the ghosts
199 grid_refinement_context.get_solution_handler().zero_out_ghosts();
200
201 // Reinit the solution vectors
202 grid_refinement_context.get_solution_handler().reinit(
203 grid_refinement_context.get_matrix_free_container());
204
205 // Step 4
206 grid_refinement_context.get_solution_handler().execute_solution_transfer();
207 grid_refinement_context.get_solution_handler().free_solution_transfer();
208 grid_refinement_context.get_solution_handler().reinit_solution_transfer(
209 grid_refinement_context.get_matrix_free_container());
210
211 // Step 6
212 grid_refinement_context.get_invm_handler().recompute_invm();
213 grid_refinement_context.get_element_volume_container().recompute_element_volume();
214
215 // Update the ghosts
216 grid_refinement_context.get_solution_handler().update_ghosts();
217 };

219private:
223 void
224 mark_cells_for_refinement_and_coarsening()
225 {
226 // Create the an object for the refinement criterion at each of the quad points. This
227 // will either contain the value for scalar fields, the magnitude for vector fields,
228 // or the magnitude of the gradient for both of the fields.
229 std::vector<number> values(num_quad_points, 0.0);
230
231 // Clear user flags
232 grid_refinement_context.get_triangulation_handler().clear_user_flags();
233
234 // Loop over the cells provided by the triangulation
235 for (const auto &cell : grid_refinement_context.get_triangulation_handler()
236 .get_triangulation()
237 .active_cell_iterators())
238 {
239 if (cell->is_locally_owned())
240 {
241 // Whether we should refine the cell
242 bool should_refine = false;
243
244 // TODO (landinjm): We can probably avoid checking some of the neighboring
245 // cells when coarsening them
246 for (const auto &criterion : grid_refinement_context.get_user_inputs()
247 .get_spatial_discretization()
248 .get_refinement_criteria())
249 {
250 // Grab the index
251 const Types::Index index = criterion.get_index();
252
253 // Grab the field type
254 const FieldType local_field_type =
255 grid_refinement_context.get_user_inputs()
256 .get_variable_attributes()
257 .at(index)
258 .get_field_type();
259
260 // Grab the DoFHandler iterator
261 const auto dof_iterator = cell->as_dof_handler_iterator(
262 grid_refinement_context.get_dof_handler().get_dof_handler(index));
263
264 // Reinit the cell
265 fe_values.at(local_field_type).reinit(dof_iterator);
266
267 if (criterion.get_criterion() & GridRefinement::RefinementFlags::Value)
268 {
269 if (local_field_type == FieldType::Scalar)
270 {
271 // Get the values for a scalar field
272 fe_values.at(local_field_type)
273 .get_function_values(
274 *grid_refinement_context.get_solution_handler()
275 .get_solution_vector(index, DependencyType::Normal),
276 values);
277 }
278 else
279 {
280 // Get the magnitude of the value for vector fields
281 // TODO (landinjm): Should be zeroing this out?
282 std::vector<dealii::Vector<number>> vector_values(
283 num_quad_points,
284 dealii::Vector<number>(dim));
285 fe_values.at(local_field_type)
286 .get_function_values(
287 *grid_refinement_context.get_solution_handler()
288 .get_solution_vector(index, DependencyType::Normal),
289 vector_values);
290 for (unsigned int q_point = 0; q_point < num_quad_points;
291 ++q_point)
292 {
293 values[q_point] = vector_values[q_point].l2_norm();
294 }
295 }
296
297 // Check if any of the quadrature points meet the refinement criterion
298 for (unsigned int q_point = 0; q_point < num_quad_points; ++q_point)
299 {
300 if (criterion.value_in_open_range(values[q_point]))
301 {
302 should_refine = true;
303 break;
304 }
305 }
306
307 // Exit if we've already determined that the cell has to be refined
308 if (should_refine)
309 {
310 break;
311 }
312 }
313 if (criterion.get_criterion() & GridRefinement::RefinementFlags::Gradient)
314 {
315 if (local_field_type == FieldType::Scalar)
316 {
317 // Get the magnitude of the gradient for a scalar field
318 // TODO (landinjm): Should be zeroing this out?
319 std::vector<dealii::Tensor<1, dim, number>> scalar_gradients(
320 num_quad_points);
321 fe_values.at(local_field_type)
322 .get_function_gradients(
323 *grid_refinement_context.get_solution_handler()
324 .get_solution_vector(index, DependencyType::Normal),
325 scalar_gradients);
326 for (unsigned int q_point = 0; q_point < num_quad_points;
327 ++q_point)
328 {
329 values[q_point] = scalar_gradients[q_point].norm();
330 }
331 }
332 else
333 {
334 // TODO (landinjm): Should be zeroing this out?
335 std::vector<std::vector<dealii::Tensor<1, dim, number>>>
336 vector_gradients(num_quad_points,
337 std::vector<dealii::Tensor<1, dim, number>>(
338 dim));
339 fe_values.at(local_field_type)
340 .get_function_gradients(
341 *grid_refinement_context.get_solution_handler()
342 .get_solution_vector(index, DependencyType::Normal),
343 vector_gradients);
344 for (unsigned int q_point = 0; q_point < num_quad_points;
345 ++q_point)
346 {
347 dealii::Vector<number> vector_gradient_component_magnitude(
348 dim);
349 for (unsigned int dimension = 0; dimension < dim; dimension++)
350 {
351 vector_gradient_component_magnitude[dimension] =
352 vector_gradients[q_point][dimension].norm();
353 }
354 values[q_point] =
355 vector_gradient_component_magnitude.l2_norm();
356 }
357 }
358
359 // Check if any of the quadrature points meet the refinement criterion
360 for (unsigned int q_point = 0; q_point < num_quad_points; ++q_point)
361 {
362 if (criterion.gradient_magnitude_above_threshold(values[q_point]))
363 {
364 should_refine = true;
365 break;
366 }
367 }
368
369 // Exit if we've already determined that the cell has to be refined
370 if (should_refine)
371 {
372 break;
373 }
374 }
375 }
376
377 Assert(cell->level() > 0,
378 dealii::ExcMessage("Cell refinement level is less than one, which "
379 "will lead to underflow."));
380 const auto cell_refinement = static_cast<unsigned int>(cell->level());
381 if (should_refine && cell_refinement < max_refinement)
382 {
383 cell->set_user_flag();
384 cell->clear_coarsen_flag();
385 cell->set_refine_flag();
386 }
387 if (should_refine)
388 {
389 cell->set_user_flag();
390 cell->clear_coarsen_flag();
391 }
392 if (!should_refine && cell_refinement > min_refinement &&
393 !cell->user_flag_set())
394 {
395 cell->set_coarsen_flag();
396 }
397 }
398 }

405 refine_grid()
406 {
407 // Prepare for grid refinement
408 grid_refinement_context.get_triangulation_handler().prepare_for_grid_refinement();
409
410 // Prepare the solution transfer objects
411 grid_refinement_context.get_solution_handler().prepare_for_solution_transfer();
412
413 // Execute grid refinement
414 grid_refinement_context.get_triangulation_handler().execute_grid_refinement();
415 }
