30 fe_values_flags.resize(static_cast<Types::Index>(FieldInfo::TensorRank::Vector),

33 for (const auto &criterion : grid_refinement_context.get_user_inputs()

37 // Grab the index and field type

96 void
97 init(const std::map<FieldInfo::TensorRank, dealii::FESystem<dim>> &fe_system)
98 {
99 // Return early if adaptive meshing is disabled
100 if (!grid_refinement_context.get_user_inputs()
101 .get_spatial_discretization()
102 .get_has_adaptivity())
103 {
104 ConditionalOStreams::pout_base() << " grid refinement disabled...\n"
105 << std::flush;
106 return;
107 }
108
109 // Set quadrature rule
110 const dealii::QGaussLobatto<dim> quadrature(degree + 1);
111
112 for (const auto &[field_type, system] : fe_system)
113 {
114 const unsigned int spacedim =
115 field_type == FieldInfo::TensorRank::Scalar ? 1 : dim;
116
117 // Recreate the FESystem for the field type. I hate to do this, but we have to
118 // ensure that this proper lifetime management and FESystem has it's copy
119 // constructor deleted.
120 fe_systems.try_emplace(field_type,
121 dealii::FE_Q<dim>(dealii::QGaussLobatto<1>(degree + 1)),
122 spacedim);
123
124 // Create the FEValues using the stored FESystem
125 fe_values.try_emplace(field_type,
126 fe_systems.at(field_type),
127 quadrature,
128 fe_values_flags[static_cast<Types::Index>(field_type) - 1]);
129 }
130
131 // Get the number of quadrature points
132 num_quad_points = quadrature.size();
133
134 // Get the min and max global refinements
135 max_refinement = grid_refinement_context.get_user_inputs()
136 .get_spatial_discretization()
137 .get_max_refinement();
138 min_refinement = grid_refinement_context.get_user_inputs()
139 .get_spatial_discretization()
140 .get_min_refinement();
141 }

154 void
155 do_adaptive_refinement()
156 {
157 // Return early if adaptive meshing is disabled
158 if (!grid_refinement_context.get_user_inputs()
159 .get_spatial_discretization()
160 .get_has_adaptivity())
161 {
162 return;
163 }
164
165 Assert(num_quad_points != 0,
166 dealii::ExcMessage("The init() function must be called before trying to "
167 "perform adaptive refinement"));
168 // Step 1
169 mark_cells_for_refinement_and_coarsening();
170 bool first_iteration = true;
171 while (
172 dealii::Utilities::MPI::logical_or(mark_cells_for_refinement(), MPI_COMM_WORLD) ||
173 first_iteration)
174 {
175 first_iteration = false;
176 for (auto &[field_index, solution] :
177 grid_refinement_context.get_solution_handler().get_solution_vector())
178 {
179 solution->update_ghost_values();
180 }
181
182 // Step 2
183 refine_grid();
184
185 // Step 3
186 grid_refinement_context.get_triangulation_handler().reinit();
187 grid_refinement_context.get_dof_handler().reinit(
188 grid_refinement_context.get_triangulation_handler(),
189 grid_refinement_context.get_finite_element_systems(),
190 grid_refinement_context.get_multigrid_info());
191 grid_refinement_context.get_constraint_handler().make_constraints(
192 grid_refinement_context.get_mapping(),
193 grid_refinement_context.get_dof_handler().get_dof_handlers());
194 if (grid_refinement_context.get_multigrid_info().has_multigrid())
195 {
196 const unsigned int min_level =
197 grid_refinement_context.get_multigrid_info().get_mg_min_level();
198 const unsigned int max_level =
199 grid_refinement_context.get_multigrid_info().get_mg_max_level();
200 for (unsigned int level = min_level; level <= max_level; ++level)
201 {
202 grid_refinement_context.get_constraint_handler().make_mg_constraints(
203 grid_refinement_context.get_mapping(),
204 grid_refinement_context.get_dof_handler().get_mg_dof_handlers(level),
205 level);
206 }
207 }
208
209 grid_refinement_context.get_matrix_free_container().template reinit<degree, 1>(
210 grid_refinement_context.get_mapping(),
211 grid_refinement_context.get_dof_handler(),
212 grid_refinement_context.get_constraint_handler(),
213 dealii::QGaussLobatto<1>(degree + 1));
214
215 grid_refinement_context.get_solution_handler().reinit(
216 grid_refinement_context.get_matrix_free_container());
217
218 // Step 4
219 grid_refinement_context.get_solution_handler().execute_solution_transfer();
220
221 // Step 6
222 grid_refinement_context.get_invm_handler().recompute_invm();
223 grid_refinement_context.get_element_volume_container().recompute_element_volume();
224 }
225 }

249 void
250 mark_cells_for_refinement_and_coarsening()
251 {
252 // Create the an object for the refinement criterion at each of the quad points. This
253 // will either contain the value for scalar fields, the magnitude for vector fields,
254 // or the magnitude of the gradient for both of the fields.
255 std::vector<number> values(num_quad_points, 0.0);
256
257 // Clear user flags
258 grid_refinement_context.get_triangulation_handler().clear_user_flags();
259
260 // Loop over the cells provided by the triangulation
261 for (const auto &cell : grid_refinement_context.get_triangulation_handler()
262 .get_triangulation()
263 .active_cell_iterators())
264 {
265 if (cell->is_locally_owned())
266 {
267 // Whether we should refine the cell
268 bool should_refine = false;
269
270 // TODO (landinjm): We can probably avoid checking some of the neighboring
271 // cells when coarsening them
272 for (const auto &criterion : grid_refinement_context.get_user_inputs()
273 .get_spatial_discretization()
274 .get_refinement_criteria())
275 {
276 // Grab the index
277 const Types::Index index = criterion.get_index();
278
279 // Grab the field type
280 const FieldInfo::TensorRank local_field_type =
281 grid_refinement_context.get_user_inputs()
282 .get_variable_attributes()
283 .at(index)
284 .field_info.tensor_rank;
285
286 // Grab the DoFHandler iterator
287 const auto dof_iterator = cell->as_dof_handler_iterator(
288 grid_refinement_context.get_dof_handler().get_dof_handler(index));
289
290 // Reinit the cell
291 fe_values.at(local_field_type).reinit(dof_iterator);
292
293 if (criterion.get_criterion() & GridRefinement::RefinementFlags::Value)
294 {
295 if (local_field_type == FieldInfo::TensorRank::Scalar)
296 {
297 // Get the values for a scalar field
298 fe_values.at(local_field_type)
299 .get_function_values(
300 *grid_refinement_context.get_solution_handler()
301 .get_solution_vector(index, DependencyType::Normal),
302 values);
303 }
304 else
305 {
306 // Get the magnitude of the value for vector fields
307 // TODO (landinjm): Should be zeroing this out?
308 std::vector<dealii::Vector<number>> vector_values(
309 num_quad_points,
310 dealii::Vector<number>(dim));
311 fe_values.at(local_field_type)
312 .get_function_values(
313 *grid_refinement_context.get_solution_handler()
314 .get_solution_vector(index, DependencyType::Normal),
315 vector_values);
316 for (unsigned int q_point = 0; q_point < num_quad_points;
317 ++q_point)
318 {
319 values[q_point] = vector_values[q_point].l2_norm();
320 }
321 }
322
323 // Check if any of the quadrature points meet the refinement criterion
324 for (unsigned int q_point = 0; q_point < num_quad_points; ++q_point)
325 {
326 if (criterion.value_in_open_range(values[q_point]))
327 {
328 should_refine = true;
329 break;
330 }
331 }
332
333 // Exit if we've already determined that the cell has to be refined
334 if (should_refine)
335 {
336 break;
337 }
338 }
339 if (criterion.get_criterion() & GridRefinement::RefinementFlags::Gradient)
340 {
341 if (local_field_type == FieldInfo::TensorRank::Scalar)
342 {
343 // Get the magnitude of the gradient for a scalar field
344 // TODO (landinjm): Should be zeroing this out?
345 std::vector<dealii::Tensor<1, dim, number>> scalar_gradients(
346 num_quad_points);
347 fe_values.at(local_field_type)
348 .get_function_gradients(
349 *grid_refinement_context.get_solution_handler()
350 .get_solution_vector(index, DependencyType::Normal),
351 scalar_gradients);
352 for (unsigned int q_point = 0; q_point < num_quad_points;
353 ++q_point)
354 {
355 values[q_point] = scalar_gradients[q_point].norm();
356 }
357 }
358 else
359 {
360 // TODO (landinjm): Should be zeroing this out?
361 std::vector<std::vector<dealii::Tensor<1, dim, number>>>
362 vector_gradients(num_quad_points,
363 std::vector<dealii::Tensor<1, dim, number>>(
364 dim));
365 fe_values.at(local_field_type)
366 .get_function_gradients(
367 *grid_refinement_context.get_solution_handler()
368 .get_solution_vector(index, DependencyType::Normal),
369 vector_gradients);
370 for (unsigned int q_point = 0; q_point < num_quad_points;
371 ++q_point)
372 {
373 dealii::Vector<number> vector_gradient_component_magnitude(
374 dim);
375 for (unsigned int dimension = 0; dimension < dim; dimension++)
376 {
377 vector_gradient_component_magnitude[dimension] =
378 vector_gradients[q_point][dimension].norm();
379 }
380 values[q_point] =
381 vector_gradient_component_magnitude.l2_norm();
382 }
383 }
384
385 // Check if any of the quadrature points meet the refinement criterion
386 for (unsigned int q_point = 0; q_point < num_quad_points; ++q_point)
387 {
388 if (criterion.gradient_magnitude_above_threshold(values[q_point]))
389 {
390 should_refine = true;
391 break;
392 }
393 }
394
395 // Exit if we've already determined that the cell has to be refined
396 if (should_refine)
397 {
398 break;
399 }
400 }
401 }
402
403 Assert(cell->level() > 0,
404 dealii::ExcMessage("Cell refinement level is less than one, which "
405 "will lead to underflow."));
406 const auto cell_refinement = static_cast<unsigned int>(cell->level());
407 if (should_refine && cell_refinement < max_refinement)
408 {
409 cell->set_user_flag();
410 cell->clear_coarsen_flag();
411 cell->set_refine_flag();
412 }
413 if (should_refine)
414 {
415 cell->set_user_flag();
416 cell->clear_coarsen_flag();
417 }
418 if (!should_refine && cell_refinement > min_refinement &&
419 !cell->user_flag_set())
420 {
421 cell->set_coarsen_flag();
422 }
423 }
424 }
425 }

433 bool
434 mark_cells_for_refinement()
435 {
436 bool any_cell_marked = false;
437 for (const auto &cell : grid_refinement_context.get_triangulation_handler()
438 .get_triangulation()
439 .active_cell_iterators())
440 {
441 if (cell->is_locally_owned())
442 {
443 const auto cell_refinement = static_cast<unsigned int>(cell->level());
444 if (std::any_of(
445 marker_functions.begin(),
446 marker_functions.end(),
447 [&](const std::shared_ptr<const CellMarkerBase<dim>> &marker_function)
448 {
449 return marker_function->flag(*cell,
450 grid_refinement_context.get_user_inputs()
451 .get_temporal_discretization());
452 }))
453 {
454 cell->set_user_flag();
455 cell->clear_coarsen_flag();
456 if (cell_refinement < max_refinement)
457 {
458 cell->set_refine_flag();
459 any_cell_marked = true;
460 }
461 }
462 }
463 }
464 return any_cell_marked;
465 }
