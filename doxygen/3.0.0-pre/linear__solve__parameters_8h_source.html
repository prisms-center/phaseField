21struct LinearSolverParameters
22{
23public:
24 // Solver tolerance
25 double tolerance = Defaults::tolerance;
26
27 // Solver tolerance type
28 SolverToleranceType tolerance_type = SolverToleranceType::RelativeResidualChange;
29
30 // Max number of iterations for the linear solve
31 unsigned int max_iterations = Defaults::iterations;
32
33 // Preconditioner
34 PreconditionerType preconditioner = PreconditionerType::GMG;
35
36 // Smoothing range for eigenvalues. This denotes the lower bound of eigenvalues that are
37 // smoothed [1.2 λ^max / smoothing_range, 1.2 λ^max], where λ^max is the estimated
38 // maximum eigenvalue. A choice between 5 and 20 is usually useful when the
39 // preconditioner is used as a smoother in multigrid.
40 double smoothing_range = Defaults::smoothing_range;
41
42 // Polynomial degree for the Chebyshev smoother
43 unsigned int smoother_degree = Defaults::smoother_degree;
44
45 // Maximum number of CG iterations used to find the maximum eigenvalue
46 unsigned int eig_cg_n_iterations = Defaults::eig_cg_n_iterations;
47
48 // The minimum multigrid level
49 unsigned int min_mg_level = 0;
50};

130inline void
131LinearSolveParameters::print_parameter_summary() const
132{
133 if (!linear_solve.empty())
134 {
135 ConditionalOStreams::pout_summary()
136 << "================================================\n"
137 << " Linear Solve Parameters\n"
138 << "================================================\n";
139
140 for (const auto &[index, linear_solver_parameters] : linear_solve)
141 {
142 ConditionalOStreams::pout_summary()
143 << "Index: " << index << "\n"
144 << " Tolerance: " << linear_solver_parameters.tolerance << "\n"
145 << " Type: " << to_string(linear_solver_parameters.tolerance_type) << "\n"
146 << " Max iterations: " << linear_solver_parameters.max_iterations << "\n"
147 << " Preconditioner: " << to_string(linear_solver_parameters.preconditioner)
148 << "\n";
149
150 if (linear_solver_parameters.preconditioner == PreconditionerType::GMG)
151 {
152 ConditionalOStreams::pout_summary()
153 << " Smoothing range: " << linear_solver_parameters.smoothing_range
154 << "\n"
155 << " Smoother degree: " << linear_solver_parameters.smoother_degree
156 << "\n"
157 << " Max eigenvalue CG iterations: "
158 << linear_solver_parameters.eig_cg_n_iterations << "\n"
159 << "Min multigrid level: " << linear_solver_parameters.min_mg_level
160 << "\n";
161 }
162 }
163
164 ConditionalOStreams::pout_summary() << "\n" << std::flush;
165 }
166}
