181inline void
182OutputParameters::postprocess_and_validate(
183 const TemporalDiscretization &temporal_discretization)
184{
185 // If the user has specified a list and we have list output use that and return early
186 if (condition == "LIST")
187 {
188 for (const auto &increment : user_output_list)
189 {
190 output_list.insert(static_cast<unsigned int>(increment));
191 }
192 return;
193 }
194
195 // If the number of outputs is 0 return early
196 if (n_outputs == 0)
197 {
198 return;
199 }
200
201 // If the number of outputs is greater than the number of increments, force them to be
202 // equivalent
203 n_outputs = std::min(n_outputs, temporal_discretization.get_total_increments());
204
205 // If the number of increments is 0, we only output the initial condition. We can set
206 // that and return early.
207 if (temporal_discretization.get_total_increments() == 0)
208 {
209 output_list.insert(0);
210 return;
211 }
212
213 // Determine the output list from the other criteria
214 if (condition == "EQUAL_SPACING")
215 {
216 for (unsigned int iteration = 0;
217 iteration <= temporal_discretization.get_total_increments();
218 iteration += temporal_discretization.get_total_increments() / n_outputs)
219 {
220 output_list.insert(iteration);
221 }
222 }
223 else if (condition == "LOG_SPACING")
224 {
225 output_list.insert(0);
226 for (unsigned int output = 1; output <= n_outputs; output++)
227 {
228 output_list.insert(static_cast<unsigned int>(std::round(
229 std::pow(static_cast<double>(temporal_discretization.get_total_increments()),
230 static_cast<double>(output) / static_cast<double>(n_outputs)))));
231 }
232 }
233 else if (condition == "N_PER_DECADE")
234 {
235 AssertThrow(temporal_discretization.get_total_increments() > 1,
236 dealii::ExcMessage("For n per decaded spaced outputs, the number of "
237 "increments must be greater than 1."));
238
239 output_list.insert(0);
240 output_list.insert(1);
241 for (unsigned int iteration = 2;
242 iteration <= temporal_discretization.get_total_increments();
243 iteration++)
244 {
245 const auto decade = static_cast<unsigned int>(std::ceil(std::log10(iteration)));
246 const auto step_size =
247 static_cast<unsigned int>(std::pow(10, decade) / n_outputs);
248 if (iteration % step_size == 0)
249 {
250 output_list.insert(iteration);
251 }
252 }
253 }
254 else
255 {
256 AssertThrow(false, UnreachableCode());
257 }
258}

260inline void
261OutputParameters::print_parameter_summary() const
262{
263 ConditionalOStreams::pout_summary()
264 << "================================================\n"
265 << " Output Parameters\n"
266 << "================================================\n"
267 << "Output file type: " << file_type << "\n"
268 << "Output file name: " << file_name << "\n"
269 << "Output subdivisions: " << patch_subdivisions << "\n"
270 << "Print output period: " << print_output_period << "\n"
271 << "Output condition: " << condition << "\n"
272 << "Number of outputs: " << n_outputs << "\n"
273 << "Print timing info: " << bool_to_string(print_timing_with_output) << "\n";
274
275 ConditionalOStreams::pout_summary() << "Output iteration list: ";
276 for (const auto &iteration : output_list)
277 {
278 ConditionalOStreams::pout_summary() << iteration << " ";
279 }
280 ConditionalOStreams::pout_summary() << "\n\n" << std::flush;
281}
