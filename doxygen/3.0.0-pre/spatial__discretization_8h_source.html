132template <unsigned int dim>
133class RectangularMesh : public Mesh<dim>
134{
138 RectangularMesh(dealii::Tensor<1, dim, double> _size,
139 std::vector<unsigned int> _subdivisions)
140 : size(_size)
141 {
142 Assert(_subdivisions.size() == dim,
143 dealii::ExcMessage(
144 "Subdivisions vector size must match the number of dimensions"));
145 subdivisions = std::move(_subdivisions);
146
147 // If the x direction is greater than 0, we set the mesh as initialized.
148 // TODO (landinjm): Check that the other directions are also greater than 0.
149 if (size[0] > 0.0)
150 {
151 this->Mesh<dim>::set_initialized();
152 }
153 };
154
158 void
159 generate_mesh(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation) override
160 {
161 // Call base class generate mesh to check initialization
162 this->Mesh<dim>::generate_mesh(triangulation);
163
164 dealii::GridGenerator::subdivided_hyper_rectangle(*triangulation,
165 subdivisions,
166 dealii::Point<dim>(),
167 dealii::Point<dim>(size));
168 };
169
173 void
174 mark_boundaries(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation,
175 const typename BoundaryParameters<dim>::BoundaryConditionMap
176 &boundary_condition_list) override
177 {
178 // Call base class mark boundaries to check that the mesh has been generated
179 this->Mesh<dim>::mark_boundaries(triangulation, boundary_condition_list);
180
181 // Check that the user has not specified extra boundary conditions
182 for (const auto &[index, boundary_conditions] : boundary_condition_list)
183 {
184 AssertThrow(boundary_conditions.get_boundary_condition_map().size() == (2 * dim),
185 dealii::ExcMessage(
186 "Rectangular meshes only have 2*dim boundaries. Please ensure that "
187 "you have not specified extra boundary conditions."));
188 }
189
190 // Loop through the cells
191 for (const auto &cell : triangulation->active_cell_iterators())
192 {
193 // Mark the faces (faces_per_cell = 2*dim)
194 for (unsigned int face_number = 0;
195 face_number < dealii::GeometryInfo<dim>::faces_per_cell;
196 ++face_number)
197 {
198 // Direction for quad and hex cells
199 auto direction = static_cast<unsigned int>(std::floor(face_number / 2));
200
201 // Mark the boundary id for x=0, y=0, z=0 and x=max, y=max, z=max
202 if (std::fabs(cell->face(face_number)->center()(direction) - 0) <
203 Defaults::mesh_tolerance ||
204 std::fabs(cell->face(face_number)->center()(direction) -
205 (size[direction])) < Defaults::mesh_tolerance)
206 {
207 cell->face(face_number)->set_boundary_id(face_number);
208 }
209 }
210 }
211 };
212
216 void
217 mark_periodic(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation,
218 const typename BoundaryParameters<dim>::BoundaryConditionMap
219 &boundary_condition_list) override
220 {
221 // Call base class mark boundaries to check that the mesh has been generated and
222 // boundaries marked
223 this->Mesh<dim>::mark_periodic(triangulation, boundary_condition_list);
224
225 // TODO (landinjm): Add some assertions here
226
227 // Add periodicity in the triangulation where specified in the boundary conditions.
228 // Note
229 // that if one field is periodic all others should be as well.
230 for (const auto &[index, boundary_conditions] : boundary_condition_list)
231 {
232 for (const auto &[component, condition] : boundary_conditions)
233 {
234 for (const auto &[boundary_id, boundary_type] :
235 condition.get_boundary_condition_map())
236 {
237 if (boundary_type == BoundaryCondition::Type::Periodic)
238 {
239 // Skip boundary ids that are odd since those map to the even faces
240 if (boundary_id % 2 != 0)
241 {
242 continue;
243 }
244
245 // Create a vector of matched pairs that we fill and enforce upon the
246 // constaints
247 std::vector<dealii::GridTools::PeriodicFacePair<
248 typename Mesh<dim>::Triangulation::cell_iterator>>
249 periodicity_vector;
250
251 // Determine the direction
252 const auto direction =
253 static_cast<unsigned int>(std::floor(boundary_id / dim));
254
255 // Collect the matched pairs on the coarsest level of the mesh
256 dealii::GridTools::collect_periodic_faces(*triangulation,
257 boundary_id,
258 boundary_id + 1,
259 direction,
260 periodicity_vector);
261
262 // Set constraints
263 triangulation->add_periodicity(periodicity_vector);
264 }
265 }
266 }
267 }
268 };
269
273 [[nodiscard]] double
274 get_size(unsigned int direction) const
275 {
276 Assert(direction < dim,
277 dealii::ExcMessage("Direction must be less than the number of dimensions"));
278 Assert(size[direction] > 0.0,
279 dealii::ExcMessage("Size in the specified direction must be greater than zero "
280 "for rectangular meshes"));
281 return size[direction];
282 };
283
287 [[nodiscard]] unsigned int
288 get_subdivisions(unsigned int direction) const
289 {
290 Assert(direction < dim,

292 Assert(subdivisions[direction] > 0,

295 "for rectangular meshes"));
296 return subdivisions[direction];
297 }
298
299private:
303 dealii::Tensor<1, dim, double> size;
304
308 std::vector<unsigned int> subdivisions = std::vector<unsigned int>(dim, 1);
309};

314template <unsigned int dim>
315class SphericalMesh
316{
317public:
321 explicit SphericalMesh(double _radius)
322 : radius(_radius)
323 {
324 if (radius > 0.0)
325 {
326 this->Mesh<dim>::set_initialized();
327 }
328 };
329
333 void
334 generate_mesh(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation) override
335 {
336 // Call base class generate mesh to check initialization
337 this->Mesh<dim>::generate_mesh(triangulation);
338
339 AssertThrow(dim != 1, dealii::ExcMessage("Spherical mesh not valid in 1D"));
340 dealii::GridGenerator::hyper_ball(*triangulation, dealii::Point<dim>(), radius);
341 };
342
346 void
347 mark_boundaries(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation,
348 const typename BoundaryParameters<dim>::BoundaryConditionMap
349 &boundary_condition_list) override
350 {
351 // Call base class mark boundaries to check that the mesh has been generated
352 this->Mesh<dim>::mark_boundaries(triangulation, boundary_condition_list);
353
354 for (const auto &[index, boundary_conditions] : boundary_condition_list)
355 {
356 // There are no special boundary ids to mark for a spherical mesh. Check that the
357 // user has specified extra boundary conditions.
358 AssertThrow(boundary_conditions.get_boundary_condition_map().size() == 1,
359 dealii::ExcMessage("Spherical meshes only have one boundary. Please "
360 "ensure that you have not specified "
361 "extra boundary conditions."));
362 }
363 };
364
368 void
369 mark_periodic(std::shared_ptr<typename Mesh<dim>::Triangulation> triangulation,
370 const typename BoundaryParameters<dim>::BoundaryConditionMap
371 &boundary_condition_list) override
372 {
373 // Call base class mark boundaries to check that the mesh has been generated and
374 // boundaries marked
375 this->Mesh<dim>::mark_periodic(triangulation, boundary_condition_list);
376
377 // There are no periodic boundaries for a spherical mesh. Check that the user has not
378 // specified any.
379 for (const auto &[index, boundary_conditions] : boundary_condition_list)
380 {
381 AssertThrow(boundary_conditions.get_boundary_condition_map().begin()->second !=
382 BoundaryCondition::Type::Periodic,
383 dealii::ExcMessage("Spherical meshes cannot have periodic boundary "
384 "conditions."));
385 }
386 };
387
391 [[nodiscard]] double
392 get_radius() const
393 {
394 Assert(radius > 0.0,
395 dealii::ExcMessage("Spherical mesh radius must be greater than zero."));
396 return radius;
397 }
398
399private:
403 double radius = 0.0;
404};

664template <unsigned int dim>
665inline void
666SpatialDiscretization<dim>::postprocess_and_validate()
667{
668 // Assign the triangulation type
669 if (radius != 0.0 && size.norm() == 0.0)
670 {
671 type = TriangulationType::Spherical;
672 }
673 else if (radius == 0.0 && size.norm() != 0.0)
674 {
675 type = TriangulationType::Rectangular;
676 }
677 else
678 {
679 AssertThrow(false, UnreachableCode());
680 }
681
682 // Check that AMR is not enabled for 1D
683 AssertThrow(
684 (!has_adaptivity || dim != 1),
685 dealii::ExcMessage(
686 "Adaptive meshing for the matrix-free method is not currently supported."));
687
688 // Some check if AMR is enabled
689 if (has_adaptivity)
690 {
691 // Check that the minimum and maximum refinement levels are valid
692 AssertThrow((max_refinement >= global_refinement),
693 dealii::ExcMessage(
694 "The maximum refinement level must be greater than or equal to the "
695 "global refinement level."));
696 AssertThrow((min_refinement <= global_refinement),
697 dealii::ExcMessage(
698 "The minimum refinement level must be less than or equal to the "
699 "global refinement level."));
700 AssertThrow((max_refinement >= min_refinement),
701 dealii::ExcMessage(
702 "The maximum refinement level must be greater than or equal to the "
703 "minimum refinement level."));
704
705 // Check that the refinement criteria are valid for the lower and upper bounds
706 for (const auto &criterion : refinement_criteria)
707 {
708 AssertThrow((criterion.get_value_lower_bound() <=
709 criterion.get_value_upper_bound()),
710 dealii::ExcMessage(
711 "The lower bound of the value-based refinement "
712 "criteria must be less than or equal to the upper bound."));
713 }
714 }
715}

717template <unsigned int dim>
718inline void
719SpatialDiscretization<dim>::print_parameter_summary() const
720{
721 ConditionalOStreams::pout_summary()
722 << "================================================\n"
723 << " Spatial Discretization\n"
724 << "================================================\n";
725
726 if (type == TriangulationType::Spherical)
727 {
728 ConditionalOStreams::pout_summary() << "Domain radius: " << radius << "\n";
729 }
730 else if (type == TriangulationType::Rectangular)
731 {
732 if constexpr (dim == 1)
733 {
734 ConditionalOStreams::pout_summary()
735 << "Domain size: x=" << size[0] << "\n"
736 << "Subdivisions: x=" << subdivisions[0] << "\n";
737 }
738 else if constexpr (dim == 2)
739 {
740 ConditionalOStreams::pout_summary()
741 << "Domain size: x=" << size[0] << ", y=" << size[1] << "\n"
742 << "Subdivisions: x=" << subdivisions[0] << ", y=" << subdivisions[1] << "\n";
743 }
744 else if constexpr (dim == 3)
745 {
746 ConditionalOStreams::pout_summary()
747 << "Domain size: x=" << size[0] << ", y=" << size[1] << ", z=" << size[2]
748 << "\n"
749 << "Subdivisions: x=" << subdivisions[0] << ", y=" << subdivisions[1]
750 << ", z=" << subdivisions[2] << "\n";
751 }
752 }
753 else
754 {
755 AssertThrow(false, UnreachableCode());
756 }
757
758 ConditionalOStreams::pout_summary()
759 << "Global refinement: " << global_refinement << "\n"
760 << "Degree: " << degree << "\n"
761 << "Adaptivity enabled: " << bool_to_string(has_adaptivity) << "\n"
762 << "Max refinement: " << max_refinement << "\n"
763 << "Min refinement: " << min_refinement << "\n"
764 << "Remeshing period: " << remeshing_period << "\n";
765
766 if (!refinement_criteria.empty())
767 {
768 ConditionalOStreams::pout_summary() << "Refinement criteria:\n";
769 }
770 for (const auto &criterion : refinement_criteria)
771 {
772 ConditionalOStreams::pout_summary()
773 << " Criterion type: " << criterion.criterion_to_string() << "\n"
774 << " Value lower bound: " << criterion.get_value_lower_bound() << "\n"
775 << " Value upper bound: " << criterion.get_value_upper_bound() << "\n"
776 << " Gradient lower bound: " << criterion.get_gradient_lower_bound() << "\n\n";
777 }
778 ConditionalOStreams::pout_summary() << "\n" << std::flush;
779}
