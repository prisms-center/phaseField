223inline bool
224NucleationParameters::should_attempt_nucleation(unsigned int increment) const

229inline void
230NucleationParameters::postprocess_and_validate(
231 const std::map<unsigned int, VariableAttributes> &var_attributes)
232{
233 // Check if the nucleation period is valid
234 AssertThrow(nucleation_period > 0,
235 dealii::ExcMessage("Nucleation period must be positive."));
236
237 // Check if the postprocessed nucleation rate exists
238 for (const auto &[index, variable] : var_attributes)
239 {
240 if (variable.is_postprocess() && variable.is_nucleation_rate())
241 {
242 pp_nucleation_rate_exists = true;
243 break;
244 }
245 }
246}
