48 void
49 reinit()
50 {
51 // Create the triangulations for the coarser levels if we have at least one instance
52 // of multigrid for any of the fields
53 if (!has_multigrid)
54 {
55 return;
56 }
57
58 Assert(triangulation->n_global_levels() > 1,
59 dealii::ExcMessage(
60 "Multigrid preconditioners require multilevel triangulations"));
61
62 // Check that the initial global refinement matches the maximum adaptive refinement
63 Assert(user_inputs->get_spatial_discretization().get_global_refinement() ==
64 user_inputs->get_spatial_discretization().get_max_refinement(),
65 dealii::ExcMessage(
66 "Currently, we don't allow the initial refinement to be lower than the "
67 "maximum adpative refinement level when using multigrid. This is because we "
68 "have to create a sequence of coarser meshes."));
69
70 coarsened_triangulations =
71 dealii::MGTransferGlobalCoarseningTools::create_geometric_coarsening_sequence(
72 *triangulation);
73
74 // TODO (landinjm): p-multigrid
75 };
