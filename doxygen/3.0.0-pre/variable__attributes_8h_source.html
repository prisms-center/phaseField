147 friend DependencyTypeFlags
148 operator|(const DependencyTypeFlags flag_1, const DependencyTypeFlags flag_2)
149 {
150 return static_cast<DependencyTypeFlags>(static_cast<std::uint8_t>(flag_1) |
151 static_cast<std::uint8_t>(flag_2));
152 }
153
154 friend DependencyTypeFlags &
155 operator|=(DependencyTypeFlags &flag_1, const DependencyTypeFlags flag_2)
156 {
157 flag_1 = flag_1 | flag_2;

163 {
164 return static_cast<DependencyTypeFlags>(static_cast<std::uint8_t>(flag_1) &
165 static_cast<std::uint8_t>(flag_2));
166 }
167
168 friend DependencyTypeFlags &
169 operator&=(DependencyTypeFlags &flag_1, const DependencyTypeFlags flag_2)
170 {
171 flag_1 = flag_1 & flag_2;
172 return flag_1;
173 }

185inline std::string
186to_string(FieldInfo::TensorRank tensor_rank)
187{
188 switch (tensor_rank)
189 {
190 case FieldInfo::TensorRank::Undefined:
191 return "Undefined";
192 case FieldInfo::TensorRank::Scalar:

195 return "Vector";
