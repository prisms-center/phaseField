165 static_cast<std::uint8_t>(flag_2));
166 }
167
168 friend DependencyTypeFlags &

170 {

172 return flag_1;
173 }

185inline std::string
186to_string(FieldInfo::TensorRank tensor_rank)
187{
188 switch (tensor_rank)
189 {
190 case FieldInfo::TensorRank::Undefined:
191 return "Undefined";
192 case FieldInfo::TensorRank::Scalar:
193 return "Scalar";
194 case FieldInfo::TensorRank::Vector:
195 return "Vector";
196 default:
197 return "Unknown TensorRank";
198 }
199}
