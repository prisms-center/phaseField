84template <int dim>
85inline void
86boundaryParameters<dim>::compute_boundary_conditions(const AttributesList &var_attributes)
87{
88 for (const auto &[index, variable] : var_attributes)
89 {
90 // Ensure that boundary conditions are specified for all variables and their
91 // components
92 if (variable.field_type == fieldType::VECTOR)
93 {
94 for (uint i = 0; i < dim; i++)
95 {
96 AssertThrow(!BC_list.at(index).at(i).empty(),
97 dealii::ExcMessage("Boundary conditions must be specified "
98 "for all components in all vector field."));
99 }
100 }
101 else
102 {
103 AssertThrow(!BC_list.at(index).at(0).empty(),
104 dealii::ExcMessage("Boundary conditions must be specified "
105 "for all scalar fields."));
106 }
107 }
108
109 // Helper function that splits boundary condition input (delimiter ",") and assigns
110 // the correct boundary condition enum.
111 auto set_boundary =
112 [&](const std::string &BC_string, const uint &index, const uint &component)
113 {
114 // Split string
115 auto BC_string_list = dealii::Utilities::split_string_list(BC_string);
116
117 // Check that there is either 1 or 2*dim entries in the vector. This can be changed
118 // later to support other geometries.
119 AssertThrow(
120 BC_string_list.size() == 1 || BC_string_list.size() == static_cast<size_t>(2 * dim),
121 dealii::ExcMessage("Either 1 or 2*dim boundary conditions must be specified."));
122
123 // If there is only 1 boundary condition resize BC_string_list, copying the first
124 // entry.
125 if (BC_string_list.size() == 1)
126 {
127 BC_string_list.resize(static_cast<size_t>(2 * dim), BC_string_list[0]);
128 }
129
130 // Assign boundary condition
131 boundaryCondition<dim> condition;
132 for (uint i = 0; i < (2 * dim); i++)
133 {
134 if (boost::iequals(BC_string_list[i], "NATURAL"))
135 {
136 condition.boundary_condition_map.emplace(i, boundaryType::NATURAL);
137 }
138 else if (boost::iequals(BC_string_list[i].substr(0, 9), "DIRICHLET"))
139 {
140 condition.boundary_condition_map.emplace(i, boundaryType::DIRICHLET);
141 std::string dirichlet_value =
142 BC_string_list[i].substr(10, BC_string_list[i].size());
143 dirichlet_value = dealii::Utilities::trim(dirichlet_value);
144 condition.dirichlet_value_map.emplace(i,
145 dealii::Utilities::string_to_double(
146 dirichlet_value));
147 }
148 else if (boost::iequals(BC_string_list[i], "PERIODIC"))
149 {
150 condition.boundary_condition_map.emplace(i, boundaryType::PERIODIC);
151 }
152 else if (boost::iequals(BC_string_list[i].substr(0, 7), "NEUMANN"))
153 {
154 AssertThrow(false, FeatureNotImplemented("Neumann boundary conditions"));
155 }
156 else if (boost::iequals(BC_string_list[i], "NON_UNIFORM_DIRICHLET"))
157 {
158 condition.boundary_condition_map.emplace(i,
159 boundaryType::NON_UNIFORM_DIRICHLET);
160 }
161 else if (boost::iequals(BC_string_list[i], "NON_UNIFORM_NEUMANN"))
162 {
163 AssertThrow(false,
164 FeatureNotImplemented("Nonuniform neumann boundary conditions"));
165 }
166 else
167 {
168 AssertThrow(false,
169 dealii::ExcMessage("Invalid boundary condition " +
170 BC_string_list[i]));
171 }
172 // If periodic boundary conditions are used, ensure that they are applied on
173 // both sides of the domain.
174 if (i % 2 == 0)
175 {
176 AssertThrow(boost::iequals(BC_string_list[i], "PERIODIC") ==
177 boost::iequals(BC_string_list[i + 1], "PERIODIC"),
178 dealii::ExcMessage("Periodic boundary condition must be "
179 "specified on both sides of domain"));
180 }
181 }
182
183 boundary_condition_list[component].emplace(index, condition);
184 };
185 // Process each boundary condition
186 for (const auto &[index, variable] : var_attributes)
187 {
188 if (variable.field_type == fieldType::VECTOR)
189 {
190 for (uint i = 0; i < dim; i++)
191 {
192 set_boundary(BC_list.at(index).at(i), index, i);
193 }
194 }
195 else
196 {
197 set_boundary(BC_list.at(index).at(0), index, 0);
198 }
199 }
200}

202template <int dim>
203inline void
204boundaryParameters<dim>::print_parameter_summary() const
205{
206 conditionalOStreams::pout_summary()
207 << "================================================\n"
208 << "\tBoundary Parameters\n"
209 << "================================================\n";
210
211 conditionalOStreams::pout_summary() << "Index: ";
212 for (const auto &[index, component_map] : boundary_condition_list)
213 {
214 for (const auto &[component, boundary_condition] : component_map)
215 {
216 conditionalOStreams::pout_summary() << index << "\n"
217 << " Component: " << component << "\n";
218 for (const auto &[domain_id, boundary_type] :
219 boundary_condition.boundary_condition_map)
220 {
221 conditionalOStreams::pout_summary()
222 << " Boundary id: " << domain_id << " "
223 << "Condition: " << to_string(boundary_type);
224 if (boundary_type == boundaryType::DIRICHLET)
225 {
226 conditionalOStreams::pout_summary()
227 << " = " << boundary_condition.dirichlet_value_map.at(domain_id);
228 }
229 conditionalOStreams::pout_summary() << "\n";
230 }
231 }
232 }
233
234 if (!pinned_point_list.empty())
235 {
236 conditionalOStreams::pout_summary() << "Pinned field index: ";
237 }
238 for (const auto &[index, point_value_map] : pinned_point_list)
239 {
240 conditionalOStreams::pout_summary()
241 << index << "\n"
242 << " Value: " << point_value_map.first << "\n"
243 << " Point: " << point_value_map.second << "\n";
244 }
245 conditionalOStreams::pout_summary() << "\n" << std::flush;
246}
