109 Assert(variable_attributes.at(index) != variable_attributes.end(),
110 dealii::ExcMessage(
111 "Invalid index. The provided index does not have an entry in the variable "

113
114 if (variable_attributes.at(index).field_type == fieldType::SCALAR)

116 Assert(scalar_needed,

122 "compute_invm() prior to calling the getter function."));
123
124 return invm_scalar;
125 }
126 else if (variable_attributes.at(index).field_type == fieldType::VECTOR)
127 {
128 Assert(vector_needed,
129 dealii::ExcMessage(
130 "The invm for vector fields is marked as not needed. Make sure the "
131 "variable attributes correspond with the provided index."));
132 Assert(invm_vector.size() == 0,
133 dealii::ExcMessage("The vector invm has size 0. Please make sure to call "
134 "compute_invm() prior to calling the getter function."));
135
136 return invm_vector;
137 }
138
139 Assert(false, dealii::ExcMessage("Invalid field type."));
140}
